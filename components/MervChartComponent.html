  removeFromDashboard(){
    this.remove.emit();
  }

  
  getFilters(){
    this.filterService.getAllPublicFilters().subscribe(
      (filters)=>{
        this.model_filters = filters;
        this.original_filters = [...filters];
        this.setFilterOptions();
      }
    )
  }
    
  clearFilter(){
    this.filterControl = new FormControl();
    this.filter_selected = {};
    this.model_filters = [...this.original_filters];
    this.setMervChart();
    this.setFilterOptions();
  }
  
  setFilterOptions(){
    this.modelFilterOptions = this.filterControl.valueChanges
      .pipe(
      startWith(''),
      map(value => this._filterFilters(value,this.model_filters))
    );
    setTimeout(() => {
    }, 1000);
  }

  private _filterFilters(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      return option
    }
  }

  _setMervFilter(filter){
    if(filter){
      this.filter_selected = {...filter};
      this.merv_rows.map(merv_row => {
        merv_row.row_selected = false;
        merv_row.numbers.map(number=>number.selected = false);
        merv_row.small.map(sm=>sm.selected = false);
        merv_row.medium.map(md=>md.selected = false);
        merv_row.large.map(lg=>lg.selected = false);
      })
  
      this.merv_rows.map(merv_row => {
        let number_row = merv_row.numbers.find(row=>row.index == this.filter_selected.merv_rating);
        if (number_row) { number_row.selected = true; merv_row.row_selected = true };
  
        let number_small = merv_row.small.find(row=>row.index == this.filter_selected.merv_rating);
        if (number_small) { number_small.selected = true;merv_row.row_selected = true };
        
        let number_med = merv_row.medium.find(row=>row.index == this.filter_selected.merv_rating);
        if (number_med) {number_med.selected = true;merv_row.row_selected = true};
        
        let number_lg = merv_row.large.find(row=>row.index == this.filter_selected.merv_rating);
        if (number_lg)  {number_lg.selected = true;merv_row.row_selected = true};
        
      })
    }
  }

}
