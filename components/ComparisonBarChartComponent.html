import { Component, OnInit, ViewChild, Input, SimpleChanges } from '@angular/core';
import Chart from 'chart.js';

@Component({
  selector: 'app-comparison-bar-chart',
  templateUrl: './comparison-bar-chart.component.html',
  styleUrls: ['./comparison-bar-chart.component.scss']
})


export class ComparisonBarChartComponent implements OnInit {

  @ViewChild('bar_chart') bar_chart:any;
  @Input('current_cost') current_cost : any;
  @Input('optimized_cost') optimized_cost : any;
  @Input('selected_cost') selected_cost :any;
  @Input('chart_id') chart_id : any;
  @Input('optimized_cost_proposed') optimized_cost_proposed : any;
  @Input('selected_cost_proposed') selected_cost_proposed : any;
  
  public myChart:any;
  private barChart:any;
  labels: string[];
  constructor() { }

  ngOnInit() {}

  ngAfterViewInit(){
    this.labels = [
      this.current_cost.toFixed(2) + ' USD',
      this.optimized_cost.toFixed(2) + ' USD',
      this.selected_cost.toFixed(2) + ' USD',
      this.optimized_cost_proposed.toFixed(2) + ' USD',
      this.selected_cost_proposed.toFixed(2) + ' USD'
    ];
    this.initBarChart();
  }

  ngOnChanges(changes: SimpleChanges){
    if (changes.selected_cost && !changes.selected_cost.firstChange){
      this.labels[2] = this.selected_cost.toFixed(2) + ' USD';
      this.barChart.data.datasets[2].data[2] = this.selected_cost.toFixed(2);
      this.barChart.update();
    }
    if (changes.selected_cost_proposed && !changes.selected_cost_proposed.firstChange){
      this.labels[4] = this.selected_cost_proposed.toFixed(2) + ' USD';
      this.barChart.data.datasets[4].data[4] = this.selected_cost_proposed.toFixed(2);
      this.barChart.update();
    }
  }

  initBarChart(){
    var id = this.chart_id;
    var canvas : any = document.getElementById(id);
    var ctx2 = canvas.getContext("2d");
    var data =  {
      labels:this.labels,
      datasets: [
        {
          barPercentage: 0.5,
          label: 'Current',
          data: [this.current_cost.toFixed(2),0,0,0,0],
          backgroundColor: '#003A70',
          borderColor:'#003A70',
          borderWidth:1,
        },
        {
          barPercentage: 0.5,
          label: 'Current Opt',
          data: [0, this.optimized_cost.toFixed(2), 0,0,0],
          backgroundColor: '#199AD8',
          borderColor:'#199AD8',
          borderWidth:1,
        },
        {
          barPercentage: 0.5,
          label: 'Current Sel.',
          data: [0, 0, this.selected_cost.toFixed(2) ,0,0],
          backgroundColor: '#c7d42d',
          borderColor:'#c7d42d',
          borderWidth:1,
        },
        {
          barPercentage: 0.5,
          label: 'Prop Opt',
          data: [0, 0,0, this.optimized_cost_proposed.toFixed(2),0],
          backgroundColor: '#6B20B7',
          borderColor:'#6B20B7',
          borderWidth:1,
        },
        {
          barPercentage: 0.5,
          label: 'Prop Sel.',
          data: [0, 0, 0, 0, this.selected_cost_proposed.toFixed(2)],
          backgroundColor: '#BD2AD3',
          borderColor:'#BD2AD3',
          borderWidth:1,
        },
      ],
    };
    var options = {
      title: {
        display: false,
        text: ''
      },
      legend: {
        position: 'top',
        display:true,
        align:'center',
        fullWidth:true,
        labels:{
          boxWidth:40,
          marginLeft:50,
          padding:20,
          fontFamily:'lato',
        }
      },
      scales: {
        xAxes: [{
          stacked: true,
          ticks: {
              fontFamily:'lato',
              display: true,
              fontSize:14,
          }
        }],
        yAxes: [{
            stacked: true,
            display: true,
            ticks: {
                beginAtZero: true,
                //stepSize:200,
                fontFamily:'lato',
                fontSize:10,
                callback: function(value, index, values) {

                  var parts = parseFloat(value.toFixed(2)).toString().split(".");
                  parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                  return "$ " + parts[0];
                }
            }
        }]
      },
      tooltips:{
        callbacks:{
          label: function(tooltipItem, data) {
            var label = data.datasets[tooltipItem.datasetIndex].label || '';
            var data = data.datasets[tooltipItem.datasetIndex];
            var value = data.data[tooltipItem.index];
            
            var parts = parseFloat(tooltipItem.yLabel.toFixed(2)).toString().split(".");
            parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
            var final_value = (parts.join("."))

            return label + ": "+  final_value  + " $";
          }
        },
      },
    }
    this.barChart = new Chart(ctx2,{
      type: 'bar',
      data: data,
      options: options
    });
  }

}
