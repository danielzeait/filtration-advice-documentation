import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { time } from 'console';
import { CaseProvider } from 'src/app/providers/case.service';
import { CommonProvider } from 'src/app/providers/common.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-case-time-line',
  templateUrl: './case-time-line.component.html',
  styleUrls: ['./case-time-line.component.scss']
})
export class CaseTimeLineComponent implements OnInit {
  @Output() consult = new EventEmitter<any>();
  @Input('case_id') case_id :any;
  @Input('optimal_date') optimal_date:any;
  public dates:any=[];
  public years:any=[];
  public current_year:any;
  public months:any=[];
  public consult_date:any;
  public grid_days:any = [];
  public display_date:any;
  public show_optimal_date: any;
  public show_card:boolean = false;
  public weeks:any = Array(53);

  constructor(private caseService:CaseProvider, private navigator:NavigationProvider, private common:CommonProvider) { }

  ngOnInit(): void {
    this.initMonths();
    this.initYears();
    this.getSnapshots();
    this.common.updateSnapshots.subscribe(()=>{
      this.getSnapshots();
    })
  }
  
  getSnapshots(){
    this.caseService.getCaseSnapshots(this.case_id).subscribe((response:any)=>{
      this.dates = [...response.historic];
      this.initGrid(new Date().getFullYear());
    })
  }
  
  initMonths(){
    this.months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  }
  
  initYears(){
    this.years = [2019,2020,2021,2022,2023,2024];
    var current_year = this.years.find(y => y == new Date().getFullYear());
    if (current_year) {
      this.current_year = current_year;
    };
  }
  
  initGrid(year){
    let init_date = new Date('01 01 '  + year);
    let last_date = new Date('12 31 ' + year).getDay();
    var original_timestamp = init_date.getTime();
    var new_day = 60 * 60 * 24 * 1000;
    var real_day = 1;
    var first_day = init_date.getDay();
    this.grid_days = [];

    var final_date =  new Date(((new Date().getMonth())+1) + ' ' + new Date().getDate() + ' ' + new Date().getFullYear()).getTime();
    
    var date_opt =  ((new Date(this.optimal_date).getMonth())+1) + ' ' + 
    new Date(this.optimal_date).getDate() + ' ' + new Date(this.optimal_date).getFullYear();
    
    var final_date_opt = new Date(date_opt).getTime();
    
    this.grid_days.push({
      'index':0,
      'first_column':false,
      'real_day':1,
      'is_optimal':(original_timestamp - (new_day * first_day)) == final_date_opt,
      'timestamp':original_timestamp - (new_day * first_day),
      'today':(original_timestamp - (new_day * first_day)) == final_date,
      'month':1,
      'hide':first_day !== 0,
      'active':false,
      'display_card':false
    });

    for(var i = 1; i < 371; i++){
      if(i !== 0){
        if((i % 53 == 0)){
          real_day = (i/53) + 1;
          if(i > 52){
            original_timestamp = init_date.getTime() + (new_day * (i/53));
          }else{
            original_timestamp = init_date.getTime() + new_day;
          }
        }else{
          real_day = real_day + 7;
          original_timestamp = original_timestamp + (new_day * 7);
        }
      }
    
      this.grid_days.push({
        'index':i,
        'first_column':i / 53 == first_day,
        'real_day':real_day - first_day,
        'timestamp':original_timestamp - (new_day * first_day),
        'is_optimal':(original_timestamp - (new_day * first_day)) == final_date_opt,
        'month':new Date(original_timestamp - (new_day * first_day)).getMonth() + 1,
        'today':(original_timestamp - (new_day * first_day)) == final_date,
        'hide':((i % 53 == 0) && (i / 53 < first_day)),
        'last_days':((i+1) % 53 == 0) && ((i+1) / 53 > (last_date + 1)),
        'active':false,
        'display_card':false
      });
      if(i == (365 + first_day)){
        this.highlightDates();
      }
    }
  }

  highlightDates(){
    this.dates.forEach(date => {
      this.grid_days.forEach((day) => {
        if(this.grid_days[day.index + 1]){
          if ((day.timestamp / 1000 < date.timestamp || day.timestamp / 1000 == date.timestamp) && (date.timestamp < ((day.timestamp / 1000) + 86400))){
            day.backend_value = date.timestamp;
            day.warning = date.has_warnings;
            day.energy_consumed_so_far = date.energy_consumed_so_far;
            day.op_hours_so_far = date.op_hours_so_far; 
            day.active = true;
          }
        }
      });
    });
  }

  handleHover(day){
    if(day.active){
      this.grid_days.map(d => d == day ? d.display_card = true : d.display_card = false)
      this.display_date = day.backend_value * 1000; 
    }
  }

  consultSnapshot(date){
    if(date.is_optimal){
      this.show_optimal_date = true;
    }else{
      this.consult.emit(date);
      this.show_optimal_date = false;
    }
  }

  changeYear(){
    this.initGrid(this.current_year);
  }

}
