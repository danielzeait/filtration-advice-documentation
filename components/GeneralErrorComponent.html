import { Component, OnInit, Inject } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { AuthProvider } from 'src/app/providers/auth.service';

@Component({
  selector: 'app-general-error',
  templateUrl: './general-error.component.html',
  styleUrls: ['./general-error.component.scss']
})
export class GeneralErrorComponent implements OnInit {

  public errors:any=[];

  constructor(
    public dialogRef: MatDialogRef<GeneralErrorComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private navigator:NavigationProvider,
    private authProvider:AuthProvider,
  ) { }

  ngOnInit() {
    if(!this.data.error.error.detail && this.data.error.status !== 524){
      for (const [key, value] of Object.entries(this.data.error.error)) {
        this.errors.push(value);
      }
    }
  }

  _cancel(){
    this.dialogRef.close();
  }

  _exitAndLog(){
    this.authProvider.destroySession();
    this.dialogRef.close();
  }

}
