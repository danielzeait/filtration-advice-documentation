import { Component, OnInit, Input, Inject, ViewChild } from '@angular/core';
import { SharedReportProvider } from 'src/app/providers/shared-reports.service';
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { global } from 'src/app/global';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { System } from 'src/app/models/system.model';
import { Observable } from 'rxjs';
import { map, share } from 'rxjs/operators';
import { FormControl , Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { ActivatedRoute } from '@angular/router';
import { NotificationService } from 'src/app/providers/notifications.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { CommonProvider } from 'src/app/providers/common.service';



@Component({
  selector: 'app-shared-reports',
  templateUrl: './shared-reports.component.html',
  styleUrls: ['./shared-reports.component.scss']
})
export class SharedReportsComponent implements OnInit {
  @Input('data') reports: any;
  @ViewChild('systemForm', { static: true }) systemForm;
  public server_name: string = global.SERVER_NAME;
  public innerWidth: any;
  public system_shared: any = {};
  public email = new FormControl('', [Validators.required]);



  constructor(
    private errorHandler:CustomErrorHandler,
    private commonProvider:CommonProvider,
    public dialogRef: MatDialogRef<SharedReportsComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private navigator: NavigationProvider,
    private responsiveProvider: ResponsiveProvider,
    private routeActivated: ActivatedRoute,
    private notificationProvider: NotificationService,
    private sharedProvider: SharedReportProvider,
    public utils: UtilsFunctions,
    ) { }

  ngOnInit() {

  }

  cancel() {
    this.dialogRef.close(false);
  }

  getErrorMessage() {
    return this.email.hasError('required') ? 'You must enter a value' :
        this.email.hasError('email') ? 'Not a valid email' :
            '';
  }

  shared(data) {
    var email_list = this.system_shared.email.split(' ');
    email_list = email_list.filter(email => email !== '' && email !== ' ');
    
    const sharedData = {
      id: this.data.id,
      emails: email_list.join(" "),
      count:email_list.length
    }
    this.sharedProvider.createNewSharedReport(sharedData).subscribe((response)=>{
      this.commonProvider.showSnackBar('Report shared.','Your report has been shared with '+ sharedData.count + ' users' ,'success-snackbar');
      this.dialogRef.close(true);
    },error=>{
      this.errorHandler.handleError(error);
      this.commonProvider.showSnackBar('Shared failed.','The report has not been shared with the user(s).','alert-snackbar');
    })

  }



}
