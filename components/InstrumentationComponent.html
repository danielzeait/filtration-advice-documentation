import { Component, OnInit, Output, EventEmitter, Input, SimpleChanges } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { InstrumentProvider } from 'src/app/providers/instrument.service';
import { ChangeProviderDialogComponent } from '../modal-dialogs/change-provider-dialog/change-provider-dialog/change-provider-dialog.component';

@Component({
  selector: 'app-instrumentation',
  templateUrl: './instrumentation.component.html',
  styleUrls: ['./instrumentation.component.scss']
})
export class InstrumentationComponent implements OnInit {
  @Input('skip_preview') skip_preview:boolean;
  @Input('data') input_data:any;
  @Input('date') input_date:any;
  @Input('review') review:any;
  @Input('instruments') instruments:any;
  @Output() instrument_set = new EventEmitter();
  @Output() back = new EventEmitter();
  @Output() skip = new EventEmitter();
  public instrument_cards:any=[];
  public instrument_data:any={
    duct_shape:'rectangle',
  }
  public new_instrumentation:boolean=true;
  public slots:any=[];
  public instrument_selector:any=false;
  public instrument_type: any;
  public probes:any;
  public view_secret_input:boolean = false;
  public exist_provider:boolean = false;
  public try_secret:number = 0;
  public instrumentation_data: any = {
    drop_pressure_inst:null,
    flow_inst:null,
    duct_shape:'rectangle',
    filterless_dp: null,
    airflow_without_filter:null,
    nominal_max_airflow:1,
    airflow_instrument_coefficient:0.0000027,
    is_circular_pipe:false,
    duct_diameter:1,
    duct_height:24,
    duct_width:24,
    probe:null,
    installation_date:'',
  };
  constructor(private instrumentProvider:InstrumentProvider,public dialog: MatDialog) { }

  ngOnInit() {
    this.instrument_cards = [
      {
        title:'Air Velocity',
        image:' ../../../assets/imgs/air_velocity.svg'
      },
      {
        title:'Pressure Drop',
        image:' ../../../assets/imgs/pressure_drop.svg'
      },
      {
        title:'Temperature',
        image:' ../../../assets/imgs/temperature.svg'
      },
      {
        title:'Humidity',
        image:' ../../../assets/imgs/humidity.svg'
      }
    ];
    if(this.input_data){
      this.new_instrumentation = true;
      this.instrumentation_data = {...this.input_data}
    }
    if(this.instruments){
      this.initSlots();
    }
    if(this.input_date){
      this.instrumentation_data.installation_date = new Date(this.input_date);
    }
    if(this.skip_preview){
      this.new_instrumentation = true;
    }
    this.getProbes();
  }

  initSlots(){
    this.slots[0] = {...this.instruments.flow_inst};
    this.slots[1] = {...this.instruments.drop_pressure_inst};
  }

  _setDuctShape(){

  }

  getProbes(){
    this.instrumentProvider.getProbes().subscribe(
      (probes:any)=>{
        this.probes = [...probes];
    })
  }

  seletcInstrument(type){
    this.instrument_type = type;
    this.instrument_selector = true;
  }
  
  ngOnChanges(changes:SimpleChanges){
    if(changes.input_date && changes.input_date.currentValue){
      this.instrumentation_data.installation_date = new Date(this.input_date);
    }
    if(changes.instruments && changes.instruments.currentValue ){
      
      this.instruments = {...changes.instruments.currentValue};
      // this.slots[0] = {...this.instruments.flow_inst};
      // this.slots[0].provider = {
      //   name:'Hobo Link',
      //   image:'../../../assets/imgs/hobo-data-loggers.png'
      // }
      // this.slots[1] = {...this.instruments.drop_pressure_inst};
      // this.slots[1].provider = {
      //   name:'Hobo Link',
      //   image:'../../../assets/imgs/hobo-data-loggers.png'
      // }
      this.instrumentation_data.is_circular_pipe ? this.instrumentation_data.duct_shape = 'circle' : this.instrumentation_data.duct_shape = 'rectangle' 
    }
  }

  setInstrument(ev){
    this.instrument_selector = false;
    if(this.instrument_type == 'air_velocity'){
      this.slots[0] = ev;
    }
    if(this.instrument_type == 'pressure_drop'){
      this.slots[1] = ev;
    }
    if(this.instrument_type == 'temperature'){
      this.slots[2] = ev;
    }
    if(this.instrument_type == 'humidity'){
      this.slots[3] = ev;
    }
  }

  setInstrumentationData(){
    this.instrumentation_data.duct_shape == 'round' ? this.instrumentation_data.is_circular_pipe = true : this.instrumentation_data.is_circular_pipe = false;
    this.slots[0] ? this.instrumentation_data.flow_inst = this.slots[0].id : '';
    this.slots[1] ? this.instrumentation_data.drop_pressure_inst = this.slots[1].id : '';
    let emit_data = {
      instruments:{...this.slots},
      data:this.instrumentation_data,
    }
    this.instrument_set.emit(emit_data);
  }

  skipStep(){
    this.skip.emit()
  }

  backStep(){
    this.back.emit();
  }

  tryOpenSecretPanel(){
    if(this.try_secret < 3){
      this.try_secret +=1; 
    }
  }

  clearSlots(slot){
    if(slot){
      this.slots[slot]=null;
      if(!this.slots[0] && !this.slots[1]){
        this.slots = [];
        localStorage.removeItem('provider_selected');
      }
    }else{
      const dialogRef = this.dialog.open(ChangeProviderDialogComponent, {
        width: '50%',
        height: 'auto',
      });
  
      dialogRef.afterClosed().subscribe((result)=>{
        if(result){
          this.slots = [];
          localStorage.removeItem('provider_selected');
        }
      })
    }
  }

  openSecretPanel(){
    this.try_secret +=1;
    if(this.try_secret == 7){
      this.view_secret_input = true;
      this.try_secret = 0;
    }
  }

  consultProviders(){
    var provider = localStorage.getItem('provider_selected');
    if(provider){
      this.exist_provider = true;
    }else{
      this.exist_provider = false;
    }
  }

}
