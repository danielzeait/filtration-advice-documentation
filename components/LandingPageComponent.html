import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { SuccessDialogComponent } from 'src/app/components/modal-dialogs/success-dialog/success-dialog.component';
import { AuthProvider } from 'src/app/providers/auth.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { MatDialog } from '@angular/material/dialog';

@Component({
  selector: 'app-landing-page',
  templateUrl: './landing-page.component.html',
  styleUrls: ['./landing-page.component.scss']
})

export class LandingPageComponent implements OnInit {

  @ViewChild('videoPlayer',{static:true}) videoplayer: ElementRef;
  prices:any = [];
  current_price:any;
  transition_class: string;
  transition_in:string = 'slideInRight';
  constructor(
    private authProvider:AuthProvider,
    private handlerError:CustomErrorHandler, 
    private navigation:NavigationProvider,
    private navigator:NavigationProvider,
    public dialog:MatDialog,
  ) { }

  ngOnInit() {
    this._checkForConfirmEmailToken();
    this.setPrices();
  }

  setPrices(){
    this.prices = [
      {
        index:0,
        title:'MANUAL CASES',
        content:'The pressure drop, air flow, running time and other information of the filtration system is  manually loaded  in to the software',
        prices:[
          '$6.25 or 10 cfm per Manual Optimization (FREE during Beta)',
          '$12.50 or 20 cfm per Manual Comparison'
        ],
        img:'../../../../assets/imgs/manual_cases.png',
        is_active:true,
      },
      {
        index:1,
        title:'MONITORED CASES ',
        content:'By on site monitoring and automatically transmitting the data to the cloud, this software continuosly updates and informs the user the current total cost of ownership and the data for the optimal filter change out',
        prices:[],
        img:'../../../../assets/imgs/monitored_cases.png',
        is_active:false,
      },
      {
        index:2,
        title:'PRIVATE LIBRARY OF TESTS REPORTS',
        content:'If you want to use filter test reports that you consider confidential, we can set up a private library for you. Only you decide with whom you want to share the test reports',
        action:'Contact us',
        img:'../../../../assets/imgs/landing/private_library.png',
        is_active:false,
      },
    ]
    this.current_price = this.prices[0];
  }

  changeCurrenPrice(move){
    if(move == 'next'){
        this.transition_class = "slideOutLeft";
        setTimeout(() => {
          let index = this.current_price.index == this.prices.length - 1 ? 0 : this.current_price.index + 1;
          this.current_price = this.prices[index];
          this.transition_class = "slideInRight";
        }, 10);
      }
      if(move == 'prev'){
        this.transition_class = "slideOutLeft";
        setTimeout(() => {
          let index = this.current_price.index == 0 ? this.prices.length - 1 : this.current_price.index - 1;
          this.current_price = this.prices[index];
          this.transition_class = "";
          this.transition_in = "slideInLeft";
        }, 10);
      }
  }

  _checkForConfirmEmailToken(){
    let confirm_token = localStorage.getItem('confirm_email_token');
    if(confirm_token){
      this._confirmUserEmail(confirm_token);
    }
  }

  _confirmUserEmail(confirm_token){
    this.authProvider.validateUserEmail(confirm_token).subscribe((response)=>{
      localStorage.setItem('confirmed_user', response.email_was_confirm);
      localStorage.removeItem('confirm_email_token');
      let dialog_data = {
        title:'WELCOME TO FILTRATION ADVICE!',
        message:'Your email account has been succesfully confirmed',
        action:'Continue'
      }
      const dialogRef = this.dialog.open(SuccessDialogComponent, {
        width: '50%',
        height: 'auto',
        data: dialog_data,
        id:'confirmed'
      });

      dialogRef.afterClosed().subscribe((data)=>{
        this.navigator._goHome();
      })
    },error=>{
      localStorage.removeItem('confirm_email_token');
      this.handlerError.handleError(error);
    })
  }

  _goLogin(){
    this.navigation._goLogin();
  }

  _goRegister(){
    this.navigation._goRegister();
  }

  _goLoginSonicu(){
    this.navigation._goLoginSonicu();
  }

  _goContact(){
    this.navigation._goContact();
  }

}
