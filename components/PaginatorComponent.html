import { HostListener } from '@angular/core';
import { Component, OnInit, Input, Output, EventEmitter, ViewChild, SimpleChanges } from '@angular/core';
import { PageEvent, MatPaginator } from '@angular/material/paginator';

@Component({
  selector: 'app-paginator',
  templateUrl: './paginator.component.html',
  styleUrls: ['./paginator.component.scss']
})
export class PaginatorComponent implements OnInit {
  @ViewChild(MatPaginator) paginator: MatPaginator;
  @Input("length") length:any;
  @Input("pageSize") pageSize:any;
  @Input("pageSizeOptions") pageSizeOptions :any;
  @Input("jumpToPageOption") jumpToPageOption :any;
  @Input("url") list_url :any;
  @Output() jump_page  = new EventEmitter<any>();
  @Output() jump_manually  = new EventEmitter<any>();
  public hide_page_size:boolean=false;
  public pageEvent: PageEvent;
  constructor() { }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.hidePageOption();
  }
  

  ngOnInit() {
    this.hidePageOption();
  }

  hidePageOption(){
    window.innerWidth < 1140 ? this.hide_page_size = true : this.hide_page_size = false;
  }

  handlePage(event){
    var page:any = 0;
    var url = this.list_url;
    if(event.pageSize !== this.pageSize){
      this.pageSize = event.pageSize;
      url = url  + '&&page_size=' + event.pageSize + '&page=1';
      event.pageIndex = 0;
      event.previousPageIndex = 0;
      this.paginator.pageIndex = 0;
    }else{
      if(event.pageIndex > 0) {
        page = event.pageIndex + 1;
      }else{
        page = 1;
      }
      url = url  + '&&page_size=' + event.pageSize + '&page=' + page;
    }
    this.pageEvent = event;
    let emitData={
      url:url,
      event:event
    }
    this.jump_page.emit(emitData);
  }

  ngOnChanges(changes:SimpleChanges){
    if(changes.jumpToPageOption && !changes.jumpToPageOption.isFirstChange()){
      var url = this.list_url;
      var page_to_jump = parseInt(changes.jumpToPageOption.currentValue.page + 1);
      url = url  + '&&page_size=' + this.pageSize + '&page=' + page_to_jump;
      this.paginator.pageIndex = changes.jumpToPageOption.currentValue.page; 
      let emitData={
        url:url,
        event:changes.jumpToPageOption.currentValue.data
      }
      this.jump_page.emit(emitData);
    }
  }

}
