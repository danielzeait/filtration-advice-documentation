import { Component, OnInit, Inject, ViewChild, ViewChildren, QueryList, ChangeDetectorRef } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { PaymentProvider } from 'src/app/providers/payment.service';
import { MatSelectionList, MatSelectionListChange, MatListOption, MatExpansionPanel, MatSnackBar } from '@angular/material';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { NotificationService } from 'src/app/providers/notifications.service';
import { ComparisonProvider } from 'src/app/providers/comparison.service';
import { AuthProvider } from 'src/app/providers/auth.service';
import { CaseProvider } from 'src/app/providers/case.service';
import { CommonProvider } from 'src/app/providers/common.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { PairedCaseInstrumentProvider } from 'src/app/providers/paired-case-instrument.service';
import { ReportsProvider } from 'src/app/providers/reports.service';
@Component({
  selector: 'app-payment-dialog',
  templateUrl: './payment-dialog.component.html',
  styleUrls: ['./payment-dialog.component.scss']
})
export class PaymentDialogComponent implements OnInit {
  @ViewChild(MatSelectionList) payments: MatSelectionList;
  @ViewChildren(MatExpansionPanel) viewPanels: QueryList<MatExpansionPanel>;
  public payment_choice:any = 'credits';
  public payment_object:any;
  public payment_methods:any;
  public payment_selected: any;
  public show_card_selected:any;
  public panelOpenState:boolean=false;
  public new_card:boolean = false;
  public set_default: boolean;
  public proposed_filter:any;
  public user_credits:any;
  public current_filter:any;
  public optimization:any;
  public report:any;
  public report_filter:any;
  instrument_data: any;
  constructor(
    private cdr: ChangeDetectorRef,
    private _snackBar: MatSnackBar,
    private authProvider:AuthProvider,
    private navigation:NavigationProvider,
    private notification:NotificationService,
    private navigator:NavigationProvider,
    private caseProvider:CaseProvider,
    private utils:UtilsFunctions,
    private reportService:ReportsProvider,
    private pairCaseInstrumentProvider:PairedCaseInstrumentProvider,
    private commonProvider:CommonProvider,
    public comparisonProvider:ComparisonProvider,
    private handlerError:CustomErrorHandler,
    private paymentProvider:PaymentProvider) { }

  ngOnInit() {
    this.getPaymentMethods();
    this.getUserCredits();
    this.payment_object = JSON.parse(localStorage.getItem('subscription'));
    this.proposed_filter = JSON.parse(localStorage.getItem('proposed_filter'));
    this.current_filter = JSON.parse(localStorage.getItem('case_to_compare'));
    this.optimization = JSON.parse(localStorage.getItem('pending_optimization'));
    this.report = JSON.parse(localStorage.getItem('report_pending'));
    this.report_filter = JSON.parse(localStorage.getItem('report_filter'));
    let instrument_data = localStorage.getItem('pending_instrumentation');
    if(instrument_data != 'undefined'){
      this.instrument_data = JSON.parse(instrument_data);
    }
    if(this.payment_object.slug == "unlimited_instrumented_system" || this.payment_object.slug == "report"){
      this.payment_choice = 'card';
    }
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  ngOnDestroy(){
    localStorage.removeItem('pending_optimization');
    localStorage.removeItem('pending_instrumentation');
    //localStorage.removeItem('proposed_filter');
    //localStorage.removeItem('case_to_compare');
    localStorage.removeItem('report_pending');
    localStorage.removeItem('report_filter');
  }

  getUserCredits(){
    this.authProvider.getUserCredits().subscribe((credits)=>{
      this.user_credits = {...credits};
    })
  }

  newCardCreated(){
    this.new_card = false;
    this.getPaymentMethods();
  }

  setPaymentChoice(type){
    this.payment_choice = type;
  }

  getPaymentMethods(){
    this.paymentProvider.getPaymentMethods().subscribe((response)=>{
      this.payment_methods = response;
      if(this.payment_methods.length > 0){
        this.payment_selected = this.getDefaultPayment();
        this.show_card_selected = true;
      }else{
        this.payment_selected = null;
        this.new_card = true;
      }
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  selectionChange(change: MatSelectionListChange){
    this.payment_selected = change.option.value;  
    this.payments.deselectAll();
    this.viewPanels.forEach(p => p.close());
    change.option.selected = true;
  }

  _cancel(){
    this.new_card = false;
  }

  _goBack(){
    if(this.payment_object.action == 'plans'){
      this._backToPlans(3);
    }
    if(this.payment_object.action == 'comparison'){
      this._backToComparisons();
    }
    if(this.payment_object.action == 'comparison_edited'){
      this._backToComparisons();
    }
    if(this.payment_object.action == 'optimization'){
      this._backToOptimization();
    }
    if(this.payment_object.action == 'update-optimization'){
      this._backToOptimization();
    }
    if(this.payment_object.action == 'report'){
      this._backToFilter();
    }
    localStorage.removeItem('subscription');
  }

  _backToComparisons(){
    this.navigation._goNewComparisonClear();
  }

  _backToCaseEdition(){
    this.caseProvider.getCaseDetails(this.optimization.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goNewCase(case_details);
      }
    )
  }

  _backToOptimization(){
    this.navigation._goNewCase(null);
  }

  _backToFilter(){
    this.navigation._goFilterListSelecect(this.report_filter);
  }

  _backToPlans(section){
    localStorage.setItem('profile_section',section);
    this.navigation._goProfile();
  }

  checkOutPayment(){
    if(this.payment_object.action == 'plans'){
      this.payPlan();
    }
    if(this.payment_object.action == 'comparison'){
      this.payComparison();
    }
    if(this.payment_object.action == 'comparison_edited'){
      this.updateCompare();
    }
    if(this.payment_object.action == 'optimization'){
      this.payOptimization();
    }
    if(this.payment_object.action == 'update-optimization'){
      this.updateOptimization();
    }
    if(this.payment_object.action == 'report'){
      this.payReport();
    }
  }

  payPlan(){
    var data = {
      plan:this.payment_object.id
    }
    this.paymentProvider.createNewSubscription(data).subscribe((response)=>{
      localStorage.removeItem('subscription');
      this.commonProvider.showSnackBar('Payment created.','Your new payment method has been created correctly','success-snackbar');     
      this._backToPlans(2);
    },error=>{
      this.handlerError.handleError(error)
    })
  }

  payReport(){
    this.reportService.buyTestReport(this.report.id,this.payment_selected.id).subscribe((response)=>{
      this.commonProvider.showSnackBar('You have succesfully bought a report.','Your ' + this.report_filter.name + ' test report N° ' + this.report.number + ' has been added to your account. You can download it from the filter action menu','success-snackbar');
      localStorage.removeItem('subscription');
      this._backToFilter();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  getDefaultPayment(){
    this.set_default = false;
    return this.payment_methods.find(payment => payment.is_default);
  }

  _setDefaultPaymentMethod(){
    if(this.set_default){
      this.paymentProvider.setDefaultPaymentMethod(this.payment_selected).subscribe(()=>{
        this.getPaymentMethods();
      },error=>{
        this.handlerError.handleError(error);
      })
    }
  }

  payComparison(){
    let comparison_data = JSON.parse(localStorage.getItem('pending_comparison'));
    comparison_data.payment_id = this.payment_selected ? this.payment_selected.id : null;
    this.payment_choice == 'credits' ? comparison_data.proposed_filter.is_paid_with_credits = true : comparison_data.proposed_filter.is_paid_with_credits = false;
    this.comparisonProvider.createNewComparison(comparison_data).subscribe((response)=>{
      if (this.payment_choice == 'credits') this.commonProvider.refreshCredits();
      localStorage.removeItem('pending_comparison');
      localStorage.removeItem('subscription');
      this.commonProvider.showSnackBar('Comparison created.','Your comparison has been created correctly','success-snackbar'); 
      this.goComparisonResults(response);
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  payOptimization(){
    this.payment_choice == 'credits' ? this.optimization.current_filter.is_paid_with_credits = true : this.optimization.current_filter.is_paid_with_credits = false;
    this.optimization.payment_id = this.payment_selected ? this.payment_selected.id : null;
    if(this.instrument_data && this.instrument_data.drop_pressure_inst !== null && this.instrument_data.flow_inst !== null){
      this.instrument_data.installation_date = this.utils.formatDateISO(this.optimization.installation_date);
      this.optimization.monitoring_module = this.instrument_data;
    }
    this.caseProvider.createNewCase(this.optimization).subscribe(
      (response) => {
        this.commonProvider.showSnackBar('New case created','Your new case has been created correctly','success-snackbar');
        if (this.payment_choice == 'credits') this.commonProvider.refreshCredits();
        this.completeNavigationCaseDone(response);
      }, error => {
        this.handlerError.handleError(error);
      });
  }

  updateOptimization(){
    this.payment_choice == 'credits' ? this.optimization.current_filter.is_paid_with_credits = true : this.optimization.current_filter.is_paid_with_credits = false;
    this.optimization.payment_id = this.payment_selected ? this.payment_selected.id : null;
    if(this.instrument_data && this.instrument_data.drop_pressure_inst !== null && this.instrument_data.flow_inst !== null){
      this.instrument_data.installation_date = this.utils.formatDateISO(this.optimization.installation_date);
      this.optimization.monitoring_module = this.instrument_data;
    }
    this.caseProvider.payAndUpdateCase(this.optimization).subscribe(
      (response) => {
        this.commonProvider.showSnackBar('Case updated','Your case has been saved correctly','success-snackbar');
        if (this.payment_choice == 'credits') this.commonProvider.refreshCredits();
        this.completeNavigationCaseDone(response);
      }, error => {
        this.handlerError.handleError(error);
      });
  }

  // Once the optimization is done might go to result, comparison or case list
  completeNavigationCaseDone(optimization_data) {
    this.caseProvider.getCase(optimization_data.alter_autoincrement_id).subscribe((case_data) => {
      localStorage.removeItem('new_case_data');
      localStorage.removeItem('filter_for_case');
      localStorage.removeItem('reports_for_case');  
      localStorage.removeItem('pending_optimization');
      localStorage.removeItem('pending_instrumentation');
      localStorage.removeItem('subscription');
      this.navigator._goOptimizationsResults(case_data);
    })
  }

  pairInstruments(case_data){
    this.instrument_data.installation_date = this.utils.formatDateISO(this.instrument_data.installation_date);
    this.pairCaseInstrumentProvider.createNewPaired(this.instrument_data,case_data.alter_autoincrement_id).subscribe((response)=>{
      this.completeNavigationCaseDone(case_data);
      this.commonProvider.showSnackBar('Instrument paired','The instrument was paired with your case correctly.','success-snackbar');
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  updateCompare(){
    let proposed_filter = JSON.parse(localStorage.getItem('proposed_filter'));
    let original_case = JSON.parse(localStorage.getItem('case_to_compare'));
    let model_filter = proposed_filter.model_filter;
    proposed_filter.annual_cost = proposed_filter.annual_costs;
    proposed_filter.model_filter = proposed_filter.model_filter.id;
    proposed_filter.payment_id = this.payment_selected ? this.payment_selected.id : null;
    this.payment_choice == 'credits' ? proposed_filter.is_paid_with_credits = true : proposed_filter.is_paid_with_credits = false;
    this.comparisonProvider.updateComparison(proposed_filter).subscribe((response)=>{
      if (this.payment_choice == 'credits') this.commonProvider.refreshCredits();
      this.commonProvider.showSnackBar('Comparison updated','Comparison updated correctly.','success-snackbar');
      let comparing_data = response;
      comparing_data.filter_extra_cost = proposed_filter.filter_extra_cost;
      comparing_data.model_filter = model_filter;
      comparing_data.reports = proposed_filter.reports;
      let data = {
        case:original_case,
        comparison_data:comparing_data
      }
       this.navigator._goComparisonResults(data);
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  goComparisonResults(comparing_data){
    let proposed_filter = JSON.parse(localStorage.getItem('proposed_filter'));
    let report = JSON.parse(localStorage.getItem('comparison_reports'));
    let comparison = comparing_data;
    comparison.filter_extra_cost = proposed_filter.filter_extra_cost;
    comparison.model_filter = proposed_filter;
    comparison.reports = report;
    let data = {
      case:JSON.parse(localStorage.getItem('case_to_compare')),
      comparison_data:comparison
    }
     this.navigator._goComparisonResults(data);
  }

}
