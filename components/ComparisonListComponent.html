import { Component, OnInit, AfterViewInit, ChangeDetectorRef } from '@angular/core';
import { CaseProvider } from 'src/app/providers/case.service';
import { ComparisonProvider } from 'src/app/providers/comparison.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { ActivatedRoute, Router } from '@angular/router';
import { FormControl } from '@angular/forms';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { global } from '../../../global'
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { ConfirmDialogComponent } from 'src/app/components/modal-dialogs/confirm-dialog/confirm-dialog.component';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material';
import { CommonProvider } from 'src/app/providers/common.service';
import { PaymentProvider } from 'src/app/providers/payment.service';
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';

@Component({
  selector: 'app-comparison-list',
  templateUrl: './comparison-list.component.html',
  styleUrls: ['./comparison-list.component.scss']
})
export class ComparisonListComponent implements AfterViewInit {
  public searchedWordsControl = new FormControl('');
  public searchedWordsComparison = new FormControl('');
  public new_comparison:boolean=true;
  public edit:boolean=false;
  public study_cases:any=[];
  public search: string = "";
  public search_comparison:string = "";
  public selected_case:any;
  public length:number;
  public pageSize = 12;
  public hoverInstrument:any=[];
  public pageSizeOptions: number[] = [5,10,12,15,20];
  public jumpToPageOption:any;
  public isImgLoaded:any;
  public toggleButtonOptions:any = [];
  public product:any;
  public list_url:any;
  public bottom_paginator_data:any = {
    previousPageIndex: 0,
    pageIndex: 0,
    pageSize: 12,
    length: 0
  }
  case_to_compare: any;
  original_cases: any;

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
    )

  public searchedWordsComparison$: Observable<string[]> = this
    .searchedWordsComparison
    .valueChanges
    .pipe(
      map((search_comparison: string) => search_comparison.trim().split(' '))
    )
  searching: boolean;
  comparison_to_edit: any;
  full_cases: any;
  expand_content: boolean = false;

  constructor(
    private handlerError:CustomErrorHandler,
    public caseProvider:CaseProvider, 
    public comparisonProvider:ComparisonProvider, 
    private routeActivated: ActivatedRoute,
    private router:Router,
    private utils:UtilsFunctions,
    private _snackBar: MatSnackBar,
    public dialog:MatDialog,
    private commonProvider:CommonProvider,
    private paymentProvider:PaymentProvider,
    private responsiveProvider:ResponsiveProvider,
    private cdr: ChangeDetectorRef,
    public navigator : NavigationProvider) {

      this.bottom_paginator_data = {
        previousPageIndex: 0,
        pageIndex: 0,
        pageSize: 12,
        length: 0
      }

    }

  ngAfterViewInit() {
    this.bottom_paginator_data = {
      previousPageIndex: 0,
      pageIndex: 0,
      pageSize: 12,
      length: 0
    }
    this.getCases();
    this.list_url = global.API_CASES;
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
      this.study_cases = this.caseProvider.filterCaseList(this.original_cases,changes);
    });
    localStorage.removeItem('case_reports_id');
    localStorage.removeItem('case_in_edition');
    localStorage.removeItem('case_to_compare');
    localStorage.removeItem('proposed_filter');
    localStorage.removeItem('pending_comparison');
    localStorage.removeItem('comparison_reports');
    
    this.getProduct();
    this.commonProvider.menuIsExpanded.subscribe((menu_expanded)=>{
      if(!menu_expanded){
        this.expand_content = true;
      }else{
        this.expand_content = false;
      }
    })
  }

  getProduct(){
    this.paymentProvider.getAllProducts().subscribe((response)=>{
      this.product = response.find(plan => plan.slug == "comparison")
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  getCases(){
    this.caseProvider.getPersonalComparisonCases().subscribe((response)=>{
      this.setCases(response);
    },error=>{
      this.handlerError.handleError(error);
    })
  }


  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  setPageSizeOptions(setPageSizeOptionsInput: string) {
    if (setPageSizeOptionsInput) {
      this.pageSizeOptions = setPageSizeOptionsInput.split(',').map(str => +str);
    }
  }

  getAllComparisons(){
    for (let index = 0; index < this.study_cases.length; index++) {
      this.getComparisonsForCases(this.study_cases[index].alter_autoincrement_id,index);
    }
  }

  searchComparisons(ev,pos){
    this.searching = true;
    this.study_cases[pos].comparisons = this.comparisonProvider.filterComparisonList(this.study_cases[pos].original_comparisons,ev);
  }

  checkForNewComparison(){
    this.routeActivated.queryParams.subscribe(async params => {
      if(params.new_comparison){
        this.new_comparison = true;
        this.case_to_compare = JSON.parse(localStorage.getItem('study_case'));
        localStorage.removeItem('study_case');
        let comparison = JSON.parse(localStorage.getItem('comparison'));
        comparison ? this.comparison_to_edit = comparison : this.comparison_to_edit = null;
        if(this.case_to_compare){
          this.selected_case =  this.study_cases.find(study_case => study_case.id == this.case_to_compare.id);
          this.selected_case.selected = true;
        }
      }else{
        this.new_comparison = false;
      }
    })
  }

  newComparisonCreated(){
    this.ngAfterViewInit();
  }

  setCases(cases){
    this.full_cases = cases;
    this.study_cases = cases.results;
    this.original_cases = cases.results;
    this.study_cases = this.study_cases.map(function(e) { 
      e.comparisons = []; 
      e.selected = false;
      return e;
    });
    this.study_cases.map(case_data => case_data.toggle_chart = false);
    this.study_cases.map(case_data => case_data.original_updated = case_data.updated);
    this.study_cases.map(case_data => case_data.created_display = this.utils.formatDateStringMonth(case_data.created));
    this.study_cases.map(case_data => case_data.updated = this.utils.formatDateStringMonth(case_data.updated));
    this.study_cases.map(case_data => case_data.comparison_search_value = '');
    this.study_cases.map(case_data => case_data.cant_compare = case_data.warnings.find(warning => warning.code == 'OPT-001'));
    this.original_cases.map(case_data => case_data.toggle_chart = false);
    this.original_cases.map(case_data => case_data.order_comparisons = '-updated');

    this.setPaginatorOptions(cases);
    this.getAllComparisons();
    this.checkForNewComparison();

    setTimeout(() => {
      let scrrol_id = localStorage.getItem('scroll_to');
      if(scrrol_id){
        this._scrollTo(scrrol_id);
        localStorage.removeItem('scroll_to')
      }
    }, 500);
  }

  resetCases(){
    this.new_comparison = false;
    this.router.navigate([], {
      queryParams: {
          new_comparison: undefined,
      },
      queryParamsHandling: 'merge',
    });
    this.clearSelectedCase();
  }

  getComparisonsForCases(id,pos){
    this.comparisonProvider.getCaseComparisons(id).subscribe((response)=>{
      let case_comparisons = response.results;
      case_comparisons.map(comparison => comparison.saving_percent = Math.abs(((this.study_cases[pos].current_filter.change_rate_annual_cost - comparison.optimize_change_rate_annual_cost) * 100) / this.study_cases[pos].current_filter.change_rate_annual_cost));
      case_comparisons.map(comparison => comparison.saving_amount = (this.study_cases[pos].current_filter.change_rate_annual_cost - comparison.optimize_change_rate_annual_cost))
      case_comparisons.map(comparison => comparison.toggle_actions = false);
      this.study_cases[pos].comparisons = case_comparisons;
      this.original_cases[pos].original_comparisons = case_comparisons;
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  
  clearSelectedCase(){
    this.caseProvider.getPersonalCases().subscribe((response)=>{
      this.setCases(response);
    },error=>{
      this.handlerError.handleError(error);
    })
    this.list_url = global.API_CASES;
  }

  viewComparisonDetails(study_case,comparison){
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe((response)=>{
      let data = {
        case:response,
        comparison_data:comparison
      }
      window.scrollTo(0, 0);
      localStorage.setItem('scroll_to',study_case.alter_autoincrement_id);
      this.navigator._goComparisonResults(data);
    })
  }

  _viewOptimization(study_case){
    this.navigator._goOptimizationsResults(study_case);
  }

  editComparison(study_case, comparison){
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goEditComparison(case_details,comparison);
      }
    )
  }

  startCaseComparison(study_case){
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        localStorage.setItem('cases',JSON.stringify(this.study_cases));
        this.navigator._goNewComparisonCaseSelected(case_details);
      }
    )
  }

  startClearComparison(){
    // this.selected_case = study_case;
    // this.new_comparison = true;
    // this.comparison_to_edit = null;
    localStorage.setItem('cases',JSON.stringify(this.full_cases));
    this.navigator._goNewComparisonClear();
  }

  setPaginatorOptions(cases){
    this.length = cases.count;
    this.bottom_paginator_data.length = cases.count;
  }

  handlePage(event){
    this.bottom_paginator_data = event.event;
    this._getCaseListByUrl(event.url);
  }

  handleManualPage(event){
    window.scrollTo(0, 0);
    this.jumpToPageOption = {
      jump:true,
      page:event.pageIndex,
      data:event
    }
  }

  _editCase(case_data){
    this.caseProvider.getCaseDetails(case_data.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goNewCase(case_details);
      }
    )
  }

  _duplicateCase(new_case) {
    new_case.current_filter.model_filter = new_case.current_filter.model_filter.id;
    var has_private_reports = new_case.current_filter.reports.find(report => report.user);
    if(has_private_reports){
      var private_report_ids = [];
      new_case.current_filter.reports.forEach(report => {
          private_report_ids.push(report.id);
      });
      new_case.current_filter.private_report_ids = private_report_ids;
    }
    this.caseProvider.createNewCase(new_case).subscribe((response) => {
      this.caseProvider.getPersonalCases().subscribe((study_cases)=>{
        this.setCases(study_cases);
      },error=>{
        this.handlerError.handleError(error);
      })
    }, error => {
      this.handlerError.handleError(error);
    })
  }

  _getCaseListByUrl(url) {
    this.caseProvider.getCasesByUrl(url).subscribe((cases)=>{
      this.setCases(cases);
      this.setPaginatorOptions(cases);
    },error=>{
      this.handlerError.handleError(error);
    });
  }

  _attemptDeleteCompariosn(case_id,comparison,case_index){
    let dialog_data = {
      title:'comparison',
      description:"You WON'T be able to recover this comparison in the future"
    };
    let dialogRef = this.dialog.open(ConfirmDialogComponent, {
      width: '50%',
      height: 'auto',
      data: dialog_data
    });

    dialogRef.afterClosed().subscribe(result => {
      if(result){
        this._deleteComparison(case_id,comparison.id,case_index);
      }
    });
  }

  _orderComparisonList(case_study,pos){
    this.comparisonProvider.orderCaseComparisons(case_study.alter_autoincrement_id,case_study.order_comparisons).subscribe(
      (comparisons)=>{
        let case_comparisons = comparisons;
        case_comparisons.map(comparison => comparison.saving_percent = Math.abs(((this.study_cases[pos].current_filter.change_rate_annual_cost - comparison.optimize_change_rate_annual_cost) * 100) / this.study_cases[pos].current_filter.change_rate_annual_cost));
        case_comparisons.map(comparison => comparison.saving_amount = (this.study_cases[pos].current_filter.change_rate_annual_cost - comparison.optimize_change_rate_annual_cost))
        case_comparisons.map(comparison => comparison.toggle_actions = false);
        this.study_cases[pos].comparisons = case_comparisons;
        this.original_cases[pos].original_comparisons = case_comparisons;
    })
  }

  _deleteComparison(case_id,comparison_id,case_index){
    this.comparisonProvider.deleteComparison(comparison_id).subscribe(
    (response)=>{
