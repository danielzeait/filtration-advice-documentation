import { Component, OnInit, Input, Output, EventEmitter, ChangeDetectorRef } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { ReportsProvider } from 'src/app/providers/reports.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import {FormControl} from '@angular/forms';
import {Observable} from 'rxjs';
import {map, startWith} from 'rxjs/operators';
import { FilterProvider } from 'src/app/providers/filter.service';
import { MetaDataProvider } from 'src/app/providers/metadata.service';

export class ReportReviewComponent implements OnInit {
  public report:any = {
    is_public:true,
  };
  public report_is_private:boolean = false;
  public myControl = new FormControl();
  public filterControl = new FormControl();
  public userControl = new FormControl();
  public labOptions: Observable<string[]>;
  public modelFilterOptions: Observable<string[]>;
  public userOptions: Observable<string[]>;
  public report_points: any;
  public report_filter:any={};
  public manufacturers: any;
  public laboratories: any;
  public dust_types: any;
  public norms: any;
  public model_filters: any;
  public new_report: boolean;
  public new_model_filter: any;
  public panelChartOpenState:boolean = true;
  public panelChartAirflowOpenState:boolean = true;
  public users:any;
  public report_has_owner:boolean = false;
  public change_pdf_report:boolean = false;
  createNewObject: boolean = false;
  new_object_type: any;
  new_object_name:any;
  laboratory_id: any;
  report_pdf: any;
  complete_report: boolean;
  public airflow_points:any;
  restricted_file: any = {
    name:'',
  };
  report_owner: any;

  constructor(
    private utils:UtilsFunctions,
    private authProvider:AuthProvider,
    private _snackBar: MatSnackBar,
    private labPoints:LabPoints,
    private cdr: ChangeDetectorRef,
    private metadata:MetaDataProvider,
    private navigator:NavigationProvider,
    private commonProvider:CommonProvider,
    private notifications:NotificationService,
    private reportService:ReportsProvider,
    private filterService:FilterProvider,
    private routeActivated:ActivatedRoute) 
  { }

  ngOnInit() {
    let id = this.routeActivated.snapshot.paramMap.get("id");
    if(id){
      this.reportService.getReportDetails(id).subscribe(
        (report_details)=>{
        this.report = {...report_details};
        if(this.report.model_filter){
          this.getReportFilter();
        }
        this.report_is_private = !this.report.is_public;
        if(this.report.pdf_file && !this.report.model_filter){
          this.complete_report=true;
        }
        this.getLabReportPoints();
        this.getLabReportAirflowPoints();
      });
    }else{
      this.new_report = true;
      this.getLabReportPoints();
      this.getLabReportAirflowPoints();
    }
    this.getMetaData();
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }
  
  private getMetaData(){
    this.metadata.getAllManufacturers().subscribe(
      (manufacturers)=>{
        this.manufacturers = manufacturers;
    });
    this.getLabs();
    this.metadata.getDustType().subscribe(
      (dust_types)=>{
        this.dust_types = dust_types;
    });
    this.metadata.getAllNorms().subscribe(
      (norms)=>{
        this.norms = norms;
    });
    this.filterService.getAllPublicFilters().subscribe(
      (filters)=>{
        this.model_filters = filters;
        this.modelFilterOptions = this.filterControl.valueChanges
          .pipe(
          startWith(''),
          map(value => this._filterFilters(value,this.model_filters))
        );
      }
    )
    this.authProvider.getUsers().subscribe((users)=>{
      this.users = users;
      this.userOptions = this.userControl.valueChanges
        .pipe(
        startWith(''),
        map(value => this._filterUsers(value,this.users))
      );
    })
  }

  private getLabs(){
    this.metadata.getAllTestLaboratories().subscribe(
      (laboratories)=>{
        this.laboratories = laboratories;
        this.labOptions = this.myControl.valueChanges
          .pipe(
          startWith(''),
          map(value => this._filter(value,this.laboratories))
      );
    });
  }

  private getReportFilter(){
    this.filterService.getFilterDetails(this.report.model_filter).subscribe(
      (model_filter)=>{
        this.report_filter = model_filter;
      }
    )
  }

  private getLabReportPoints(){
    this.reportService.getReportPoints(this.report.id).subscribe(
      (points)=>{
        points.sort((a, b) => (a.drop_pressure > b.drop_pressure) ? 1 : -1)
        if(points.length > 0){
          this.report_points = points;
          this.report_points.map(point=>point.arrestance ? point.arrestance : point.arrestance = 100);
          this.report_points.map(point=>point.dust_fed = point.dust);
          this.report_points.map(point=>point.dust_held = ((point.dust_fed * point.arrestance) / 100).toFixed(2));
          if(points.length < 6){
            for (let i = points.length; i < 6; i++) {
              this.addNewEmptyPoint();
            }
          }
        }else{
          this.report_points = this.labPoints.setEmptyPoints(this.report.id);
          //this.airflow_points = this.labPoints.setEmptyAirflowPoints(this.report.id);
        }
      },(error)=>{
        this.report_points = this.labPoints.setEmptyPoints(null);
        //this.airflow_points = this.labPoints.setEmptyAirflowPoints(this.report.id);
      }
    )
  }

  private getLabReportAirflowPoints(){
    this.reportService.getReportAirflowPoints(this.report.id).subscribe(
      (points)=>{
        points.sort((a, b) => (a.drop_pressure > b.drop_pressure) ? 1 : -1)
        if(points.length > 0){
          this.airflow_points = points;
        }else{
          this.airflow_points = this.labPoints.setEmptyAirflowPoints(this.report.id);
        }
      },(error)=>{
        this.airflow_points = this.labPoints.setEmptyAirflowPoints(this.report.id);
      }
    )
  }

  private _filter(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      return option
    }
  }

  setLab(lab){
    this.report.laboratory_id = lab.id;
  } 

  public addNewEmptyPoint(){
    this.report_points.push(this.labPoints.emptyPoint(this.report.id));
  }


  private _filterFilters(value: any,array:any): string[] {
    const filterValue = value ? value.toLowerCase() : '';
    return array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
  }

  private _filterUsers(value: any,array:any): string[] {
    const filterValue = value ? value.toLowerCase() : '';
    return array.filter(option => option ? option.email.toLowerCase().includes(filterValue) : '');
  }

  public _setFilterReport(filter){
    if(filter){
      this.report.model_filter = filter;
      this.report_filter = filter;
    }
  }

  public _setReportOwner(user){
    if(user){
      this.report_owner = user;
      this.report.user = user;
      //this.report.user.email = user.email;
    }
  }

  public _toggleOwner(){
    if(!this.report_has_owner){
      this.report_owner = null;
      this.userControl = new FormControl();
      this.userOptions = this.userControl.valueChanges
        .pipe(
        startWith(''),
        map(value => this._filterUsers(value,this.users))
      );
    }
  }

  _changeModelFilter(){
    this.report.model_filter = null;
    this.report_filter = null;
    this.new_model_filter = null;
  }

  _backToReportList(){
    this.navigator._goHomeValidator();
  }
  
  _enbaleCreationForm(type){
    this.new_object_type = type
    this.createNewObject = true;
  }

  saveNewObject(){
    if(this.new_object_type == 'Laboratory'){
      this.saveNewLab();
    }
    if(this.new_object_type == 'Filter'){
      
    }
  }

  saveNewLab(){
    this.metadata.addLaboratory(this.new_object_name).subscribe(
      (response:any)=>{
        this.getLabs();
        this.report.laboratory = response.name;
        this.report.laboratory_id = response.id;
        this.createNewObject = false;
        this.commonProvider.showSnackBar('Success.','New ' + this.new_object_type + ' created correctly.' ,'success-snackbar');
      },error=>{
        this.report.laboratory = this.new_object_name;
        
      }
    )
  }

  _saveReportAirflowPoints(id_report){
    this.airflow_points.forEach(point => {
      if(point.id){
        this.reportService.editReportAirflowPoints(point).subscribe((response)=>{});  
      }else{
        point.report = id_report;
        if(point.air_flow && point.drop_pressure){
          this.reportService.setReportAirflowPoints(point).subscribe((response)=>{});
        }
      }
    });
  }

  _approveReport(){
    if(this.new_report){
      if(!this.report_has_owner){
        delete this.report.user;
        this.report.is_public = true;
      }
    }
    if(this.report_is_private){
      this.report.is_public = false;
    }else{
      this.report.is_public = true;
    }
    this.airflow_points = this.airflow_points.filter(point=>point.air_flow && point.drop_pressure);
    if(!this.new_report){
      this.report.status = "approved";
      if(!this.report.laboratory_id){
        this.report.laboratory_id = this.laboratories.find(lab => lab.name == this.report.laboratory).id;
      }
      this.reportService.editFullReport(this.report,this.report_pdf,this.restricted_file).subscribe((response:any)=>{
        this._updateReportPoints(response);
        this.commonProvider.showSnackBar('Report saved','The report has been saved correctly','success-snackbar');
        this.navigator._goHomeValidator();
      },error=>{
        console.log(error);
      })
    }else{
      this._createNewReport();
    }
    let points = this.report_points.filter(point => point.drop_pressure && (point.dust_fed || point.dust));
  }

  _createNewReport(){
    this.report.status = "approved";
    this.report.report_dust_points = this.report_points;
    this.report.report_air_flow_points = this.airflow_points;
    this.reportService.createFullReport(this.report,this.report_pdf,this.restricted_file).subscribe((response:any)=>{
      this._updateReportPoints(response);
      this.commonProvider.showSnackBar('New report created','The report has been created correctly','success-snackbar');
      this.navigator._goHomeValidator();
    },error=>{
      console.log(error);
    })
  }
  
  _updateReportPoints(lab_report){
    let report = lab_report;
    delete report.pdf_file_limited;
    delete report.pdf_file; 
    report.report_dust_points = this.report_points;
    report.report_air_flow_points = this.airflow_points;
    this.reportService.updateReport(report).subscribe(()=>{});
  }

  _rejectReport(){
    this.report.status = "rejected";
    delete this.report.pdf_file;
    this.reportService.rejectReport(this.report).subscribe((response)=>{
      this.commonProvider.showSnackBar('Report rejected','The report has been rejected','alert-snackbar');
      this.navigator._goHomeValidator();
    },error=>{
      console.log(error);
    })
  }

  _setNewPdfFile(ev){
    var type = ev.type;
    if (type !== "application/pdf") {
      this.commonProvider.showSnackBar('Alert','File selected is not a PDF file','alert-snackbar');
    } else {
      this.change_pdf_report = false;
      this.report_pdf = ev;
      this.report.pdf_file = ev.name;
    }
  }

  receiveFilterInfo(event){
    this.createNewObject = false;
    if(event){
      this.report.model_filter = event;
      this.report_filter = event;
    }
  }

  reportRestrictedChange(ev){
    var type = ev.target.files[0].type;
    if (type !== "application/pdf") {
      this.commonProvider.showSnackBar('Alert','File selected is not a PDF file','alert-snackbar');
    } else {
      this.report.pdf_file_limited = ev.target.files[0].name;
      this.restricted_file = ev.target.files[0];
    }
  }


}
