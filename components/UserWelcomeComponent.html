import { Component, OnInit } from '@angular/core';
import { MatSnackBar } from '@angular/material/snack-bar';
import { CustomSnackBarComponent } from '../custom-snackbar/custom-snack-bar.component';
import { NotificationService } from 'src/app/providers/notifications.service';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import * as moment from 'moment';
import { Message } from '@angular/compiler/src/i18n/i18n_ast';
import { connect } from 'http2';

@Component({
  selector: 'app-user-welcome',
  templateUrl: './user-welcome.component.html',
  styleUrls: ['./user-welcome.component.scss']
})
export class UserWelcomeComponent implements OnInit {
  public user_data:any;
  public notifications: any[];
  public unread_notifications: any;
  public credits: any;
  constructor(
    private navigator:NavigationProvider,
    private _snackBar: MatSnackBar, 
    private notificationProvider:NotificationService, 
    private authProvider:AuthProvider) { }
  
  ngOnInit() {
    this.user_data = JSON.parse(localStorage.getItem('user_data'));
    this.getNotifications();
    this.getCredits();
  }

  getNotifications(){
    this.notificationProvider.getNotifications().subscribe((notifications:any)=>{
      this.notifications = [...notifications];
      this.unread_notifications = this.notifications.filter(notification => !notification.is_read);
    })
  }

  getCredits(){
    this.authProvider.getUserCredits().subscribe((credits:any)=>{
      this.credits = {...credits};
    })
  }

  goToProfile(){
    this.navigator._goProfile();
  }

  openNotifications(){
    this.navigator._goNotificationInbox();
  }

  getWelcomeMessage(){
    let date = new Date().toLocaleString();
    let time = date.split(' ')[1];
    var message = '';
    // if(parseInt(time) < 12)\
    var final_time = parseInt(moment(time, 'hh:mm:ss').format('HH'));
    if(final_time > 0 && final_time < 12){
      message = "Good Morning"
    }
    if(final_time > 11 && final_time < 19){
      message = "Good Afternoon"
    }
    if(final_time > 18 && final_time < 23){
      message = "Good Evening"
    }
    return message;
  }

}
