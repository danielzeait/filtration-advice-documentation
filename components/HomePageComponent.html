import { Component, OnInit, HostListener } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { CaseProvider } from 'src/app/providers/case.service';
import { FilterProvider } from 'src/app/providers/filter.service';
import { filter } from 'rxjs/operators';
import { ComparisonProvider } from 'src/app/providers/comparison.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { DomSanitizer, SafeResourceUrl, SafeUrl } from '@angular/platform-browser';
import { LoaderProvider } from 'src/app/providers/loader.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { LoaderComponent } from 'src/app/components/loader/loader.component';
import { MatDialog } from '@angular/material/dialog';
import { CustomizeDashboardComponent } from 'src/app/components/modal-dialogs/customize-dashboard/customize-dashboard.component';
import { CommonProvider } from 'src/app/providers/common.service';
import { DashboardProvider } from 'src/app/providers/dashboard.provider';
import { SwPush } from '@angular/service-worker';

@Component({
  selector: 'app-home-page',
  templateUrl: './home-page.component.html',
  styleUrls: ['./home-page.component.scss']
})
export class HomePageComponent implements OnInit {
  public list:Array<any> = ['','','','','',''];
  public news_list:Array<any> = [];
  public counters:Array<any>;
  public breakpoint: number;
  public row_height:string;
  private num_cases:number;
  private num_filters:number;
  private num_comparisons:number=0;
  public waiting:boolean;
  public num_instruments: any;
  public stats: any;
  user_widgets:any=[];
  overview_data: any;
  expand_news: boolean;
  constructor(
    private handlerError:CustomErrorHandler, 
    private authProvider:AuthProvider, 
    private navigator:NavigationProvider, 
    private metaService:MetaDataProvider,
    private _sanitizer: DomSanitizer,
    private caseProvider:CaseProvider,
    private utils:UtilsFunctions,
    readonly swPush:SwPush,
    private loader:LoaderProvider,
    private dashboardProvider:DashboardProvider,
    public dialog:MatDialog,
    private responsiveProvider:ResponsiveProvider) { }

  ngOnInit() {
    this.breakpoint = this.responsiveProvider.getNumCols(window.innerWidth);
    this.row_height = this.responsiveProvider.getRowHeight(window.innerWidth);
    this.waiting = true;
    this.initStats();
    this.initCounters();
    this.initNews();
    this.initUserWidgets();
    this.pushSubscription();

    this.swPush.messages.subscribe((messages)=>{
    })

    this.swPush.notificationClicks.subscribe(
      ({action, notification}) => {
        window.open(notification.data.url);
    });
    //this.loader.presentLoader("Waiting for love...");
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.breakpoint = this.responsiveProvider.getNumCols(window.innerWidth);
    this.row_height = this.responsiveProvider.getRowHeight(window.innerWidth);
  }

  initCounters(){
    this.metaService.getHomeInfo().subscribe((response)=>{
      this.overview_data = response;
      this.num_cases = response.cases;
      this.num_comparisons = response.comparissons;
      this.num_filters = response.public_filters;
      this.num_instruments = response.instruments;
      this._setCounters();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  initStats(){
    this.metaService.getHomeStats().subscribe((response)=>{
      this.stats = response;
      this.stats.personal_savings ? this.stats.personal_savings = this.utils.numberThousandSeparator(this.stats.personal_savings) : 0;
      this.stats.general_savings ?  this.stats.general_savings = this.utils.numberThousandSeparator(this.stats.general_savings) : 0;
      this.stats.personal_co2_savings ? this.stats.personal_co2_savings = this.utils.numberThousandSeparator(this.stats.personal_co2_savings) : 0;
      this.stats.general_co2_savings ? this.stats.general_co2_savings = this.utils.numberThousandSeparator(this.stats.general_co2_savings) : 0;
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  initNews(){
    this.metaService.getHomeNews().subscribe((response)=>{
      this.news_list = response;
      //this.news_list.push(response[2]);
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  initUserWidgets(){
    this.authProvider.getUserDashboard().subscribe(
      ( dashboard:any )=>{
        if(dashboard){
          this.user_widgets = [...dashboard];
          this.getWidgetCaseData();
        }else{
          this.user_widgets = [
            { name:'Welcome', cols: 2, rows: 2, image:'../../../../assets/imgs/case_widget.png', type:'welcome',data:null,in_use:true},
            { name:'User overview', cols: 1, rows: 6, image:'../../../../assets/imgs/overview_widget.png', type:'overview',data:null,in_use:false},
            { name:'Merv Efficiency', cols: 3, rows: 8, image:'../../../../assets/imgs/merv_widget.png', type:'merv',data:null,in_use:true, height:'700px'},
            { name:'Optimal date of change-out', cols: 2, rows: 4, image:'../../../../assets/imgs/changeout_widget.png', type:'changeout',data:null,in_use:false},
          ]
          this.authProvider.setUserDashboard(this.user_widgets).subscribe();
        }
        this.dashboardProvider.setUserDashboard(this.user_widgets);
      }
    )
  }

  _setCounters(){
    this.waiting = false;
    this.counters = [ 
      {'count':this.num_cases,'title':'STUDY CASES','action_text':'START NEW CASE','action_url':'./case-list','list_url':'./case-list'},
      {'count':this.num_comparisons,'title':'COMPARISONS','action_text':'NEW COMPARISON','action_url':'./case-list','list_url':'./new-comparison'},
      {'count':this.num_filters,'title':'PUBLIC FILTERS','action_text':'UPLOAD REPORT','action_url':'./filter-list','list_url':'./filter-list'},
      {'count':this.num_instruments,'title':'INSTRUMENTS','action_text':'MANAGE INSTRUMENTS','action_url':'./case-list','list_url':'./instrument-list'},
    ]
  }

  openCustomDashboardModal(){
    let dialogRef = this.dialog.open(CustomizeDashboardComponent, {
      width: '75%',
      height: '95%',
      data:this.user_widgets,
    });

    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.refreshWidgets(result);
        //this.user_widgets = [...result]
      }
    });
  }

  _logOut(){
    this.authProvider.userLogut().subscribe((response)=>{
      this.authProvider.destroySession();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  _homeCases(){
    this.navigator._goHomeCases(null);
  }

  _filterList(){
    this.navigator._goFilterList();
  }

  _goToAction(url){
    window.open(url,'_blank');
  }

  _startNewCase(){
    this.navigator._goNewCase(null);
  }

  refreshWidgets(widgets){
    widgets.forEach(element => {
      if((!this.user_widgets.find(widget => widget.type == element.type) && element.in_use)){
        this.user_widgets.push(element);
      }else{
        this.user_widgets.map(card =>  {
          if(card.type == element.type){
            card.in_use = element.in_use;
          }
        });
      }
    });
    this.getWidgetCaseData();
    this.authProvider.setUserDashboard(this.user_widgets).subscribe();
  }

  getBackground(image) {
    return this._sanitizer.bypassSecurityTrustStyle(`linear-gradient(0deg, rgba(204,204,204,1) 0%, rgba(255,255,255,0) 15%, rgba(255,255,255,0) 85%, rgba(204,204,204,1) 100%), url(${image})`);
  }

  getWidgetCaseData(){
    this.user_widgets.forEach(widget => {
      if(widget.type == "case" && widget.in_use && widget.data){
        this.caseProvider.getCaseById(widget.data).subscribe(
          ( case_info ) => {
              if(case_info.results[0]){
                widget.case_data = { ...case_info.results[0] };
              }else{
                var pos = this.user_widgets.findIndex(case_widget => case_widget.type == 'case' && case_widget.data == widget.data);
                if(pos !== -1){
                  this.dashboardProvider.removeCaseFromDashboardById(widget.data);   
                  this.user_widgets.splice(pos,1);
                }
              }
          })
        }
    });
  }

  expandNews(){
    this.expand_news = true;
  }

  compressNews(){
    this.expand_news = false;
  }

  pushSubscription(){
    if(!this.swPush.isEnabled){
      console.log('Notificactions not enabled')
      return
    }
    this.swPush.requestSubscription({
      serverPublicKey:'BNjD_NSdzq0yYQMQlMhYY9ymyAw_wuhIIpCyKNvfQV98zFPyVTCo36qvaVyfHT-4WzeRUnUAFxzUV5Kwe4fqcHw',
    })
    .then((sub)=> {
    })
    .catch((error)=> console.log(error));

  }

}
