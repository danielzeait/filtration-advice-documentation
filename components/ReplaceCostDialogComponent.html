import { Component, OnInit, Inject } from '@angular/core';
import {MatDialog, MatDialogRef, MAT_DIALOG_DATA} from '@angular/material/dialog';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-replace-cost-dialog',
  templateUrl: './replace-cost-dialog.component.html',
  styleUrls: ['./replace-cost-dialog.component.scss']
})
export class ReplaceCostDialogComponent implements OnInit {
  public array_costs:any=[];
  public invalid_costs: boolean = false;
  max_num_costs: boolean = false;
  constructor(public dialogRef: MatDialogRef<ReplaceCostDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any, public utils: UtilsFunctions) { }
    
    ngOnInit() {
      this.array_costs = this.data.costs;
      if(this.data.type == "cycle"){
        this.array_costs = this.data.costs.slice(3,this.array_costs.length);
        if(this.array_costs.length == 0){
          this.array_costs = [{'name':'','value':null, 'is_annual':false}];
        }
      }else{
        if(this.array_costs.length == 1 && this.array_costs[0].value == '0.00'){
          this.array_costs[0].value = null;
        }
      }
    }

    newCostRow(){
      var bool =false;
      bool = this.data.type == 'annual';
      if(this.array_costs.length < 3){
        this.array_costs.push({
          'name':'',
          'value':'',
          'is_annual':bool,
        })
      }else{
        this.max_num_costs = true;
      }
    }

    getInvalidCost(){
      var empty_cost  = this.array_costs.find(cost => cost.name == "");
      if(empty_cost){
        this.invalid_costs = true;
        return true
      }
    }

    cancel(){
      this.dialogRef.close();
    }

    setAmountDecimal(ev){
      this.utils.transformAmount(ev);
    }

    deleteCostRow(pos){
      this.array_costs.splice(pos,1);
      if(this.array_costs.length < 3){
        this.max_num_costs = false;
      }
    }

    done(){
      var return_array = this.array_costs;
      if(!this.getInvalidCost()){
        if(this.data.type == "cycle"){
          return_array = this.data.costs.slice(0, 3).concat(this.array_costs);
        }
        this.dialogRef.close(return_array);
      }
    }

}
