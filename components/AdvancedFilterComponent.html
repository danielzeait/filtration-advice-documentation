@Component({

  @Input('data') data : any;
  @Input('models') input_models : any;

  constructor( 
    private filterProvider:FilterProvider,

  ngOnInit(){
    this.getMetaData();

    if(this.data){
      this.search_data = {...this.data};
    }

    if(this.input_models){
      this.model = {...this.input_models}
    }
  }
  
  ngAfterViewInit(){}
  
  ngOnChanges(changes:SimpleChanges){
    if(changes.simple_clear && changes.simple_clear.currentValue && changes.simple_clear.currentValue.clear_me){
      this.clearAllFilters(false);
    }
  }

  getMetaData(){
    this.metaDataProvider.getFilterTypes()
    .subscribe((filter_types)=>{
      this.options.FILTER_TYPE = filter_types;
      if(this.data.default_values){
        this.setInitValues(this.options.FILTER_TYPE,this.model.filter_type,this.data.default_values.filter_type);
      }
    });

    this.metaDataProvider.getFrameMaterial()
    .subscribe((frame_materials)=>{
      this.options.FRAME_MATERIAL = frame_materials;
      if(this.data.default_values){
        this.setInitValues(this.options.FRAME_MATERIAL,this.model.frame_material,this.data.default_values.filter_frame_material);
      }
    });

    this.metaDataProvider.getFrameTypes()
    .subscribe((frame_types)=>{
      this.options.FRAME_TYPE = frame_types;
      if(this.data.default_values){
        this.setInitValues(this.options.FRAME_TYPE,this.model.frame_type,this.data.default_values.filter_frame_type);
      }
    });

    this.metaDataProvider.getMediaTypes()
    .subscribe((media_types)=>{
      this.options.MEDIA_MATERIAL = media_types;
      if(this.data.default_values){
        this.setInitValues(this.options.MEDIA_MATERIAL,this.model.media_type,this.data.default_values.filter_media_material);
      }
    });

    this.metaDataProvider.getDustType()
    .subscribe((dust_types)=>{
      this.options.DUST_TYPE = dust_types;
      if(this.data.default_values){
        this.setInitValues(this.options.DUST_TYPE,this.model.dust_type,this.data.default_values.filter_dust_type);
      }
    });

    this.metaDataProvider.getAllNorms()
    .subscribe((norms)=>{
      this.norms = norms;
      this.options.NORMS = norms;
      if(this.search_data.default_values){
        this.search_data.standard = this.norms.find(norm => norm.id == this.search_data.default_values.standard.id).id;
        this.setInitObjectValues(this.options.NORMS,this.model.standard,this.search_data.standard);
      }
    });

    this.metaDataProvider.getAllManufacturers()
    .subscribe((manufacturers)=>{
      this.options.MANUFACTURERS = manufacturers;
      if(this.data.default_values){
        this.setInitObjectValues(this.options.MANUFACTURERS,this.model.manufacturer,this.data.manufacturer[0]);
      }
    });

    this.metaDataProvider.getAllTestLaboratories()
    .subscribe((labs)=>{
      this.options.LABORATORIES = labs;
    });

    this.metaDataProvider.getProviderType()
    .subscribe((providers)=>{
      this.options.PROVIDERS = providers;
    });
  }

  setFiltersType(ev,value){
    if(ev.checked){

  async prepareUrl(){
    let filter_type = "";
    let frame_type = "";
    let dust_type = "";
    let frame_material ="";
    let manufacturer = "";
    let media_type = "";
    let provider = "";
    let standard = "";
    let lab = "";
    let is_private = "";
    for (let index = 0; index < this.search_data.filter_type.length; index++) {
      filter_type = filter_type + "filter_type="+this.search_data.filter_type[index];
      if(this.search_data.filter_type.length > 1 && index < this.search_data.filter_type.length-1){
        filter_type = filter_type + "&";
      }
    }

    for (let index = 0; index < this.search_data.frame_type.length; index++) {
      frame_type = frame_type + "frame_type="+this.search_data.frame_type[index];
      if(this.search_data.frame_type.length > 1 && index < this.search_data.frame_type.length-1){
        frame_type = frame_type + "&";
      }
    }

    for (let index = 0; index < this.search_data.dust_type.length; index++) {
      dust_type = dust_type + "dust_type="+this.search_data.dust_type[index];
      if(this.search_data.dust_type.length > 1 && index < this.search_data.dust_type.length-1){
        dust_type = dust_type + "&";
      }
    }

    for (let index = 0; index < this.search_data.frame_material.length; index++) {
      frame_material = frame_material + "frame_material="+this.search_data.frame_material[index];
      if(this.search_data.frame_material.length > 1 && index < this.search_data.frame_material.length-1){
        frame_material = frame_material + "&";
      }
    }

    for (let index = 0; index < this.search_data.manufacturer.length; index++) {
      manufacturer = manufacturer + "manufacturer="+this.search_data.manufacturer[index];
      if(this.search_data.manufacturer.length > 1 && index < this.search_data.manufacturer.length-1){
        manufacturer = manufacturer + "&";
      }
    }

    for (let index = 0; index < this.search_data.media_type.length; index++) {
      media_type = media_type + "media_type="+  this.search_data.media_type[index];
      if(this.search_data.media_type.length > 1 && index < this.search_data.media_type.length-1){
        media_type = media_type + "&";
      }
    }

    for (let index = 0; index < this.search_data.provider_type.length; index++) {
      provider = provider + "provider_type="+this.search_data.provider_type[index];
      if(this.search_data.provider_type.length > 1 && index < this.search_data.provider_type.length-1){
        provider = provider + "&";
      }
    }

    for (let index = 0; index < this.search_data.laboratory.length; index++) {
      lab = lab + "laboratory="+this.search_data.laboratory[index];
      if(this.search_data.laboratory.length > 1 && index < this.search_data.laboratory.length-1){
        lab = lab + "&";
      }
    }

    let url_list = "";
    if(this.search_data.filter_type.length > 0){
      url_list = filter_type 
    }
    if(this.search_data.frame_type.length > 0){
      url_list = url_list + "&" + frame_type 
    }
    if(this.search_data.frame_material.length > 0){
      url_list = url_list + "&" + frame_material 
    }
    if(this.search_data.dust_type.length > 0){
      url_list = url_list + "&" + dust_type;
    }
    if(this.search_data.manufacturer.length > 0){
      url_list = url_list + "&"+ manufacturer;
    }
    if(this.search_data.media_type.length > 0){
      url_list = url_list + "&" + media_type;
    }
    if(this.search_data.provider_type.length > 0){
      url_list = url_list + "&" + provider;
    }
    if(this.utils.isNumber(this.search_data.standard)){
      url_list = url_list + "&" + "standard="+this.search_data.standard;
    }
    if(this.search_data.laboratory.length > 0){
      url_list = url_list + "&" + lab;
    }

    if(this.search_data.incinerable){
      url_list = url_list + '&' + "incinerable=true"
    }

    if(this.search_data.no_incinerable){
      url_list = url_list + '&' + "incinerable=false"
    }

    if(this.search_data.face_screen){
      url_list = url_list + '&' + "face_screen=true"
    }

    if(this.search_data.no_face_screen){
      url_list = url_list + '&' + "face_screen=false"
    }
    
    if(url_list[0] === "&"){
      url_list =  url_list.substr(1);
    }
    return url_list
  }

  async searchFilters(){
    var url = await this.prepareUrl();
    if(this.search_data.pdf_date_min){
      this.search_data.pdf_date_min = await this.utils.formatDate(this.search_data.pdf_date_min)
    }
    if(this.search_data.pdf_date_max){
      this.search_data.pdf_date_max = await this.utils.formatDate(this.search_data.pdf_date_max)
    }
    this.filterProvider.searchFilters(this.search_data,url).subscribe((filters)=>{
      let emit_data = {
        filters:filters,
        data:this.search_data,
        url:url,
        model:this.model,
      }
      this.filtered.emit(emit_data);
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  clearAllFilters(emit){
    this.model = {
      filter_type:[],
      media_type:[],
      manufacturer:[],
      frame_type:[],
      frame_material:[],
      dust_type:[],
      laboratory:[],
      provider:[],
      standard:[],
    }
    this.search_data = {...Objects.search_data};
    this.search_data.filter_type = [];
    this.search_data.media_type = [],
    this.search_data.manufacturer = [],
    this.search_data.frame_type = [],
    this.search_data.frame_material = [],
    this.search_data.dust_type = [],
    this.search_data.laboratory = [],
    this.search_data.provider = [],
    this.search_data.standard = [],
    this.clear_all = true;
    if(emit){
      this.emitClear();
    }
  }
  
  emitClear(){
    this.clear_filters.emit();
  }

  hideSearch(){
    this.hide.emit();
  }
  

}
