import { Component, OnInit, Output, EventEmitter, ChangeDetectorRef } from '@angular/core';
import { PaymentProvider } from 'src/app/providers/payment.service';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-billings',
  templateUrl: './billings.component.html',
  styleUrls: ['./billings.component.scss']
})
export class BillingsComponent implements OnInit {
  @Output() show_plans = new EventEmitter<boolean>();
  public original_billings:any;
  public billings: any;
  public search: string = "";
  public searchedWordsControl = new FormControl('');
  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
    )
  constructor(
    private handlerError:CustomErrorHandler,
    private utils : UtilsFunctions,
    private navigator:NavigationProvider,
    private commonProvider:CommonProvider,
    private paymentProvider:PaymentProvider,private cdr: ChangeDetectorRef,) { }

  ngOnInit() {
    this._getUserBillings();
  }

  ngAfterViewInit() {
    // this.searchedWordsControl.valueChanges.subscribe((changes) => {
    //   this.billings = this.paymentProvider.filterBillingList(this.original_billings,changes);
    // })
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  _getUserBillings(){
    this.paymentProvider.getUserBillings().subscribe((response)=>{
      //response.map(bill => bill.created = this.utils.formatDateTime(bill.created))
      this.billings = response;
      this.original_billings = response;
      //this.billings.reverse();
      //this.original_billings.reverse();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  _viewReceipt(bill){
    // window.open(bill.meta_data.charges.data[0].receipt_url, "_blank");
    this.navigator._viewInvoice(bill);
    // this.paymentProvider.getPaymentInvoice(bill).subscribe((invoice)=>{
    //   localStorage.setItem('invoice',invoice.html);
    // })
  }

  _listPlans(){
    this.show_plans.emit(true);
  }

}
