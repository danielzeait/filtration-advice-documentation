import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { CommonProvider } from 'src/app/providers/common.service';
import { PaymentProvider } from 'src/app/providers/payment.service';

@Component({
  selector: 'app-invoice',
  templateUrl: './invoice.component.html',
  styleUrls: ['./invoice.component.scss']
})
export class InvoiceComponent implements OnInit {
  invoice:any;
  constructor(private routeActivated:ActivatedRoute, private paymentProvider:PaymentProvider) { }

  ngOnInit(): void {
    // localStorage.removeItem('invoice');
    let id = this.routeActivated.snapshot.paramMap.get("id");
    this.paymentProvider.getPaymentInvoice(id).subscribe((invoice)=>{
      // this.invoice = localStorage.getItem('invoice');
      this.invoice =invoice.html
    })
  }
  
  ngAfterViewInit(){
    var nameEls = document.getElementsByClassName("header-image");
    var logo = document.getElementsByTagName("img");
    setTimeout(() => {
      logo[0].setAttribute('width','250px');
      nameEls[0].setAttribute('width','500px');
    }, 1000);
  }

}
