import { Component, Input, OnInit } from '@angular/core';

@Component({
  selector: 'app-filter-list-review',
  templateUrl: './filter-list-review.component.html',
  styleUrls: ['./filter-list-review.component.scss']
})
export class FilterListReviewComponent implements OnInit {
  @Input('filter') filter;
  public show_sellers:boolean = false;
  public filter_reports:any = [];
  constructor() { }

  ngOnInit(): void {
    this.filter_reports = this.filter.reports.private.concat(this.filter.reports.public);
  }

}
