import { Component, OnInit, AfterViewInit, Input, SimpleChanges } from '@angular/core';
import Chart from 'chart.js';

@Component({
  selector: 'app-comparison-dp-vs-time-chart',
  templateUrl: './comparison-dp-vs-time-chart.component.html',
  styleUrls: ['./comparison-dp-vs-time-chart.component.scss']
})

export class ComparisonDpVsTimeChartComponent implements AfterViewInit {
    @Input('current_data') current_data : any;
    @Input('current_data_optimized') current_data_optimized : any;  
    @Input('proposed_data_optimized') proposed_data_optimized : any;  
    @Input('limit_dp') limit_dp : any;
    @Input('chart_id') chart_id : any;
    @Input('current_dp_limit') current_dp_limit:any;
    @Input('current_optimized_dp_limit') current_optimized_dp_limit:any;
    @Input('current_selected_dp_limit') current_selected_dp_limit:any;
    @Input('proposed_optimized_dp_limit') proposed_optimized_dp_limit:any;
    @Input('current_data_selected') current_data_selected:any;
    @Input('proposed_data_selected') proposed_data_selected:any;
    @Input('proposed_selected_dp_limit') proposed_selected_dp_limit:any;
    @Input('chart_x_type') chart_type : any;

    public chartDataCurrent:any=[];
    public chartDataCurrentOptimized:any=[];
    public chartDataProposedOptimized:any=[];
    public chartDataSelectedCurrent: any=[];
    public chartDataSelectedProposed: any=[];
    public already_fixed_x_axis:boolean;
    public chart: any;
    private verticalLimitChartData:any=[];
    private datas: any = [];
    
    constructor() { 
    }
    
    ngAfterViewInit() {
      
      this.chartDataCurrent.data = this.setCurrentChartData(this.chart_type);
      this.chartDataCurrent.label = "CF"
      this.chartDataCurrent.color = "#003A70";
      this.chartDataCurrent.name = "current";
      this.chartDataCurrent.dp_value = this.current_dp_limit;
      
      this.chartDataCurrentOptimized.data = this.setCurrentOptimizedChartData(this.chart_type);
      this.chartDataCurrentOptimized.label = "CF (Optimized)"
      this.chartDataCurrentOptimized.color = "#199AD8";
      this.chartDataCurrentOptimized.name = "current_optimized";
      this.chartDataCurrentOptimized.dp_value = this.current_optimized_dp_limit;
      
      this.chartDataProposedOptimized.data = this.setProposedOptimizedChartData(this.chart_type);
      this.chartDataProposedOptimized.label = "PF (Optimized)"
      this.chartDataProposedOptimized.color = "#6B20B7";
      this.chartDataProposedOptimized.name = "proposed_optimized";
      this.chartDataProposedOptimized.dp_value = this.proposed_optimized_dp_limit;

      this.chartDataSelectedCurrent.data = [];
      this.chartDataSelectedCurrent.label = "CF (Selected)"
      this.chartDataSelectedCurrent.color = "#c7d42d";
      this.chartDataSelectedCurrent.name = "current_selected";
      this.chartDataSelectedCurrent.dp_value = this.current_selected_dp_limit;

      this.chartDataSelectedProposed.data = [];
      this.chartDataSelectedProposed.label = "PF (Selected)"
      this.chartDataSelectedProposed.color = "#BD2AD3";
      this.chartDataSelectedProposed.name = "proposed_selected";
      this.chartDataSelectedProposed.dp_value = this.proposed_selected_dp_limit;


      this.datas = [
        this.chartDataCurrent,
        this.chartDataCurrentOptimized,
        this.chartDataProposedOptimized,
        this.chartDataSelectedCurrent,
        this.chartDataSelectedProposed,
      ];
      
      let longest_data = this.getLongestData();

      if(longest_data.data.length > 0){
        this.verticalLimitChartData = [
          {x:0,y:Number.parseFloat(this.limit_dp).toFixed(2)},
          {x:Math.ceil(longest_data.data[longest_data.data.length-1].x),y:Number.parseFloat(this.limit_dp).toFixed(2)}
        ]
      }
      
      this.orderCharts();
      this.initChart();
    }
  
    ngOnChanges(changes: SimpleChanges){
      var update = false;

      if(changes.chart_type && !changes.chart_type.firstChange){
        this.chart_type = changes.chart_type.currentValue;
        this.chart.scales.xAxis2.hidden = this.chart_type == 'dust' ? false : true;
        this.chart.scales.xAxis2.options.display = this.chart_type == 'dust' ? true : false;
        update = true;
      }

      if(changes.current_data_selected && changes.current_data_selected.currentValue){
        this.current_data_selected = changes.current_data_selected.currentValue;
        
        this.chartDataCurrent = this.setCurrentChartData(this.chart_type);
        this.chartDataCurrentOptimized = this.setCurrentOptimizedChartData(this.chart_type);
        this.chartDataSelectedCurrent = this.setSelectedChartCurrent(this.chart_type);

        this.chart.data.datasets.find(data => data.label == "CF").data = this.chartDataCurrent;
        this.datas.find(data => data.name == "current").data = this.chartDataCurrent;
        this.datas.find(data => data.name == "current").dp_value = this.current_dp_limit;

        this.chart.data.datasets.find(data => data.label == "CF (Optimized)").data = this.chartDataCurrentOptimized;
        this.datas.find(data => data.name == "current_optimized").data = this.chartDataCurrentOptimized;
        this.datas.find(data => data.name == "current_optimized").dp_value = this.current_optimized_dp_limit;

        this.chart.data.datasets.find(data => data.label == "CF (Selected)").data = this.chartDataSelectedCurrent;
        this.datas.find(data => data.name == "current_selected").data = this.chartDataSelectedCurrent;
        this.datas.find(data => data.name == "current_selected").dp_value = this.current_selected_dp_limit;

        this.chart.options.scales.xAxes[0].ticks.max = this.chartDataSelectedCurrent[this.chartDataSelectedCurrent.length - 1].x;
        this.chart.options.scales.xAxes[1].ticks.max = this.chartDataSelectedCurrent[this.chartDataSelectedCurrent.length - 1].z;

        update = true;
      }

      if(changes.proposed_data_selected && changes.proposed_data_selected.currentValue && !changes.proposed_data_selected.isFirstChange()){

        this.proposed_data_selected = changes.proposed_data_selected.currentValue;
        this.proposed_data_selected = changes.proposed_data_selected.currentValue;
        
        this.chartDataSelectedProposed = this.setSelectedChartProposed(this.chart_type);
        this.chartDataProposedOptimized =  this.setProposedOptimizedChartData(this.chart_type);

        this.chart.data.datasets.find(data => data.label == "PF (Selected)").data = this.chartDataSelectedProposed;
        this.datas.find(data => data.name == "proposed_selected").data = this.chartDataSelectedProposed;
        this.datas.find(data => data.name == "proposed_selected").dp_value = this.proposed_selected_dp_limit;

        this.chart.data.datasets.find(data => data.label == "PF (Optimized)").data = this.chartDataProposedOptimized;
        this.datas.find(data => data.name == "proposed_optimized").data = this.chartDataProposedOptimized;
        this.datas.find(data => data.name == "proposed_optimized").dp_value = this.proposed_optimized_dp_limit;

        this.chart.options.scales.xAxes[0].ticks.max = this.chartDataSelectedProposed[this.chartDataSelectedProposed.length - 1].x;
        this.chart.options.scales.xAxes[1].ticks.max = this.chartDataSelectedProposed[this.chartDataSelectedProposed.length - 1].z;

        update = true;
      }

      if(update){
        var x2 = this.current_data[this.current_data.length-1].time;
        this.verticalLimitChartData = [
          {x:0,y:Number.parseFloat(this.limit_dp).toFixed(2)},
          {x:Math.ceil(x2),y:Number.parseFloat(this.limit_dp).toFixed(2)}
        ]
        this.chart.data.datasets[0].data = this.verticalLimitChartData;

        this.orderCharts();
        this.chart.update();
      }
    }

    getLongestData(){
      let datas  = [this.chartDataCurrent,this.chartDataCurrentOptimized,this.chartDataProposedOptimized];
      datas.sort(function(a, b) {  return a.data.length - b.data.length; })
      return datas[0]
    }
  
    orderCharts(){
      this.datas.sort(function(a, b) {  return a.dp_value - b.dp_value; })
      if(this.chart){
        this.chart.data.datasets[1].data = this.datas[0].data;
        this.chart.data.datasets[2].data = this.datas[1].data;
        this.chart.data.datasets[3].data = this.datas[2].data;
        this.chart.data.datasets[4].data = this.datas[3].data;
        this.chart.data.datasets[5].data = this.datas[4].data;
        
        this.chart.data.datasets[1].label = this.datas[0].label;
        this.chart.data.datasets[2].label = this.datas[1].label;
        this.chart.data.datasets[3].label = this.datas[2].label;
        this.chart.data.datasets[4].label = this.datas[3].label;
        this.chart.data.datasets[5].label = this.datas[4].label;
        
        this.chart.data.datasets[1].backgroundColor = this.datas[0].color;
        this.chart.data.datasets[2].backgroundColor = this.datas[1].color;
        this.chart.data.datasets[3].backgroundColor = this.datas[2].color;
        this.chart.data.datasets[4].backgroundColor = this.datas[3].color;  
        this.chart.data.datasets[5].backgroundColor = this.datas[4].color;
      };
    }
  
    setSelectedChartCurrent(type){
      var chart_selected = [];
      for (let i = 0; i < this.current_data_selected.length; i++) {
        var point = {
          'x':Number.parseFloat(this.current_data_selected[i].time),
          'y':Number.parseFloat(this.current_data_selected[i].drop_pressure),
          'z':Number.parseFloat(this.current_data_selected[i].dust)
        }
        chart_selected.push(point);
      }
      return chart_selected;
    }

    setSelectedChartProposed(type){
      var chart_selected = [];
      for (let i = 0; i < this.proposed_data_selected.length; i++) {
        var point = {
          'x':Number.parseFloat(this.proposed_data_selected[i].time),
          'y':Number.parseFloat(this.proposed_data_selected[i].drop_pressure),
          'z':Number.parseFloat(this.proposed_data_selected[i].dust)
        }
        chart_selected.push(point);
      }
      return chart_selected;
    }
  
    setCurrentChartData(type){
      var chart_current = [];
      for (let i = 0; i < this.current_data.length; i++) {
        var point = {
          'x':Number.parseFloat(this.current_data[i].time),
          'y':Number.parseFloat(this.current_data[i].drop_pressure),
          'z':Number.parseFloat(this.current_data[i].dust)
        }
        chart_current.push(point); 
      }
      return chart_current
    }
  
    setCurrentOptimizedChartData(type){
      var chart_optimized = [];
      for (let i = 0; i < this.current_data_optimized.length; i++) {
        var point = {
          'x':Number.parseFloat(this.current_data_optimized[i].time),
          'y':Number.parseFloat(this.current_data_optimized[i].drop_pressure),
          'z':Number.parseFloat(this.current_data_optimized[i].dust)
        }
        chart_optimized.push(point);
      }
      return chart_optimized
    }

    setProposedOptimizedChartData(type){
      var chart_proposed_optimized = [];
      for (let i = 0; i < this.proposed_data_optimized.length; i++) {
        var point = {
          'x':Number.parseFloat(this.proposed_data_optimized[i].time),
          'y':Number.parseFloat(this.proposed_data_optimized[i].drop_pressure),
          'z':Number.parseFloat(this.proposed_data_optimized[i].dust)
        }
        chart_proposed_optimized.push(point);
      }
      return chart_proposed_optimized
    }
    
    updateChartTicks(max,step) {
      setTimeout(() => {
        if(!this.already_fixed_x_axis){
          this.already_fixed_x_axis = true;
          this.chart.options.scales.xAxes[1].ticks.min = 0;
          this.chart.options.scales.xAxes[1].ticks.max = max;
          this.chart.options.scales.xAxes[1].ticks.stepSize = step;
          this.chart.update();
        }
      }, 250); 
    };
  
  
    initChart(){
      var id = this.chart_id;
      var ctx:HTMLElement = <HTMLElement> document.getElementById(id);
      ctx.style.backgroundColor = '#FFF';
      ctx.style.padding = '10px';
  
      var canvas : any = document.getElementById(id);
      var ctx2 = canvas.getContext("2d");
      
      Chart.Legend.prototype.afterFit = function() {
        this.height = this.height + 10;
      };
      
      this.chart = new Chart(ctx2, {
          type: 'line',
          pointBackgroundColor: "#55bae7",
          pointBorderColor: "#55bae7",
          data: {
              datasets: [
                {
                  label:'0',
                  data:this.verticalLimitChartData,
                  pointRadius:'0',
                  borderColor:'#ffa629',
                  borderWidth:3,
                  fill:false,
                  backgroundColor:'#f3f3f3'
                },
                {
                  label:this.datas[0].label,
                  data: this.datas[0].data,
                  backgroundColor:this.datas[0].color,
                  borderWidth:1,
                  lineTension:0,
                  pointRadius:'0',
                  clip:-1,
                },
                {
                  label:this.datas[1].label,
                  data: this.datas[1].data,
                  backgroundColor:this.datas[1].color,
                  borderWidth:1,
                  lineTension:0,
                  pointBorderColor: "#222",
                  pointRadius:'0',
                  clip:-1,
                },
                {
                  label:this.datas[2].label,
                  data: this.datas[2].data,
                  backgroundColor:this.datas[2].color,
                  borderWidth:1,
                  lineTension:0,
                  pointBorderColor: "#222",
                  pointRadius:'0',
                  clip:-1,
                },
                {
                  label:this.datas[3].label,
                  data: this.datas[3].data,
                  backgroundColor:this.datas[3].color,
                  borderWidth:1,
                  lineTension:0,
                  pointBorderColor: "#222",
                  pointRadius:'0',
                  clip:-1,
                },
                {
                  label:this.datas[4].label,
                  data: this.datas[4].data,
                  backgroundColor:this.datas[4].color,
                  borderWidth:1,
                  lineTension:0,
                  pointBorderColor: "#222",
                  pointRadius:'0',
                  clip:-1,
                },
              ]
          },
          options: {
            legend: {
              align:'start',
              labels: {
                filter: function(item, chart) {
                  // Logic to remove a particular legend item goes here
                  return !item.text.includes('0');
                },
                fontFamily:'lato',
                fontSize:11.5
              },
            },
            elements: {
              line: {
                tension: 0.4
              }
            },
            tooltips:{
              callbacks: {
                title:function(tooltipItem) {
                  var label = 'Dp value: ' + Number.parseFloat(tooltipItem[0].yLabel).toFixed(3);
                  return label;
                },
                label: function(tooltipItem){
                  var value = Number.parseFloat(tooltipItem.xLabel).toFixed(2);
                  return 'Time: '  + value + ' months';
                },
                afterBody: (tooltipItem,data) => {
                  var value = '';
                  if(this.chart_type == 'dust'){                  
                    value = data.datasets[tooltipItem[0].datasetIndex].data.find(point => point.x == tooltipItem[0].xLabel).z.toFixed(2);
                    value =  'Dust: ' + value + ' grams'
                  }
                  return value;
                }
              }
            },
            scales: {
                xAxes: [
                  {
                    type: 'linear',
                    position: 'bottom',
                    scaleLabel: {
                      display: true,
                      labelString: 'Time of Change in Months',
                      fontColor:'#222',
                      fontSize:10,
                    },
                    ticks: {
                      //stepSize: 2,
                      fontSize:10,
                      callback: function(value, index, values) {
                        return Number.parseFloat(value).toFixed(1);
                      },
                      max:this.current_data[this.current_data.length - 1].time
                    }
                  },
                  {
                    id:'xAxis2',
                    position: 'bottom',
                    type:"linear",
                    scaleLabel: {
                      display: true,
                      labelString:'Dust Fed (in grams)',
                      fontColor:'#222',
                      fontSize:10,
                    },
                    ticks: {
                      //stepSize: 2,
                      fontSize:10,
                      callback: function(value, index, values) {
                        return Number.parseFloat(value).toFixed(1);
                      },
                      //max:this.chart_data[this.chart_data.length - 1].dust
                    },
                    afterBuildTicks: (scale) => {
                      var val_1 = Math.ceil((this.current_data[this.current_data.length-1]).dust) / 5;
                      //var val_2 = Math.floor(Math.abs((this.chart_data[this.chart_data.length-1]) - (this.chart_data[this.chart_data.length-3]).dust))
                      let new_scale = [0];
                      for (let index = 0; index < 5; index++) {
                        new_scale.push(new_scale[index] + val_1);
                      }
                      this.updateChartTicks(new_scale[new_scale.length-1],val_1);
                      return;
                    },
                  },
                ],
                yAxes: [{
                  scaleLabel: {
                    display: true,
                    labelString: 'Pressure Drop (inch of WC)',
                    fontColor:'#222',
                    fontSize:10,
                  },
                  ticks: {
                    beginAtZero: true,
                    fontSize:10,
                    //stepSize:0.5
                  },
                },
              ],
            }
          }
      });
      
      
  
    }
