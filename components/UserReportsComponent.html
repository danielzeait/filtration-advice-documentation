import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ReportRejectedComponent } from 'src/app/components/modal-dialogs/report-rejected/report-rejected/report-rejected.component';
import { UploadReportComponent } from 'src/app/components/modal-dialogs/upload-report/upload-report/upload-report.component';
import { ReportsProvider } from 'src/app/providers/reports.service';

@Component({
  selector: 'app-user-reports',
  templateUrl: './user-reports.component.html',
  styleUrls: ['./user-reports.component.scss']
})
export class UserReportsComponent implements OnInit {

  user_reports:any = [];

  constructor(private reportsProvider:ReportsProvider,public dialog: MatDialog,) { }

  ngOnInit(): void {
    this.getUserReports();
  }

  getUserReports(){
    this.reportsProvider.getUserReports().subscribe((reports)=>{
      this.user_reports = reports;
    })
  }

  showRejectedModal(){
    this.dialog.open(ReportRejectedComponent, {
      width: '70%',
      height: 'auto',
    });
  }

  _uploadReport(){
    const dialogRef = this.dialog.open(UploadReportComponent, {
      width: '35%',
      height: '75%',
      id: 'upload-report',
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.getUserReports();
      }
    });
  }

}
