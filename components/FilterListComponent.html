import { Component, OnInit, HostListener, ChangeDetectorRef } from '@angular/core';
import { FilterProvider } from 'src/app/providers/filter.service';
import { FormControl } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { AdvancedSearchComponent } from 'src/app/components/modal-dialogs/advanced-search-dialog/advanced-search.component';
import { clear_search_data } from "../../../mocks/mocks.objects";
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { Filter } from 'src/app/models/filter.model';
import { ActivatedRoute, Router } from '@angular/router';
import { SearchModel } from 'src/app/models/search.model';
import { UploadReportComponent } from 'src/app/components/modal-dialogs/upload-report/upload-report/upload-report.component';
import { RequestFilterComponent } from 'src/app/components/modal-dialogs/request-filter/request-filter/request-filter.component';
import { RequestFilterOptionsComponent } from 'src/app/components/modal-dialogs/request-filter-options/request-filter-options/request-filter-options.component';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-filter-list',
  templateUrl: './filter-list.component.html',
  styleUrls: ['./filter-list.component.scss']
})

export class FilterListComponent implements OnInit {
  public searchedWordsControl = new FormControl('');

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
    )

  private original_filters: Array<Filter> = [];
  private advanced_search_data: any = {};

  public filters: Array<Filter> = new Array<Filter>();
  public filters_search_result: Array<Filter> = [];
  public filter_selected: Filter = null;
  public new_report: boolean = false;
  public showFilterDetails: boolean = false;
  public num_cols: number;
  public search: string = "";
  public texts_to_hightlight: NodeListOf<Element>;
  public didSearch: boolean = false;
  public isImgLoaded:any;
  show_advanced_filter: boolean;
  url_search_data: any;
  search_models: any;
  clear_search: any;

  constructor(
    private route: ActivatedRoute,
    private router: Router,
    private filterProvider: FilterProvider,
    private responsiveProvider: ResponsiveProvider,
    private routeActivated: ActivatedRoute,
    private navigator:NavigationProvider,
    private cdr: ChangeDetectorRef,
    public dialog: MatDialog) { }


  ngOnInit() {
    this.routeActivated.params.subscribe((params) => {
      if(params && params.id){
        this.filterProvider.getFilterDetails(params.id).subscribe((filter)=>{
          this._openFilterDetails(filter);
        })
      }else{
        this.filterProvider.getAllPublicFilters().subscribe(this.setFilters); 
        this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
        Object.assign(this.advanced_search_data, clear_search_data);
        this.routeActivated.queryParams.subscribe(async params => {
          if(params.new_report){
            this.new_report = true;
          }else{
            this.new_report = false;
          }
        })
      }
    })
  }

  setFilters = (filters: Array<Filter>) => {
    this.filters = [...filters];
    this.original_filters = [...filters];
    this.routeActivated.queryParams.subscribe(async params => {
      if(params.filter){
        let filter = this.filters.find(filter=> filter.id === parseInt(params.filter));
        this._openFilterDetails(filter);
      }
    })
  }

  ngAfterViewInit() {
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
    })
  }
  
  searchBywords(){
    if(this.filters_search_result.length > 0){
      this.filters = [...this.filterProvider.filterFilterList(this.filters_search_result, this.search)]
    }else{
      this.filters = [...this.filterProvider.filterFilterList(this.original_filters, this.search)]  
    }
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }

  _openSearchModal(): void {
    this.show_advanced_filter = !this.show_advanced_filter;
  }
  
  refreshFilters(result){
    this.filters = [...result.filters];
    this.didSearch = true;
    this.filters_search_result = [...result.filters];
    this.advanced_search_data = {...result.data};
    this.url_search_data = result.url;
    this.search_models = {...result.model};
  }

  clearFiltersFilter() {
    this.filters = [...this.original_filters];
    this.advanced_search_data = {...clear_search_data};
    this.url_search_data = '';
    this.search_models = {
      filter_type:[],
      media_type:[],
      manufacturer:[],
      frame_type:[],
      frame_material:[],
      dust_type:[],
      laboratory:[],
      provider:[],
      standard:[],
    };
    this.filters_search_result = [];
    this.didSearch = false;
    this.clear_search = {...{clear_me:true}};
  }

  _openFilterDetails(filter) {
    this.filter_selected = filter;
    this.showFilterDetails = true;
    this.show_advanced_filter = false;
  }

  goFilterDetails(filter){
    this.filter_selected = filter;
    this.showFilterDetails = true;
    this.show_advanced_filter = false;
    this.navigator._goFilterDetailsId(filter.id);
  }

  backClick(){
    this.showFilterDetails = false;
    //this.router.navigate([], { queryParams:null});
    this.navigator._goFilterList();
    if(this.didSearch){
      // this.filterProvider.searchFilters(this.advanced_search_data,this.url_search_data).subscribe((filters)=>{
      //   this.filters = filters;
      //   this.filters_search_result = filters;
      // })
      setTimeout(() => {
      }, 500);
    }
  }

  //ABOUT FILTER REQUEST

  openRequestFilterOptions(){
    const dialogRef = this.dialog.open(RequestFilterOptionsComponent, {
      width: '70%',
      height: 'auto',
    });
    
    dialogRef.afterClosed().subscribe(option => {
      if (option && option == 1) {
        this._requestFilter()
      }
      if (option && (option == 2 || option == 3) ) {
        this._uploadTestReport()
      } 
    });
  }

  _requestFilter(){

    let dialog_data = {
      filters:this.original_filters
    };

    const dialogRef = this.dialog.open(RequestFilterComponent, {
      width: '27.5%',
      height: 'auto',
      id: 'request-filter',
      data: dialog_data
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
      }
    });
  }

  _uploadTestReport(){
    
    let dialog_data = {
      filters:this.original_filters
    };

    const dialogRef = this.dialog.open(UploadReportComponent, {
      width: '30%',
      height: 'auto',
      id: 'upload-report',
      data: dialog_data
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
      }
    });
  }

}
