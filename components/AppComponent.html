import { Component, OnInit, ViewChild } from '@angular/core';
import { Router,Event, ActivatedRoute, ActivatedRouteSnapshot, NavigationStart, NavigationEnd, NavigationError, NavigationCancel, RouterOutlet, ActivationStart } from '@angular/router';
import { AuthProvider } from './providers/auth.service';
import { AuthSonicuProvider } from './providers/authsonicu.service';
import { NavigationProvider } from './providers/navigation.provider';
import { CustomErrorHandler } from './providers/error.service';
import { SuccessDialogComponent } from './components/modal-dialogs/success-dialog/success-dialog.component';
import { MatDialog } from '@angular/material/dialog';
import { CommonProvider } from './providers/common.service';
import { HostListener } from '@angular/core';
import { environment } from '../environments/environment';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent implements OnInit {
  @ViewChild(RouterOutlet) outlet: RouterOutlet;
  title = 'howk';
  routes:any=[];
  outer_header:boolean = true;
  link_active:any = 'landing';
  validator:any;
  random: boolean;
  invoice:boolean = false;
  display_responsive:boolean =false;
  
  constructor(
    private handlerError:CustomErrorHandler, 
    private navigator:NavigationProvider,
    private commonService:CommonProvider,
    public dialog:MatDialog,
    private router:Router,private authProvide:AuthProvider, ){

    }

    @HostListener('window:resize', ['$event'])
    onResize(event) {
      this.checkDisplay();
    }

    checkDisplay(){
      if(window.innerWidth < 768){
        this.display_responsive  = true;
      }
      if(window.innerWidth > 769){
        this.display_responsive  = false;
      }
    }
    
    
    ngOnInit(){
      this.checkCookies();
      this.checkDisplay();

      this.commonService.userValidator.subscribe((validator)=>{
        this.validator = validator;
      })
  
      this.routes = this.router.config;
  
      this.router.events.subscribe( async (event: Event) => {
        
        let attempt = localStorage.getItem('attemptUrl');

        if(event instanceof NavigationCancel){
        }  
        
        
        if(event instanceof NavigationStart){
            if((event.url.includes('confirm-register'))){
              var confirm_email_token = event.url.split('/')[2]
              localStorage.setItem('confirm_email_token',confirm_email_token);
            }else{
              let clean_url = event .url.split("?")[0];
              clean_url = clean_url.replace('/','');
              var route = this.routes.find((route => route.path.split('/')[0] == clean_url.split('/')[0]));
              var token = localStorage.getItem('auth_token');
              
              if(!route && !token){
                window.localStorage.setItem('previousUrl', '/'+clean_url);
                this.navigator._land();
              }
              if(!route && token && event.url.includes('optimization-restuls')){
                window.localStorage.setItem('previousUrl', '/'+clean_url);
                this.router.navigate(['/home'],{replaceUrl:true})
              }
              if((event.url.includes('recover-password')) && (event.url.split('/')[2])){
                this.navigator._goNewPassword((event.url.split('/')[2]))
              }
              if(event.url.includes('landing')){
                this.outer_header = true;
              }
              if(event.url.includes('validator') && (localStorage.getItem('user_validator') == 'true')){
                this.validator = true;
              }
              if(event.url.includes('invoice')){
                this.invoice = true;
              }
              
              if(route && !token && event.url != '' && event.url != '/' && event.url != '/landing' && event.url != '/login' && event.url != '/register' && (!(event.url.includes('new-password'))) && 
               event.url != '/recover-password' && event.url != '/new-password' && event.url != '/product' && event.url != '/contact' 
               && event.url != '/faqs' && (event.url.split('?')[0] != '/confirm-register') && (event.url.split('?')[0] != '/chart-playground')){
                 localStorage.setItem('attemptUrl',event.url.split('?')[0]);
                 this.navigator._goLogin();
              }
            }
          }else{
            if (event instanceof NavigationEnd) {
              let clean_url = event .url.split("?")[0];
              clean_url = clean_url.replace('/','');
              var route = this.routes.find((route => route.path == clean_url));
              var token = localStorage.getItem('auth_token');
              var tokens = localStorage.getItem('auths_token');
              if((event.url.includes('recover-password')) && (event.url.split('/')[2])){
              }else{
                
                if(!token && event.url != '/' && event.url != '/login' && event.url != '/register' && (!(event.url.includes('new-password')))
                && event.url != '/recover-password' && event.url != '/new-password' && event.url != '/product' && event.url != '/contact' && event.url != '/faqs' && (event.url.split('?')[0] != '/confirm-register') && (event.url.split('?')[0] != '/chart-playground')){
                  this.navigator._land();
                  this.validator = false;
                  this.outer_header = true;
                }
                if(token && (event.url == "/login" || event.url == "/register" || event.url == "/recover-password" || event.url == '/product' || event.url == '/contact' || event.url == '/faqs' || event.url == "/new-password")){
                  this.router.navigate(["/home"],{replaceUrl:true});
                }
                // Hide loading indicator
                event.url = event.url.replace('/','');
                this.link_active =  event.url;
                if(((event.url.includes('validator')) || (event.url.includes('report-review'))) && (localStorage.getItem('user_validator') == 'true') ){
                  this.validator = true;
                }
                if(event.url.includes('chart-playground')){
                  this.outer_header = false;
                  this.validator = false;
                  this.random = true;
                };
                if((event.url == 'register') || (event.url == 'login') || (event.url == 'product') || (event.url == 'contact') || (event.url == 'faqs') || (event.url == 'recover-password') || (event.url == '') || ((event.url.includes('new-password')))
                || (event.url == 'new-password') || (event.url == 'landing') || (event.url.split('?')[0] == 'confirm-register')){
                  this.outer_header = true;
                }else{
                  this.outer_header = false;
                }
                let clean_url = event .url.split("?")[0];
                clean_url = clean_url.replace('/','');
                var route = this.routes.find((route => route.path == clean_url));
                var token = localStorage.getItem('auth_token');
                if(route){
                  window.localStorage.setItem('previousUrl', '/'+clean_url);
                }
                if(!token && event.url == ""){
                  this.navigator._land();
                  this.validator = false;
                  this.outer_header = true;
                }else if(token && event.url == ""){
                  this.router.navigate(['/home'],{replaceUrl:true});
                  this.outer_header = false
                }
                if(!tokens && event.url == "/instrument-list"){
                  this.router.navigate(["/loginsonicu"],{replaceUrl:true});
                }else if(tokens && event.url == "/instrument-list"){
                  this.router.navigate(['/instrument-list'],{replaceUrl:true});
                  this.outer_header = false
    
                }
    
              }
            }
          }
  
          // if(event instanceof NavigationEnd){
          //   let clean_url = event .url.split("?")[0];
          //   let attempt = localStorage.getItem('attemptUrl');
          //   clean_url = clean_url.replace('/','');
          //   var route = this.routes.find((route => route.path == clean_url));

          //      event.url != 'recover-password' && event.url != 'new-password' && event.url != 'product' && event.url != 'contact' 
          //      && event.url != 'faqs' && (event.url.split('?')[0] != 'confirm-register') && (event.url.split('?')[0] != 'chart-playground')){
          //        this.navigator._goLogin();
          //   }
          // }

          
  
          if (event instanceof NavigationError) {
              // Hide loading indicator
  
              // Present error to user
          }
        })
      
  }

  checkCookies(){
    if(environment.production){
      let cc = window as any;
      cc.cookieconsent.initialise({
        palette: {
          popup: {
            background: "#32467A"
          },
          button: {
            background: "#199AD8",
            text: "#fff"
          },
        },
        theme: "classic",
        content: {
          header: 'Cookies used on the website!',
          message: 'This website uses cookies to improve your experience.',
          dismiss: 'Ok. Got it!',
          allow: 'Allow cookies',
          deny: '<a aria-label="deny cookies" tabindex="0" class="cc-btn cc-deny"></a>',
          link: '',
          href: '',
          target: '_blank',
          },
      });     
    }
  };

}
