import { Component, OnInit, Output, EventEmitter, HostListener, ViewChild, ElementRef } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { SupplierUnitsProvider } from 'src/app/providers/supplier-units.provider';
import { SettingsProvider } from 'src/app/providers/settings.service';
import { MatSnackBar, MatDialog } from '@angular/material';
import { LeaveDialogComponent } from 'src/app/components/modal-dialogs/leave-dialog/leave-dialog.component';
import { CommonProvider } from 'src/app/providers/common.service';
import { FeedbackDialogComponent } from 'src/app/components/modal-dialogs/feedback-dialog/feedback-dialog/feedback-dialog.component';



@Component({
  selector: 'app-profile-page',
  templateUrl: './profile-page.component.html',
  styleUrls: ['./profile-page.component.scss']
})
export class ProfilePageComponent implements OnInit {
  @ViewChild('section') section: ElementRef;
  @Output() update = new EventEmitter<boolean>();
  public profile_option:any=[false,false,false,false,false,false];
  public user_data:any={};
  public original_data:any={};
  public metrics_units;
  public resize:boolean=false;
  public isImgLoaded:any;
  public confirm_profile_image:boolean=false;
  public profile_file:any;
  public profile_image: any;
  public selectedIndex:any = 0;
  public personal_data;
  public snack_message: string;
  public unit_system: string;
  public data_change: any;
  public section_to_go: string = null;
  public credits: any;
  public edit_data:boolean = false;
  constructor(
    private handlerError:CustomErrorHandler, 
    private unitsProvider: SupplierUnitsProvider,
    private commonProvider:CommonProvider,
    private _snackBar: MatSnackBar , 
    public dialog: MatDialog,
    private settingsProvider:SettingsProvider, 
    private authProvider:AuthProvider) { }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.setProfileColumns(window.outerWidth);
  } 

  ngOnInit() {
    this.unitsProvider.getUnitsActive().then(metrics => {
    this.metrics_units = metrics;
    this.setProfileColumns(window.outerWidth);
  });

  this.section_to_go = localStorage.getItem('profile_section');
  
  if(this.section_to_go){
    localStorage.removeItem('profile_section');
  }
  this.getCredits();
  this._getUserInformation();
  }

  setProfileColumns(width){
    width < 1025 ? this.resize = true : this.resize = false;
  }

  _openProfileSection(section){
    if(this.resize){
      this.scroll();
    }
    if(this.data_change && (this.profile_option[0] == 1 || this.profile_option[5] == 1)){
      const dialogRef = this.dialog.open(LeaveDialogComponent, {
        width: '50%',
        height: '30%',
      });
      dialogRef.afterClosed().subscribe((result)=>{
        if(result){
          this.changeSection(section);
        }
      }); 
    }else{
      this.changeSection(section);
    }
  }

  changeSection(section){
    this.data_change = false;
    this.profile_option = [0,0,0,0,0,0];
    this.profile_option[section]=1; 
  }

  setDataChanges(ev){
    this.data_change = ev;
  }

  _update(){
    this._getUserInformation();
  }

  _changeMetricsUnits(e) {
    if(e.index == 1){
      ///////GUARDANDO EN DATA DEL USUARIO LA UNIDAD
      const data = {
        first_name: this.user_data.first_name,
        last_name: this.user_data.last_name,
        email: this.user_data.email,
        birth_date: this.user_data.birth_date,
        gender: this.user_data.gender,
        phone: this.user_data.phone,
        email_was_confirm: this.user_data.email_was_confirm,
        token: this.user_data.token,
        unit_system:'m',
      }
      this.settingsProvider.patchUserProfileUnit(data).subscribe((response)=>{
        this.refreshPersonalData();
      },error=>{
        this.handlerError.handleError(error); 
      });

      ///////GUARDANDO EN STORAGE
      localStorage.setItem('metrics_units','1');
    } else {
      const data = {
        first_name: this.user_data.first_name,
        last_name: this.user_data.last_name,
        email: this.user_data.email,
        birth_date: this.user_data.birth_date,
        gender: this.user_data.gender,
        phone: this.user_data.phone,
        email_was_confirm: this.user_data.email_was_confirm,
        token: this.user_data.token,
        unit_system:'i',
      }
      this.settingsProvider.patchUserProfileUnit(data).subscribe((response)=>{
        this.refreshPersonalData();
      },error=>{
        this.handlerError.handleError(error); 
      });

      localStorage.removeItem('metrics_units');
    }
  }

  refreshPersonalData(){
    this.update.emit(true);
  }

  _getUserInformation(){
    this.authProvider.getUserInfo().subscribe((response)=>{
      this.user_data = response;
      if (!this.user_data.zip_code) this.user_data.zip_code = 'Not specified';
      this.original_data = {...response};
      if(!this.section_to_go){
        this._openProfileSection(0);
      }else{
        this._openProfileSection(parseInt(this.section_to_go));
      }
      
    },error=>{  
      this.handlerError.handleError(error)
    })
  }

  getCredits(){
    this.authProvider.getUserCredits().subscribe((credits:any)=>{
      this.credits = {...credits};
    })
  }

  showModalFeedback(){
    this.dialog.open(FeedbackDialogComponent, {
      width: '70%',
      height: 'auto',
      id:'feedback',
      data:this.user_data
    });
  }

  fileSelectedChange(ev){
    this.profile_file = ev.target.files[0];
    const reader = new FileReader();
    if(ev.target.files[0].type !== 'image/png' && ev.target.files[0].type !== 'image/jpeg' &&  ev.target.files[0].type !== 'image/jpg' &&  ev.target.files[0].type !== 'image/gif'){
      this.commonProvider.showSnackBar('Image error','Unsupported image file','alert-snackbar');
    }else if(ev.target.files[0].size > 200000){
      this.commonProvider.showSnackBar('Image error','Your image file is to heavy. Try to upload files with 2MB max','alert-snackbar');
    }else{  
      reader.readAsDataURL(ev.target.files[0]);
      reader.onload = () => {
        this.profile_image = reader.result as string;
        this.confirm_profile_image = true;
      };
    }
  }

  tabChanged(event){
    this.selectedIndex = event.index;
  }

  goToTab(tab){
    this.selectedIndex = tab;
  }

  savePersonalInfo(){
    if(this.profile_file){
      this.settingsProvider.updateUserProfile(this.profile_file,this.user_data).subscribe((response)=>{
        this.snack_message = "Your profile has been updated"
        this.showSnackNotification(5000);
        this.edit_data = false;
        this.commonProvider.updateUserData(response);
        this.refreshPersonalData();
      },error=>{
        this.handlerError.handleError(error); 
      });
    }else{
      this.settingsProvider.patchUserProfile(this.user_data).subscribe((response)=>{
        this.snack_message = "Your profile has been updated"
        this.showSnackNotification(5000);
        this.edit_data = false;
        this.commonProvider.updateUserData(response);
        this.refreshPersonalData();
      },error=>{
        this.handlerError.handleError(error); 
      });
    }
  }

  showSnackNotification(time) {
    this.commonProvider.showSnackBar('Message',this.snack_message,'alert-snackbar');
  }

  cancelEdit(){
    this.edit_data = false;
    this.user_data = {...this.original_data};
  }

  scroll() {
    setTimeout(() => {
      document.getElementById('section').scrollIntoView({behavior: 'smooth'});
    }, 300);
  }

  async canExit() : Promise<boolean>  {
    var leave = true;
    if(this.profile_option[0] == 1 || this.profile_option[5] == 1){
      if(this.data_change){
        const dialogRef = this.dialog.open(LeaveDialogComponent, {
          width: '50%',
          height: '30%',
        });
    
        leave = await dialogRef.afterClosed().toPromise();
      }
    }    
    return leave
  }

  

}
