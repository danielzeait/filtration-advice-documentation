import { Component, OnInit, Output, Input, EventEmitter, Inject, ChangeDetectorRef, ViewChild } from '@angular/core';

import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { Report } from 'src/app/models/report.model';

import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { System } from 'src/app/models/system.model';

import { ActivatedRoute } from '@angular/router';

import { CustomErrorHandler } from 'src/app/providers/error.service';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-users-shared-reports',
  templateUrl: './users-shared-reports.component.html',
  styleUrls: ['./users-shared-reports.component.scss']
})
export class UserSharedReportsComponent implements OnInit {
  @Input('data') reports : any = {authorize_users:[]};
  public server_name: string = global.SERVER_NAME;
  public system_unshared: any;
  public users: any;

  constructor(
    private handlerError:CustomErrorHandler,
    public dialogRef: MatDialogRef<UserSharedReportsComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private commonProvider:CommonProvider,
    private sharedProvider: SharedReportProvider,
    ) { }

  ngOnInit() {
    this._getUserSharedReport(this.data);
  }

  _getUserSharedReport(data) {
    this.sharedProvider.getUserSharedReports(data.id).subscribe((response)=>{
      this.reports = response;
      },error=>{
        this.handlerError.handleError(error);
      })
  }

  _closeDialog(){
    this.dialogRef.close(false);
  }

  unshared(data,users) {
    const unsharedData = {
      id: this.data.id,
      email: users[0],
    }
    this.sharedProvider.deleteUserSharedReport(unsharedData).subscribe((response)=>{
      this.commonProvider.showSnackBar('Permission revoked.','The permission was revoked to the user' ,'success-snackbar');
      this.dialogRef.close(true);
    },error=>{
      this.handlerError.handleError(error);
      this.commonProvider.showSnackBar('Rovoked error.','We were unable to revoke the permission to use this report' ,'alert-snackbar');
    })

  }






}
