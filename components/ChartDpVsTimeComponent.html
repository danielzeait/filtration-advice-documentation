import { Component, OnInit, Input, SimpleChanges, AfterViewInit } from '@angular/core';
import Chart from 'chart.js';
import { timeStamp } from 'console';
@Component({
  selector: 'app-chart-dp-vs-time',
  templateUrl: './chart-dp-vs-time.component.html',
  styleUrls: ['./chart-dp-vs-time.component.scss']
})
export class ChartDpVsTimeComponent implements AfterViewInit {
  @Input('chart_data') chart_data : any;  
  @Input('limit_dp') limit_dp : any;
  @Input('chart_id') chart_id : any;
  @Input('selected_values') selected_values : any; 
  @Input('chart_data_optimized') chart_data_optimized : any;
  @Input('top_current_dp') top_current_dp : any; 
  @Input('top_optimized_dp') top_optimized_dp : any;
  @Input('top_selected_dp') top_selected_dp : any;
  @Input('chart_x_type') chart_type : any;
  @Input('chart_data_selected') chart_data_selected : any;
  
  private already_fixed_x_axis:boolean=false;
  public max_x:any;
  private chart: any;
  private chartData:any=[];
  private chartDataOptimized:any=[];
  private chartDataSelected: any=[];
  private variableData:any={};
  private verticalLimitChartData:any=[]
  private datas: { dp_value: any; name: string; data: any; label: string; color: string; }[];
  
  constructor() { 
  }
  
  ngAfterViewInit() {

    if(this.chart_data.length > 0){
      var x2 = this.chart_data[this.chart_data.length-1].time;
      //var x2 = this.chart_type == 'dp' ? this.chart_data[this.chart_data.length-1].time : this.chart_data[this.chart_data.length-1].dust;
      this.verticalLimitChartData = [
        {x:0,y:this.limit_dp},
        {x:Math.ceil(x2),y:this.limit_dp}
      ]
      this.chartData = this.setCurrentChartData(this.chart_type);
    }
    if(this.chart_data_optimized.length > 0){
      this.chartDataOptimized = this.setOptimizedChartData(this.chart_type);
    }
     
    if(this.selected_values){
      this.variableData = [{
        'x':parseFloat(this.selected_values.time),
        'y':parseFloat(this.selected_values.drop_pressure)
      }]
    }
    this.orderCharts();
    var all_data = this.chart_data.concat(this.chart_data_optimized);
    this.initChart(all_data);
  }

  ngOnChanges(changes: SimpleChanges){
    var update = false
    
    if(changes.chart_type && !changes.chart_type.firstChange){
      this.chart_type = changes.chart_type.currentValue;
      this.chart.scales.xAxis2.hidden = this.chart_type == 'dust' ? false : true;
      this.chart.scales.xAxis2.options.display = this.chart_type == 'dust' ? true : false;
      // if(this.chart_data_selected){
      //   this.chartDataSelected = this.setSelectedChart(this.chart_type);
      //   this.chart.data.datasets.find(data => data.label == "Selected").data = this.chartDataSelected;
      // }
      // this.chartData = this.setCurrentChartData(this.chart_type);
      // this.chart.data.datasets.find(data => data.label == "Current").data = this.chartData;
      // this.chartDataOptimized = this.setOptimizedChartData(this.chart_type);
      // this.chart.data.datasets.find(data => data.label == "Optimized").data = this.chartDataOptimized;

      //var x2 = this.chart_type == 'dp' ? this.chart_data[this.chart_data.length-1].time : this.chart_data[this.chart_data.length-1].dust;
      // var x2 = this.chart_data[this.chart_data.length-1].time;
      // this.verticalLimitChartData = [
      //   {x:0,y:this.limit_dp},
      //   {x:Math.ceil(x2),y:this.limit_dp}
      // ]
      // this.chart.data.datasets[0].data = this.verticalLimitChartData;
      //this.chart.scales['x-axis-0'].options.ticks.max = this.chart_type == 'dp' ? this.chart_data[this.chart_data.length-1].time : this.chart_data[this.chart_data.length-1].dust
      //this.chart.scales['x-axis-0'].options.scaleLabel.labelString = this.chart_type == 'dp' ?  'Time of Change in Months' : 'Dust Fed (in grams)';
      update = true;
    }
    
    if(changes.chart_data_selected && changes.chart_data_selected.currentValue){
      this.chart_data_selected = changes.chart_data_selected.currentValue;
      this.chartDataSelected = this.setSelectedChart(this.chart_type);
      this.chart.data.datasets.find(data => data.label == "Selected").data = this.chartDataSelected;
      this.chart.options.scales.xAxes[0].ticks.max = this.chartDataSelected[this.chartDataSelected.length - 1].x;
      this.chart.options.scales.xAxes[1].ticks.max = this.chartDataSelected[this.chartDataSelected.length - 1].z;
      update = true; 
    }
    
    if(changes.chart_data && !changes.chart_data.firstChange){
      this.chart_data = changes.chart_data.currentValue;
      this.chartData = this.setCurrentChartData(this.chart_type);
      this.chart.data.datasets.find(data => data.label == "Current").data = this.chartData;
      update = true; 
    }
    
    if(changes.chart_data_optimized && !changes.chart_data_optimized.firstChange){
      this.chart_data_optimized = changes.chart_data_optimized.currentValue;
      this.chartDataOptimized = this.setOptimizedChartData(this.chart_type);
      this.chart.data.datasets.find(data => data.label == "Optimized").data = this.chartDataOptimized;
      update = true; 
    }
    
    
    if(this.chart_data_selected){
      //var x2 = this.chart_type == 'dp' ? this.chart_data_selected[this.chart_data_selected.length-1].time : this.chart_data_selected[this.chart_data_selected.length-1].dust;
      var x2 = this.chart_data_selected[this.chart_data_selected.length-1].time
      this.verticalLimitChartData = [
        {x:0,y:this.limit_dp},
        {x:Math.ceil(x2),y:this.limit_dp}
      ]
      this.chart.data.datasets[0].data = this.verticalLimitChartData;
    }
    
    if(update){
      this.orderCharts();
      this.chart.update();
    }
  }

  orderCharts(){
    this.datas = [
      { 
        dp_value:this.top_current_dp,
        name:'current',
        data:this.chartData,
        label:"Current",
        color:'#003A70',
      },
      {
        dp_value:this.top_optimized_dp,
        name:'optimized',
        data:this.chartDataOptimized,
        label:"Optimized",
        color:'#199AD8',
      },
      {
        dp_value:this.top_selected_dp,
        name:'selected',
        data:this.chartDataSelected,
        label:"Selected",
        color:'#c7d42d',
      }
    ]; 
    this.datas.sort(function(a, b) {  return a.dp_value - b.dp_value; });

    if(this.chart){
      this.chart.data.datasets[1].data = this.datas[0].data;
      this.chart.data.datasets[2].data = this.datas[1].data;
      this.chart.data.datasets[3].data = this.datas[2].data;

      this.chart.data.datasets[1].label = this.datas[0].label;
      this.chart.data.datasets[2].label = this.datas[1].label;
      this.chart.data.datasets[3].label = this.datas[2].label;

      this.chart.data.datasets[1].backgroundColor = this.datas[0].color;
      this.chart.data.datasets[2].backgroundColor = this.datas[1].color;
      this.chart.data.datasets[3].backgroundColor = this.datas[2].color;

    };
  }

  setSelectedChart(type){
    var chart_selected = [];
    for (let i = 0; i < this.chart_data_selected.length; i++) {
      var point = {
        'x':Number.parseFloat(this.chart_data_selected[i].time),
        'y':parseFloat(this.chart_data_selected[i].drop_pressure),
        'z':Number.parseFloat(this.chart_data_selected[i].dust)
      }
      chart_selected.push(point);
    }
    return chart_selected;
  }

  setCurrentChartData(type){
    var chart_current = [];
    for (let i = 0; i < this.chart_data.length; i++) {
      var point = {
        'x':Number.parseFloat(this.chart_data[i].time),
        'y':parseFloat(this.chart_data[i].drop_pressure),
        'z':Number.parseFloat(this.chart_data[i].dust)
      }
      chart_current.push(point); 
    }
    //this.max_x = type == 'dp' ? this.chart_data[this.chart_data.length-1].time : this.chart_data[this.chart_data.length-1].dust
    return chart_current
  }

  setOptimizedChartData(type){
    var chart_optimized = [];
    for (let i = 0; i < this.chart_data_optimized.length; i++) {
      var point = {
        'x':Number.parseFloat(this.chart_data_optimized[i].time),
        'y':parseFloat(this.chart_data_optimized[i].drop_pressure),
        'z':Number.parseFloat(this.chart_data_optimized[i].dust)

      }
      chart_optimized.push(point);
    }
    return chart_optimized
  }

  updateChartTicks(max,step) {
    setTimeout(() => {
      if(!this.already_fixed_x_axis){
        this.already_fixed_x_axis = true;
        this.chart.options.scales.xAxes[1].ticks.min = 0;
        this.chart.options.scales.xAxes[1].ticks.max = max;
        this.chart.options.scales.xAxes[1].ticks.stepSize = step;
        this.chart.update();
      }
    }, 250); 
  };

  initChart(global_data){
    var id = this.chart_id;
    var ctx:HTMLElement = <HTMLElement> document.getElementById(id);
    ctx.style.backgroundColor = '#FFF';
    ctx.style.padding = '10px';

    var canvas : any = document.getElementById(id);
    var ctx2 = canvas.getContext("2d");
    
    Chart.Legend.prototype.afterFit = function() {
      this.height = this.height + 10;
    };
    
    this.chart = new Chart(ctx2, {
        type: 'line',
        pointBackgroundColor: "#55bae7",
        pointBorderColor: "#55bae7",
        data: {
            datasets: [
              {
                label:'0',
                data:this.verticalLimitChartData,
                pointRadius:'0',
                borderColor:'#ffa629',
                borderWidth:3,
                fill:false,
                backgroundColor:'#f3f3f3'
              },
              {
                label:this.datas[0].label,
                data: this.datas[0].data,
                backgroundColor:this.datas[0].color,
                borderWidth:1,
                lineTension:0,
                borderColor:'#1C036F',
                pointBorderColor: "#222",
                pointRadius:'0',
                clip:-1,
              },
              {
                label:this.datas[1].label,
                data: this.datas[1].data,
                backgroundColor:this.datas[1].color,
                borderWidth:1,
                lineTension:0,
                borderColor:'#1C036F',
                pointBorderColor: "#222",
                pointRadius:'0',
                clip:-1,
              },
              {
                label:this.datas[2].label,
                data: this.datas[2].data,
                backgroundColor:this.datas[2].color,
                borderWidth:1,
                lineTension:0,
                borderColor:'#1C036F',
                pointBorderColor: "#222",
                pointRadius:'0',
                clip:-1,
              },
            ]
        },
        options: {
          legend: {
            align:'start',
            labels: {
              filter: function(item, chart) {
                // Logic to remove a particular legend item goes here
                return !item.text.includes('0');
              },
              fontFamily:'lato',
              fontSize:11.5
            },
          },
          elements: {
            line: {
              tension: 0.4
            }
          },
          tooltips:{
            callbacks: {
              title:function(tooltipItem) {
                var label = 'Dp value: ' + Number.parseFloat(tooltipItem[0].yLabel).toFixed(3);
                return label;
              },
              label: function(tooltipItem){
                var value = Number.parseFloat(tooltipItem.xLabel).toFixed(2);
                return 'Time: '  + value + ' months';
              },
              afterBody: (tooltipItem,data) => {
                var value = '';
                if(this.chart_type == 'dust'){                  
                  value = data.datasets[tooltipItem[0].datasetIndex].data.find(point => point.x == tooltipItem[0].xLabel).z.toFixed(2);
                  value =  'Dust: ' + value + ' grams'
                }
                return value;
              }
            }
            // callbacks:{
            //     label:function(tooltipItem) {
            //         var label = 'Dp value: ' + Number.parseFloat(tooltipItem.yLabel).toFixed(3);
            //         return label;
            //     },
            //     title: function(tooltipItem){
            //       console.log(tooltipItem);
            //       var value = Number.parseFloat(tooltipItem[0].xLabel).toFixed(2);
            //       var value2 = Number.parseFloat(tooltipItem[0].x).toFixed(2);
            //       return 'Time: '  + value + ' /n' + 'Dust: ' + value2;
            //     }
            // },
          },
          scales: {
              xAxes: [
                {
                  id:'xAxis1',
                  type: 'linear',
                  position: 'bottom',
                  scaleLabel: {
                    display: true,
                    labelString:'Time of Change in Months',
                    fontColor:'#222',
                    fontSize:10,
                  },
                  ticks: {
                    //stepSize: 2,
                    fontSize:10,
                    callback: function(value, index, values) {
                      return Number.parseFloat(value).toFixed(1);
                    },
                    max:this.chart_data[this.chart_data.length - 1].time
                  },
                },
                {
                  id:'xAxis2',
                  position: 'bottom',
                  type:"linear",
                  scaleLabel: {
                    display: true,
                    labelString:'Dust Fed (in grams)',
                    fontColor:'#222',
                    fontSize:10,
                  },
                  ticks: {
                    //stepSize: 2,
                    fontSize:10,
                    callback: function(value, index, values) {
                      return Number.parseFloat(value).toFixed(1);
                    },
                    //max:this.chart_data[this.chart_data.length - 1].dust
                  },
                  afterBuildTicks: (scale) => {
                    var val_1 = Math.ceil((this.chart_data[this.chart_data.length-1]).dust) / 5;
                    //var val_2 = Math.floor(Math.abs((this.chart_data[this.chart_data.length-1]) - (this.chart_data[this.chart_data.length-3]).dust))
                    let new_scale = [0];
                    for (let index = 0; index < 5; index++) {
                      new_scale.push(new_scale[index] + val_1);
                    }
                    this.updateChartTicks(new_scale[new_scale.length-1],val_1);
                    return;
                  },
                },
              ],
              yAxes: [{
                scaleLabel: {
                  display: true,
                  labelString: 'Pressure Drop (inch of WC)',
                  fontColor:'#222',
                  fontSize:10,
                },
                ticks: {
                  beginAtZero: true,
                  fontSize:10,
                  //stepSize:0.5
                  callback: function(value, index, values) {
                    return Number.parseFloat(value).toFixed(2);
                  }
                },
              },
            ],
          }
        }
    });
    
    

  }

}
