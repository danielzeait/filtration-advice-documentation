import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { MatSnackBar } from '@angular/material';
import { SettingsProvider } from 'src/app/providers/settings.service';
import { Company } from 'src/app/models/company.model';
import { User } from 'src/app/models/user_data.model';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-personal-information',
  templateUrl: './personal-information.component.html',
  styleUrls: ['./personal-information.component.scss']
})
export class PersonalInformationComponent implements OnInit {
  logo_file: any;
  logo_image: any;
  confirm_logo: boolean;
  error_title: string;
  snack_message: string;
  error_file: boolean;
  personal_data:any;
  @Input('user_data') user_data:any;
  @Output() show_pdf = new EventEmitter<boolean>();
  @Output() update = new EventEmitter<boolean>();
  @Output() data_change = new EventEmitter<boolean>();
  constructor(
    private commonProvider:CommonProvider,
    private handlerError:CustomErrorHandler,
    private _snackBar: MatSnackBar , private settingsProvider:SettingsProvider, private utils:UtilsFunctions) { }

  ngOnInit() {
    this.personal_data = this.user_data;
    this.personal_data.gender = "N";
    this.personal_data.birth_date = this.utils.formatDate(new Date());
    if(this.personal_data.profile_image){
      this.confirm_logo = true;
      this.logo_image = this.personal_data.profile_image
    }
  }

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      var type = files[0].file.type;
      if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
        this.setErrorFile();
      } else {
        this.logo_file = files[0].file;
        this.logo_image = files[0].imgURL;
        this.confirm_logo = true;
        this.data_change.emit(true);
      }
    }
  }

  fileSelectedChange(ev) {
    var type = ev.target.files[0].type
    if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
      this.setErrorFile();
    } else {
      this.logo_file = ev.target.files[0];
      const reader = new FileReader();
      reader.readAsDataURL(ev.target.files[0]);
      reader.onload = () => {
        this.logo_image = reader.result as string;
        this.confirm_logo = true;
      };
      this.data_change.emit(true);
    }
  }

  setErrorFile() {
    this.error_title = "File not supported"
    this.snack_message = "Make sure that the file you are trying to upload is an image"
    this.error_file = true;
    this.showSnackNotification(5000);
  }
  
  showSnackNotification(time) {
    this.commonProvider.showSnackBar('Message',this.snack_message,'alert-snackbar');
  }

  savePersonalInfo(){
    if(this.logo_file){
      this.settingsProvider.updateUserProfile(this.logo_file,this.personal_data).subscribe((response)=>{
        this.snack_message = "Your profile has been updated"
        this.showSnackNotification(5000);
        this.refreshPersonalData();
        this.data_change.emit(false);
      },error=>{
        this.handlerError.handleError(error); 
      });
    }else{
      this.settingsProvider.patchUserProfile(this.personal_data).subscribe((response)=>{
        this.snack_message = "Your profile has been updated"
        this.showSnackNotification(5000);
        this.refreshPersonalData();
        this.data_change.emit(false)
      },error=>{
        this.handlerError.handleError(error); 
      });
    }
  }

  refreshPersonalData(){
    this.update.emit(true);
  }

  emitChange(){
    this.data_change.emit(true)
  }

  _show_pdf(){
    this.show_pdf.emit(true);
  }

}
