import { Component, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { CustomErrorHandler } from 'src/app/providers/error.service';

@Component({
  selector: 'app-new-password',
  templateUrl: './new-password.component.html',
  styleUrls: ['./new-password.component.scss']
})

export class NewPasswordComponent implements OnInit {
  public confirmed: boolean = false;
  public confirm_password = new FormControl('', [Validators.required]);
  public hide: boolean = true;
  public hide_confirm: boolean = true;
  public newData: any = {
    password: '',
    confirm_password: ''
  }
  public password = new FormControl('', [Validators.required]);
  
  constructor(
    private authProvider: AuthProvider, 
    private handlerError:CustomErrorHandler, 
    private navigator: NavigationProvider,
    private routeActivated: ActivatedRoute 
  ) { }

  ngOnInit() {
     this.routeActivated.queryParams.subscribe(async params => {
      this.newData.token = params.token;
    })
  }

  _confimrNewPassword() {
    this.authProvider.resetUserPassword(this.newData).subscribe((response) => {
      this.confirmed = true;
    }, error => {
      this.handlerError.handleError(error);
    })
  }

  _continue() {
    this.navigator._goLogin();
  }

  getMissMatchPasswordErrorMessage() {
    return "Confirm password doesn't match"
  }

}
