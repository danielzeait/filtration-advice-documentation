import { Component, OnInit, Output, EventEmitter } from '@angular/core';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { MatSnackBar } from '@angular/material';
import { MatDialog } from '@angular/material/dialog';
import { Company } from 'src/app/models/company.model';
import { SettingsProvider } from 'src/app/providers/settings.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { CommonProvider } from 'src/app/providers/common.service';
import { PdfHelperDialogComponent } from 'src/app/components/modal-dialogs/pdf-help/pdf-helper-dialog/pdf-helper-dialog.component';

@Component({
  selector: 'app-company',
  templateUrl: './company.component.html',
  styleUrls: ['./company.component.scss']
})
export class CompanyComponent implements OnInit {
  error_file: boolean;
  error_title: string;
  snack_message: string;
  confirm_logo:boolean=false;
  logo_file:any;
  logo_image: any;
  edit_data:boolean = false;
  public is_new_company:boolean;
  @Output() show_profile = new EventEmitter<boolean>();
  @Output() data_change = new EventEmitter<boolean>();
  constructor(
    private commonProvider:CommonProvider,
    private handlerError:CustomErrorHandler,
    private dialog:MatDialog,
    private _snackBar: MatSnackBar , private settingsProvider:SettingsProvider) { }
  company_data:any=new Company();
  original_company_data:any = new Company();
  ngOnInit() {
    this.settingsProvider.getPdfSettings().subscribe((response)=>{
      if(response.name !== ""){
        this.is_new_company = false;
        this.company_data = response;
        this.original_company_data = {...response};
        if(this.company_data.logo !== "" && this.company_data.logo){
          this.confirm_logo = true;
          this.logo_image = this.company_data.logo;
        }
      }else{
        this.is_new_company = true;
      }
      if (!this.company_data.zip_code) this.company_data.zip_code = 'Not specified';
    })
  }

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      var type = files[0].file.type;
      if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
        this.setErrorFile();
      } else {
        this.logo_file = files[0].file;
        this.logo_image = files[0].imgURL;
        this.confirm_logo = true;
        this.data_change.emit(true)
      }
    }
  }

  fileSelectedChange(ev) {
    var type = ev.target.files[0].type;
    var size = ev.target.files[0].size;
    if ((type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") || (size > 200000)) {
      this.setErrorFile();
    } else {
      this.logo_file = ev.target.files[0];
      const reader = new FileReader();
      reader.readAsDataURL(ev.target.files[0]);
      reader.onload = () => {
        this.logo_image = reader.result as string;
        this.confirm_logo = true;
      };
      this.data_change.emit(true)
    }
  }

  setErrorFile() {
    this.error_title = "File not supported"
    this.snack_message = "Make sure that the file you are trying to upload is an image and the size in bellow 2MB"
    this.error_file = true;
    this.showSnackNotification(5000);
  }
  
  showSnackNotification(time) {
    this.commonProvider.showSnackBar('Message',this.snack_message,'alert-snackbar');
  }

  saveCompanyInfo(){
    this.edit_data = false;
    if(this.logo_file){
      this.settingsProvider.createUserSettings(this.logo_file,this.company_data).subscribe((response)=>{
        this.snack_message = "Your PDF settings has been updated";
        this.showSnackNotification(5000);
        this.data_change.emit(false)
      },error=>{
        this.handlerError.handleError(error);
      });
    }else{
      delete this.company_data.logo;
      this.settingsProvider.updatePdfSettings(this.company_data).subscribe((response)=>{
        this.snack_message = "Your PDF settings has been updated";
        this.showSnackNotification(5000);
        this.data_change.emit(false)
      },error=>{
        this.handlerError.handleError(error);
      });
    }
  }

  cancelEdit(){
    this.edit_data = false;
    this.company_data = {...this.original_company_data};
  }

  showPDFHelper(){
    this.dialog.open(PdfHelperDialogComponent, {
      width: '70%',
      height: 'auto',
    });
  }

  _show_profile(){
    this.show_profile.emit(true);
  }

  emitChange(){
    this.data_change.emit(true);
  }

}
