/**
 * Empty component for list of cases

@Component({
  selector: 'app-case-list',
  templateUrl: './case-list.component.html',
  styleUrls: ['./case-list.component.scss']
})

export class CaseListComponent implements OnInit {
  @ViewChild(MatPaginator) paginator: MatPaginator;
  public bottom_paginator_data: any = {}
  public case_data: System = null;
  public cases: Array<any> = new Array<any>();
  public innerWidth: any;
  public isImgLoaded: any;
  public isSearching: boolean = false;
  public jumpToPageOption: any;
  public length: number;
  public list_url: any = global.API_CASES;
  public original_cases: Array<any> = new Array<any>();
  public num_cols: number = 4;
  public numPages: any;
  public pageEvent: PageEvent;
  public pageSize = 12;
  public pageSizeOptions: number[] = [5, 10, 12, 15, 20];
  public search: string = "";
  public searchedWordsControl = new FormControl('');
  public server_name: string = global.SERVER_NAME;
  public hide_check:boolean = false;
  public display_value:any='all';
  public empty_refine:any=false;
  public order_date:any='-updated';

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
    )
  optimization_plan: any;
  case_list_width: any = window.outerWidth < 1025 ? '69%' : '77%';
  case_list_margin: any;

  constructor(
    public dialog: MatDialog,
    private caseProvider: CaseProvider,
    private meta:MetaDataProvider,
    private authProvider:AuthProvider,
    private cdr: ChangeDetectorRef,
    private paymentProvider:PaymentProvider,
    private handlerError: CustomErrorHandler,
    private dashboardProvider:DashboardProvider,
    private navigator: NavigationProvider,
    private responsiveProvider: ResponsiveProvider,
    private _snackBar: MatSnackBar,
    private commonProvider:CommonProvider,
    private utils: UtilsFunctions
  ) {

    this.bottom_paginator_data = {
      previousPageIndex: 0,
      pageIndex: 0,
      pageSize: 12,
      length: 0
    }

  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }

  ngOnInit() {
    localStorage.removeItem('case_reports_id');
    localStorage.removeItem('case_in_edition');
    localStorage.removeItem('new_case_data');
    localStorage.removeItem('filter_for_case');
    localStorage.removeItem('reports_for_case');
    localStorage.removeItem('max_airflow_allowed');
    this.getPaymentObject();
    this._getCaseList();
    this.getContentSize();
  }

  ngAfterViewInit() {
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
      this.isSearching = true;
      this.cases = this.caseProvider.filterCaseList(this.original_cases, changes);
    });
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  getPaymentObject(){
    this.paymentProvider.getAllProducts().subscribe((response)=>{
      this.optimization_plan = response.find(plan => plan.slug == "optimization");
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  getContentSize(){
    this.responsiveProvider.setContentWidth.subscribe((width)=>{
      this.case_list_width = width;
    })
    this.responsiveProvider.setContentMargin.subscribe((margin)=>{
      this.case_list_margin = margin;
    })
  }

  /******************************************************************
  *                         CASE ACTIONS                            *
  *******************************************************************/

  /**
   * TODO:  _attemptDeleteCase
   * Try to delete a study case selected, but pop up a warning to confir
   * if confirm then delete case 
   * @param study_case
  */
  _attemptDeleteCase(study_case: any) {
    
    let dialog_data = {
      title: 'case',
      description: "You WON'T be able to recover this case in the future. <br> By deleting your case all the comparisons you made for this case will be deleted as well"
    };
    let dialogRef = this.dialog.open(ConfirmDialogComponent, {
      width: '50%',
      height: 'auto',
      data: dialog_data
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.authProvider.getUserDashboard().subscribe((dashboard:any)=>{
          let ids = dashboard.find(widget => widget.type == "changeout").data;
          let pos = ids.indexOf(study_case.alter_autoincrement_id);
          if(pos != -1){
            ids.splice(pos,1);
            dashboard.find(widget => widget.type == "changeout").data = [...ids];
            this.authProvider.setUserDashboard(dashboard).subscribe((dashboard)=>{
              if(study_case.is_pinned){
                this.removeFromDashboard(study_case);
              }
            });
          }else{
            if(study_case.is_pinned){
              this.removeFromDashboard(study_case);
            }
          }
        })
        this._deleteStudyCase(study_case.alter_autoincrement_id);
      }
    });
  }

  // Clear search wekwords for case list
  clearSearch() {
    this.isSearching = false;
    this.search = '';
    if(this.empty_refine) {
      this.empty_refine = false;
      this.cases = [...this.original_cases];
      this.refreshPaginatorOptions(this.cases);
    }
  }

  /**
   * TODO:  _compareCase
   * Go to comparison with selected case to compare
   * @param study_case
   */
  _compareCase(study_case) {
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goNewComparisonCaseSelected(case_details);
      }
    )
  }

  /**
  * TODO: _deleteStudyCase
  * Given a study case delete selected study case 
  * @param case_id
  */
  _deleteStudyCase(case_id) {
    this.caseProvider.deleteStudyCase(case_id).subscribe(
      () => {
      this.commonProvider.preservePaginator();
      this.commonProvider.showSnackBar('Case deleted.','Your case has been deleted correctly.' ,'success-snackbar');
      if(this.cases.length == 1){
        this.cases = [];
        this.original_cases = [];
      }
    }, error => {
      console.log(error)
    })
  }

  /**
  * TODO: _duplicateCase
  * Given a study case duplicate selected case
  * @param study_case
  */
  _duplicateCase(study_case) {
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        var private_reports = case_details.current_filter.reports.filter(report => report.user);
        if(private_reports.length > 0){
          case_details.current_filter.private_report_ids = [];
          private_reports.forEach(report => {
            case_details.current_filter.private_report_ids.push(report.id);
          });
        }
        case_details.current_filter.model_filter = case_details.current_filter.model_filter.id;
        this.caseProvider.createNewCase(case_details).subscribe((response) => {
          this._getCaseList();
        }, error => {
          this.handlerError.handleError(error);
        })
      }
    )
  }

  /**
  * TODO: _editCase
  * Given a study case go to new case form with data to edit
  * @param study_case
  */
  _editCase(study_case) {
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goNewCase(case_details);
      }
    )
  }

  /**
  * TODO: _getCaseList
  * Get the study case list for a user
  * @param study_case
  */
  _getCaseList() {
    this.caseProvider.getPersonalCases().subscribe((cases) => {
      this._setCases(cases.results);
      this.setPaginatorOptions(cases);
    });
  }

  /**
  * TODO: _getCaseListByUrl
  * Given a URL from paginator get the study cases based on page and size
  * @param url
  */
  _getCaseListByUrl(url) {
    url = url.replace('-updated',this.order_date);
    this.caseProvider.getCasesByUrl(url).subscribe((cases) => {
      this._setCases(cases.results);
      this.setPaginatorOptions(cases);
    }, error => {
      this.handlerError.handleError(error);
    });
  }

  /**
  * TODO: _setCases
  * Given a case list set values for cases and columns distribution
  * @param cases
  */
  _setCases = (cases: Array<System>) => {
    this.cases = cases;
    this.cases.map(case_data => case_data.toggle_chart = false);
    this.cases.map(case_data => case_data.created_display = this.utils.formatDateStringMonth(case_data.created));
    this.cases.map(case_data => case_data.updated = this.utils.formatDateStringMonth(case_data.updated));
    this.original_cases = cases;
    this.original_cases.map(case_data => case_data.toggle_chart = false);
    this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);

    let progress_case = this.cases.find(_case => _case.is_processing);
    if(progress_case){
      this._refreshCaseStatus(progress_case,'is_processing');
    }

    this.authProvider.getUserDashboard().subscribe((dashboard:any)=>{
      let cases = dashboard.filter(widget=>widget.type == 'case' && widget.data);
      if(cases){
        cases.forEach(case_widget => {
          let case_pinned = this.cases.find(study_case => study_case.alter_autoincrement_id == case_widget.data);   
          if(case_pinned){
            case_pinned.is_pinned = true
          }
        });  
      }
    })
  
  }

  /**
   * TODO: _refreshCaseStatus
   * Gven a  case, refresh the status of the case
   * @param case
   * @param status
   */

  _refreshCaseStatus(study_case,status){
    study_case.display_status = status;
    if(status != 'done'){
      this.caseProvider.getCaseStatus(study_case.alter_autoincrement_id).subscribe((res:any)=>{
        setTimeout(() => {
          if(res.is_processing){
            this._refreshCaseStatus(study_case,res.status.status[res.status.status.length - 1].status);
          }else{
            if(!res.update_error){
              study_case.is_processing  = false;
              this._getCaseList();
              this._refreshCaseStatus(study_case,'done');
            }else{
              study_case.update_error = res.update_error
            }
          }
        }, 5000);
      })
    }else{
      setTimeout(() => {
        this.hide_check = true;
      }, 5000);
      study_case.is_done = true;
    }
  }
  /**
  * TODO: _refineCases
  * Given a case list set values for cases and columns distribution
  * @param cases
  */
 _refineCases = (cases: Array<System>) => {
  this.cases = cases;
  this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
}

  /******************************************************************
  *                          NAVIGATION                             *
  *******************************************************************/

  // Go to create new case
  _goNewCase() {
    if(this.original_cases.length > 0 ){
      let case_position = Object.keys(this.original_cases).reduce((a, b) => this.original_cases[a].alter_autoincrement_id > this.original_cases[b].alter_autoincrement_id ? a : b);
      localStorage.setItem('next_case_number',this.original_cases[case_position].alter_autoincrement_id);
    }else{
      localStorage.setItem('next_case_number','0');  
    }
    this.navigator._goNewCase(null);
  }

  // View optimization of case
  _viewOptimization(case_data: System) {
    this.navigator._goOptimizationsResults(case_data);
  }

  /******************************************************************
  *                         CASE PAGINATOR                          *
  *******************************************************************/

  setPageSizeOptions(setPageSizeOptionsInput: string) {
    if (setPageSizeOptionsInput) {
      this.pageSizeOptions = setPageSizeOptionsInput.split(',').map(str => +str);
    }
  }

  setPaginatorOptions(cases) {
    this.length = cases.count;
    this.bottom_paginator_data.length = cases.count;
  }

  refreshPaginatorOptions(cases) {
    this.length = cases.length;
    this.bottom_paginator_data.length = cases.length;
  }

  handlePage(event) {
    this.bottom_paginator_data = event.event;
    this._getCaseListByUrl(event.url);
  }

  handleManualPage(event) {
    this.jumpToPageOption = {
      jump: true,
      page: event.pageIndex,
      data: event
    }
  }

  pinCaseToDashboard(case_study){
    case_study.is_pinned = true;
    this.dashboardProvider.casePinned(case_study);
  }

  removeFromDashboard(case_study){
    this.dashboardProvider.removeCaseFromDashboard(case_study);
  }

  filterView(){
    this.empty_refine = false;
    var cases = [...this.original_cases];
    if(this.display_value == 'manual'){
      cases = cases.filter( _case =>!(_case.is_instrumented));
    }
    if(this.display_value == 'instrumented'){
      cases = cases.filter( _case => _case.is_instrumented );
    }
    if(cases.length == 0) this.empty_refine = true;
    this._refineCases(cases);
    this.refreshPaginatorOptions(cases);
  }

  _orderCaseList():void{
    this.caseProvider.reOrderCases(this.order_date).subscribe((cases)=>{
      this.bottom_paginator_data = {
        previousPageIndex: 0,
        pageIndex: 0,
        pageSize: 12,
        length: 0
      }
      this._setCases(cases.results);
      this.setPaginatorOptions(cases);
    })
  }

}
