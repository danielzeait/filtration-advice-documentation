import { Component, OnInit, Output, EventEmitter, Input, SimpleChanges } from '@angular/core';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-case-cost',
  templateUrl: './case-cost.component.html',
  styleUrls: ['./case-cost.component.scss']
})
export class CaseCostComponent implements OnInit {
  @Input('review') review:boolean;
  @Input('data') input_data:any;
  @Input('energy_data') energy_data:any;

  @Output() set = new EventEmitter();
  @Output() emit_data = new EventEmitter();
  public additional_costs:any=[];
  public annual_costs:any=[];
  public list_height:string="340px";
  public case_cost:any={
    filter_extra_cost:[],
    freight_cost:null,
    labor_cost:null,
    disposal_cost:null,
    filter_cost:null,
    energy_cost:null,
    carbon_footprint:1.352,
    filters_number:null,
  };
  extra_costs_empty: boolean;
  constructor(private utils:UtilsFunctions) { }
  
  ngOnInit() {
    if(this.input_data){
      this.case_cost = {...this.input_data};
    }else{
      this.case_cost.energy_cost = 0.12;
    }
  }

  ngOnChanges(changes: SimpleChanges){
    if(changes.input_data && changes.input_data.currentValue){
      this.case_cost = {...changes.input_data.currentValue};
      if(this.case_cost.filter_extra_cost.length > 3){
        this.getAdditionalCosts();
        this.getAnnualCosts();
      }
    }
    if(changes.review && !changes.review.firstChange){
      this.review = changes.review.currentValue;
    }
    // if(changes.energy_data && !changes.energy_data.firstChange){
    //   this.case_cost.energy_cost ? this.case_cost.energy_cost = changes.energy_data.currentValue.energy_cost : '';
    //   this.case_cost.carbon_footprint ? this.case_cost.carbon_footprint = changes.energy_data.currentValue.carbon_footprint : '';
    //   this.emitData();
    // }
  }

  
  newAdditionalCost(){
    this.additional_costs.push(
      {
        name:'',
        value:null,
        is_annual:false,
      }
    );
    this.list_height = this.getRowHeight();
  }

  removeAdditionalCost(pos){
    this.additional_costs.splice(pos,1);
    this.list_height = this.getRowHeight();
  }

  newAnnualCost(){
    this.annual_costs.push(
      {
        name:'',
        value:null,
        is_annual:true,
      }
    );
    this.list_height = this.getRowHeight();
  }

  removeAnnuallCost(pos){
    this.annual_costs.splice(pos,1);
    this.list_height = this.getRowHeight();
  }

  setAmountDecimal(ev) {
    this.utils.transformAmount(ev);
  }

  getRowHeight(){ 
    var grater;
    this.additional_costs.length > this.annual_costs.length - 3 ? grater = this.additional_costs.length : grater = this.annual_costs.length - 3;
    return (340 + (40 * grater)).toString() + 'px';
  }

  setCaseCost(){
    !this.case_cost.labor_cost || this.case_cost.labor_cost == "" ?  this.case_cost.labor_cost = 0 : this.case_cost.labor_cost;
    !this.case_cost.freight_cost || this.case_cost.freight_cost == "" ? this.case_cost.freight_cost = 0 : this.case_cost.freight_cost;
    !this.case_cost.disposal_cost || this.case_cost.disposal_cost == "" ?  this.case_cost.disposal_cost = 0 : this.case_cost.disposal_cost;

    let base_cost = [
      {name:'Freight cost', value:this.case_cost.freight_cost, is_annual:false},
      {name:'Labor cost', value:this.case_cost.labor_cost, is_annual:false},
      {name:'Disposal cost', value:this.case_cost.disposal_cost, is_annual:false},
    ]

    this.case_cost.filter_extra_cost = [...base_cost, ...this.additional_costs, ...this.annual_costs];;

    this.set.emit(this.case_cost);
  }

  getAdditionalCosts(){
    this.additional_costs = this.case_cost.filter_extra_cost.filter(cost => 
      cost.name !== "Labor cost" && cost.name !== "Freight cost" && cost.name !== "Disposal cost" && !cost.is_annual)
  }

  getAnnualCosts(){
    this.annual_costs = this.case_cost.filter_extra_cost.filter(cost => 
      cost.name !== "Labor cost" && cost.name !== "Freight cost" && cost.name !== "Disposal cost" && cost.is_annual)
  }
  
  getDisableExpression(){
    if(this.annual_costs.find(cost=>!cost.value || cost.value == '' || cost.value < 0) ||
    this.additional_costs.find(cost=>!cost.value || cost.value == '' || cost.value < 0 )){
      this.extra_costs_empty = true;
    }else{
      this.extra_costs_empty = false;
    }
    
    return  !this.case_cost.filter_cost || !this.case_cost.filters_number ||
            !this.case_cost.energy_cost || !this.case_cost.carbon_footprint ||
            this.case_cost.filter_cost == '' || this.case_cost.filters_number == '' ||
            this.case_cost.energy_cost == '' || this.case_cost.carbon_footprint == '' ||
            this.case_cost.filter_cost < 0 || this.case_cost.filters_number < 0 || 
            this.case_cost.energy_cost < 0 || this.case_cost.carbon_footprint < 0 ||
            this.annual_costs.find(cost=>!cost.value || cost.value == '' || cost.value < 0) ||
            this.additional_costs.find(cost=>!cost.value || cost.value == '' || cost.value < 0)
  }

  emitData(){
    this.setCaseCost();
    this.emit_data.emit(this.case_cost);
  }

}
