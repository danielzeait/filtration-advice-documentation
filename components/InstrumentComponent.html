@Component({
  selector: 'app-instrument-list',
  templateUrl: './instrument-list.component.html',
  styleUrls: ['./instrument-list.component.scss']
})

export class InstrumentComponent implements OnInit {
  public searchedWordsControl = new FormControl('');
  public show_me:Array<boolean> = []; 
  displayedColumns: string[] = ['serial_number', 'name', 'unit', 'status', 'type'];

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
    )

  private original_instrument: Array<Instrument> = [];
  public num_cols: number;
  public search: string = "";
  public texts_to_hightlight: NodeListOf<Element>;
  public instrument: Array<Instrument>;
  public instrument_search_result: Array<Instrument> = [];
  public instrument_selected: Instrument = new Instrument();
  public showInstrument: boolean = false;
  public instrument_providers = [];
  public original_providers:any;
  public log_provider:boolean = false;
  public user_has_instruments:boolean = true;
  public monitoring_subscription:any;
  provider_to_log: any;
  hobo_dashboard: boolean;
  providers_tabs: boolean;
  provider_selected: any;


  constructor(
    private handlerError:CustomErrorHandler, 
    private instrumentProvider: InstrumentProvider,
    private navigator: NavigationProvider,
    private caseProvider:CaseProvider,
    private commonProvider:CommonProvider,
    private paymentProvider:PaymentProvider,
    private instrumentsProvider:InstrumentProvider,
    public dialog: MatDialog) { }

  @ViewChild(MatPaginator, {static: true}) paginator: MatPaginator;

  ngOnInit() {
    //this._getInstrumentList();
    this.paymentProvider.getUserSubscriptions().subscribe((response)=>{
      this.user_has_instruments = response.find(subscription => subscription.plan.slug == "unlimited_instrumented_system") ? true : false;
      if(this.user_has_instruments){
        this.instrumentProvider.getAllInstrumentProviders().subscribe((providers)=>{
          this.original_providers = [...providers.reverse()];
          this.instrument_providers = [...providers.reverse()];
        });
      }
    })
  }

  ngAfterViewInit() {
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
      if(this.instrument){
        this.instrument = this.instrumentProvider.filterInstrumentList(this.original_instrument,changes);
      }else{
        this.instrument_providers = this.instrumentProvider.filterInstrumentProvidersList(this.original_providers,changes);
      }
    })
  }


  _upadateSonicuInstrumentList(){
    this.instrumentProvider.updateInstrumentList().subscribe(()=>{
      this.getSonicuInstruments();
      //this.setInstrument(response);
    })
  }
  
  _getInstrumentsFromProvider(provider){
    this.provider_selected = provider;
    if(!provider.is_already_logged){
      this.provider_to_log = provider;
      this.log_provider = true;
    }else{
      this.providers_tabs = true;
      if(provider.slug == 'sonicu'){
        this.getSonicuInstruments();
      }else{
        this.log_provider = false;
        this.hobo_dashboard = true;
      }
    }
  }

  getHoboInstruments(){
    this.instrumentProvider.getHoboInstrument().subscribe((response)=>{})
  }

  getSonicuInstruments(){
    this.instrumentProvider.getSonicuInstruments().subscribe((response)=>{
      this.setInstrument(response);
    })    
  }

  redirectOnUnauthorized() {
    this.navigator._goLoginSonicu();
  }

  setInstrument = (instrument: Array<Instrument>) => {
    this.instrument = instrument;
    this.original_instrument = this.instrument;
    this.instrument.sort((a,b) => 
      (a['units'] > b['units']) ? 1 : ((b['units'] > a['units']) ? -1 : 0)
    )
    this.instrument.reverse();
    this.instrument.map(i => i.paired_with ? i.case_name = i.paired_with.case_name : i.case_name = 'Unpaired' );
    this.instrument.map(i => i.slot ? '' : i.slot = '-');
  }

  _startNewCase() {
    this.navigator._goHomeCases(this.instrument);
  }

  _viewCaseResults(instrument:Instrument){
    var case_id = instrument.paired_with.case_id;

    this.caseProvider.getPersonalCases().subscribe((response)=>{
      var _case = response.find(paired_case=>paired_case.id = case_id);
      this.navigator._goOptimizationsResults(_case);
    },error=>{

    })
  }

  _viewOptimization(case_data:System) {
    this.navigator._goOptimizationsResults(case_data);
  }

  _unpair(instrument){
    this.instrumentProvider.unpairInstrument(instrument).subscribe((response)=>{
      this.commonProvider.showSnackBar('Paired.','The instrument was paired with your case correctly.' ,'success-snackbar');
      window.location.reload();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  _openPairedModalDialog(instrument): void {
    const dialogRef = this.dialog.open(PairedCaseInstrumentComponent, {
      width: '80%',
      height: '90%',
      id: 'pairInstrument',
      data: instrument
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {}
    });
  }

  _prividerLogged(data){
    this.log_provider = false;
    this.setInstrument(data);
  }

  _clearSearch(){
    this.instrument_providers = this.original_providers;
    if(this.instrument){
      this.instrument = this.original_instrument;
    }
    this.search = '';
  }

  _beginInstrumentation(){
    this.navigator._goNewInstrumentation();
  }





}
