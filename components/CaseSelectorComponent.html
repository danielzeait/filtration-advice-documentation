import { Component, OnInit, Inject } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { System } from 'src/app/models/system.model';
import { CaseProvider } from 'src/app/providers/case.service';
import { global } from '../../../global'
import { element } from 'protractor';

@Component({
  selector: 'app-case-selector',
  templateUrl: './case-selector.component.html',
  styleUrls: ['./case-selector.component.scss']
})
export class CaseSelectorComponent implements OnInit {
  public cases:any=[];
  public original_cases:any=[];
  public length:number;
  public pageSize = 12;
  public pageSizeOptions: number[] = [5, 10, 12, 15, 20];
  public jumpToPageOption:any;
  public list_url:any = global.API_CASES;
  public case_to_changeout: any = [];
  max_limit: boolean;
;
  public bottom_paginator_data:any = {
    previousPageIndex: 0,
    pageIndex: 0,
    pageSize: 8,
    length: 0
  }
  constructor(
    private caseProvider:CaseProvider,
    public dialogRef: MatDialogRef<CaseSelectorComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
  ) { }

  ngOnInit() {
    if(this.data){
      this.case_to_changeout = [...this.data];
    }
    
    this.getCases();
  }

  getCases(){
    this.caseProvider.getPersonalCases().subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    });
  }

  handlePage(event){
    this._getCaseListByUrl(event.url);
  }

  _getCaseListByUrl(url) {
    this.caseProvider.getCasesByUrl(url).subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    },error=>{
      //this.handlerError.handleError(error);
    });
  }

  setCases = (cases: Array<System>) => {
    this.cases = cases;
    this.original_cases = cases;
    if(this.data){
      this.data.forEach(id => {
        this.cases.find(_case=>_case.alter_autoincrement_id == id) ? this.cases.find(_case=>_case.alter_autoincrement_id == id).selected = true : {};
      });
    }
  }

  setPageSizeOptions(setPageSizeOptionsInput: string) {
    if (setPageSizeOptionsInput) {
      this.pageSizeOptions = setPageSizeOptionsInput.split(',').map(str => +str);
    }
  }

  setPaginatorOptions(cases){
    this.length = cases.count;
    this.bottom_paginator_data.length = cases.count;
  }

  close(){
    this.dialogRef.close(null);
  }

  setCase(study_case,index){
    if(this.case_to_changeout.length < 4){
      this.case_to_changeout.push(study_case.alter_autoincrement_id);
      this.cases[index].selected = true;
    }else{
      this.max_limit = true;
    }
  }
  
  removeCase(study_case,index){
    var pos = this.case_to_changeout.findIndex(element => element == study_case.alter_autoincrement_id);
    this.case_to_changeout.splice(pos,1);
    this.cases[index].selected = false;
    if(this.case_to_changeout.length < 4){
      this.max_limit = false;
    }
  }

  setChangeoutListCases(){
    this.dialogRef.close(this.case_to_changeout);
  }

}
