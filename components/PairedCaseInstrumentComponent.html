@Component({
  selector: 'app-paired-list',
  templateUrl: './paired-case-instrument.component.html',
  styleUrls: ['./paired-case-instrument.component.scss']
})
export class PairedCaseInstrumentComponent implements OnInit {
  @Input('data') instrument: any;
  @Input('cases') cases:any;
  @Input('case') selected_case:any;
  @Input('filter') filter: any;
  @ViewChild('systemForm', { static: true }) systemForm;
  maxDate: Date;
  public existing_case: any;
  public original_cases: Array<System> = new Array<System>();
  public original_case: any;
  public case_data: System = null;
  public num_cols: number = 4;
  public search: string = "";
  public searchedWordsControl = new FormControl('');
  public server_name: string = global.SERVER_NAME;
  public new_case: boolean = false;
  public new_case_filter: any = null;
  public edit: boolean = false;
  public is_paired: boolean = false;
  public innerWidth: any;
  public instrument_type: any;
  public system: System = null;
  public system_paired: any = {
    id: '',
    case: '',
    instrument_type: '',
    filterless_dp: 0.02,
    airflow_of_filterless_dp:118,
    air_flow_instrument_coefficient:'',
    is_circular_pipe:false,
    diameter:'',
    height:'',
    width:'',
    installation_date:'',
  };
  public log_error: string = '';
  public length:number;
  public pageSize = 20;
  public pageSizeOptions: number[] = [20,40,60,80,100];
  public jumpToPageOption:any;
  public list_url:any;
  public bottom_paginator_data:any = {
    previousPageIndex: 0,
    pageIndex: 0,
    pageSize: 20,
    length: 0
  }


  public requiredFields = {
    filterless_dp: new FormControl('', [Validators.required]),
    installation_date: new FormControl('', [Validators.required]),
  }


  public searchedWords$: Observable<string[]> = this
  .searchedWordsControl
  .valueChanges
  .pipe(
    map((search: string) => search.trim().split(' '))
  )
  rectangle: any = true;
  circle: boolean = false;
  unavailable_slot: boolean;


  constructor(
    private commonProvider:CommonProvider,
    private handlerError:CustomErrorHandler,
    public dialogRef: MatDialogRef<PairedCaseInstrumentComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private navigator: NavigationProvider,
    private responsiveProvider: ResponsiveProvider,
    private errorHandler:CustomErrorHandler,
    private caseProvider: CaseProvider,
    private routeActivated: ActivatedRoute,
    private pairCaseInstrumentProvider: PairedCaseInstrumentProvider,
    private notificationProvider: NotificationService,
    public utils: UtilsFunctions,
    ) { }

  ngOnInit() {
    this.system = new System();
    this.system_paired.is_circular_pipe = false;
    this._getCaseList();
    this.maxDate = new Date();
    this.routeActivated.queryParams.subscribe(async params => {
      if (params.filter && params.filter !== "null") {
        this.new_case_filter = JSON.parse(params.filter);
        this.new_case = true;
      } else {
        localStorage.removeItem('new_case_data');
      }
    });
  }

  handlePage(event){
    this._getCaseListByUrl(event.url);
  }

  _getCaseListByUrl(url) {
    this.caseProvider.getCasesByUrl(url).subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    },error=>{
      this.handlerError.handleError(error);
    });
  }

  _startNewCase() {
    this.dialogRef.close(false);
    this.navigator._goNewCase(null);
  }


  _getCaseList() {
    this.caseProvider.getPersonalCases().subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    });
  }

  
  setCases = (cases: Array<System>) => {
    this.cases = cases;
    this.cases.map(case_data => case_data.toggle_chart = false);
    this.original_cases = cases;
    this.original_cases.map(case_data => case_data.toggle_chart = false);
    this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }

  setPaginatorOptions(cases){
    this.length = cases.count;
    this.bottom_paginator_data.length = cases.count;
  }

   setDefaultCaseSelected(){
    this.original_case = null;
    this.cases = this.cases.map(function(e) {
      e.selected ? e.selected = true : e.selected = false;
      return e;
    });
  }

   clearSelectedCase(){
    this.cases = this.cases.map(function(e) {
      e.selected = false;
      return e;
    });
  }

  setOriginalCase(study_case, index){
    this.unavailable_slot = false;
    this.clearSelectedCase();
    this.cases[index].selected = true;
    this.original_case = study_case;
    this.system_paired.case = study_case.id;
    if(study_case.is_instrumented){
      this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
        (case_details)=> {
          if(case_details.monitoring_module.drop_pressure_inst){
            this.setDpInstrumentData(case_details);
          }
          if(case_details.monitoring_module.flow_inst){
            this.setAirFlowIntrumentData(case_details);
          }
          if(case_details.monitoring_module.flow_inst && case_details.monitoring_module.drop_pressure_inst && this.data.units == 'inWC'){
            this.unavailable_slot = true;
          }
        }
      )
    }
  }

  setAirFlowIntrumentData(case_details){
    this.system_paired.diameter = case_details.diameter;
    this.system_paired.is_circular_pipe = case_details.is_circular_pipe;
    this.system_paired.width = case_details.width;
    this.system_paired.height = case_details.height;
    this.system_paired.airflow_of_filterless_dp = case_details.airflow_of_filterless_dp;
    this.system_paired.air_flow_instrument_coefficient = case_details.monitoring_module.air_flow_instrument_coefficient;
  }

  setDpInstrumentData(case_details){
    this.system_paired.filterless_dp = case_details.filterless_dp;
    this.system_paired.monitoring_module.installation_date = case_details.monitoring_module.installation_date;
  }

  cancel(){
    this.dialogRef.close(false);
  }

    setAmountDecimal(ev) {
    this.utils.transformAmount(ev);
  }

  getDisabledPair(system_paired) {

    if(system_paired.instrument_type == 'drop_pressure' || system_paired.instrument_type == 'flow'){
      return (!system_paired.monitoring_module.installation_date || !system_paired.filterless_dp || !this.original_case)
    }else{
      return ( !system_paired.instrument_type || !this.original_case)
    }
  }

  setDuctShape(shape){
    if(shape == 'rectangle'){
      !this.rectangle ? this.circle = true : this.circle = false
      this.system_paired.is_circular_pipe = false;  
    }else{
      this.system_paired.is_circular_pipe = true;
      !this.circle ? this.rectangle = true : this.rectangle = false
    }
  }

  paired(data) {
    this.system_paired.id = this.data.id;
    this.system_paired.monitoring_module.installation_date ? 
    this.system_paired.monitoring_module.installation_date = this.utils.formatDateISO(this.system_paired.monitoring_module.installation_date) : null;
    
    this.pairCaseInstrumentProvider.createNewPaired(this.system_paired,this.original_case.id).subscribe((response)=>{
      this.commonProvider.showSnackBar('Instrument paired.','The instrument was paired with your case correctly.','success-snackbar');
      window.location.reload();
    },error=>{
      this.handlerError.handleError(error);
    })
  }



}
