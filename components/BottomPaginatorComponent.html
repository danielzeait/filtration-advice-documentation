import { Component, OnInit, Input, Output, EventEmitter, AfterViewInit, SimpleChanges, SimpleChange } from '@angular/core';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-bottom-paginator',
  templateUrl: './bottom-paginator.component.html',
  styleUrls: ['./bottom-paginator.component.scss']
})
export class BottomPaginatorComponent implements OnInit {
  @Input("data") data:any;  
  @Input("count") count:any;
  @Output() jump_page  = new EventEmitter<string>();
  public pages: any;
  public hide_zone:any={
    min:0,
    max:0,
    pivot:0,
  }
  constructor(private commonProvider:CommonProvider) { }

  ngOnInit() {
    this.getNumOfPages();

    this.commonProvider.updatePaginator.subscribe(()=>{
      let page = this.data.pageIndex;
      if( this.count == 1){
        page = this.data.pageIndex !== 0 ? this.data.pageIndex - 1 : 0
      }
      setTimeout(() => {
        this.data.length = this.data.length - 1;
        this.getNumOfPages();
        this.updateManualPage(page);
      }, 500);
    })

  }

  ngOnChanges(changes:SimpleChanges){
    if(changes.data && !changes.data.isFirstChange()){
      this.data = changes.data.currentValue;
      this.getNumOfPages();
    }
  }

  getNumOfPages(){
    var pages:any = this.data.length / this.data.pageSize;
    if(this.data.length % this.data.pageSize !== 0){
      pages = pages + 1;
    } 
    this.setHideZone(pages);
    this.setPages(parseInt(pages));
  }

  setHideZone(pages){
    this.hide_zone.min = this.data.pageIndex + 2;
    this.hide_zone.max = parseInt(pages) - 3;
    this.hide_zone.pivot = this.hide_zone.min + 2;
  }

  setPages(num_pages){
    this.pages = [];
    for (let i = 0; i < num_pages; i++) {
      let element = {
        page:i,
        active:this.data.pageIndex == i,
      };
      this.pages.push(element);
    }
  }

  updateManualPage(page){
    this.data.pageIndex = page;
    this.pages.map(page => page.active = false);
    this.pages && page ? this.pages[page].active = true : null;
    this.setHideZone(this.pages.length);
    this.jump_page.emit(this.data);
  }

}
