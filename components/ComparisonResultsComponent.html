  setSavingsValues(){
    let saving_current_vs_current_opt = {
      is_best:false,
      index:0,
      value:this.result_data.case.current_filter.change_rate_annual_cost - this.result_data.case.current_filter.optimize_change_rate_annual_cost,
      percent: (((
        this.result_data.case.current_filter.change_rate_annual_cost - this.result_data.case.current_filter.optimize_change_rate_annual_cost) 
        * 100) / (this.result_data.case.current_filter.change_rate_annual_cost)).toFixed(2)
    }
    let saving_current_vs_current_selected = {
      is_best:false,
      index:1,
      value:this.result_data.case.current_filter.change_rate_annual_cost - this.selected_data.annual_cost - 0.005,
      percent: (((
        this.result_data.case.current_filter.change_rate_annual_cost - this.selected_data.annual_cost) 
        * 100) / (this.result_data.case.current_filter.change_rate_annual_cost)).toFixed(2)
    }

    let saving_current_vs_proposed = {
      is_best:false,
      index:2,
      value:parseFloat(this.savings.current_savings),
      percent:this.savings_percent 
    }

    let saving_current_vs_proposed_selected = {
      is_best:false,
      index:3,
      value:(this.result_data.case.current_filter.change_rate_annual_cost - this.selected_data_proposed_filter.annual_cost) - 0.005,
      percent: (((
        this.result_data.case.current_filter.change_rate_annual_cost - this.selected_data_proposed_filter.annual_cost) 
        * 100) / (this.result_data.case.current_filter.change_rate_annual_cost)).toFixed(2)
    }

    this.savings_array = [
      saving_current_vs_current_opt,
      saving_current_vs_proposed,
      saving_current_vs_current_selected,
      saving_current_vs_proposed_selected
    ]

    var original_savings = [
      saving_current_vs_current_opt,
      saving_current_vs_proposed,
      saving_current_vs_current_selected,
      saving_current_vs_proposed_selected
    ]

    var best = original_savings.sort(function(a, b) {  return a.value - b.value; }).reverse()[0];

    this.savings_array.find(saving => saving.index == best.index).is_best = true;

  }

  editComparison(study_case, comparison){
    comparison.filter_extra_cost = this.result_data.comparison_data.filter_extra_cost;
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goEditComparison(case_details,comparison);
      }
    )
  }

  _getCurrentFilterChart(){
      this.caseProvider.getCaseCostVsTimeChart(this.result_data.case.alter_autoincrement_id).subscribe((response)=>{
        this.current_filter_cost_vs_time_chart = response.cost_chart;
        this.current_limit = this.current_filter_cost_vs_time_chart.find(point => point.is_current);
        if(this.result_data.case.current_filter.selected_time && !this.result_data.case.monitoring_module){
          //8.777876683307117
          this.selected_data = this.current_filter_cost_vs_time_chart.find(a=>a.change_out_time.toFixed(12) == this.result_data.case.current_filter.selected_time.toFixed(12));
          var i = this.current_filter_cost_vs_time_chart.indexOf(this.selected_data);
          this.range_value = i;
          this.current_circle_change = true;
          this.can_consult = true;
        }else{
          this.selected_data = this.current_filter_cost_vs_time_chart.find(a=>a.is_optimal);
          this.range_limit = this.current_filter_cost_vs_time_chart.length-1;
          let middle = this.current_filter_cost_vs_time_chart[Math.round((this.current_filter_cost_vs_time_chart.length - 1) / 2)];
          var i = this.current_filter_cost_vs_time_chart.indexOf(middle);
          this.range_value = i;
        }

        this.max_operational_point = this.current_filter_cost_vs_time_chart.find(point => point.is_at_max_operational); 
        if(this.max_operational_point){
          this.max_operational_limit = this.max_operational_point.change_out_time;
        }else{
          this.max_operational_limit = (this.current_filter_cost_vs_time_chart.length -1);
        }
      },error=>{
        this.handlerError.handleError(error);
      });
  }

      }
    )
  }

  _getProposedFilterChart(){
    return new Promise(resolve => {
      this.comparisonProvider.getComparisonChartCostVsTime(this.result_data).subscribe((response)=>{
        this.proposed_filter_cost_vs_time_chart = response.cost_chart;
        this.range_limit_proposed_filter = this.proposed_filter_cost_vs_time_chart.length-1;
