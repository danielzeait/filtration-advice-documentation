import { Component, OnInit } from '@angular/core';
import { version } from 'punycode';
import { CommonProvider } from 'src/app/providers/common.service';
import { MetaDataProvider } from 'src/app/providers/metadata.service';

import { environment } from '../../../../environments/environment';

@Component({
  selector: 'app-about',
  templateUrl: './about.component.html',
  styleUrls: ['./about.component.scss']
})
export class AboutComponent implements OnInit {

  public versions:any;
  public front_version:any = environment.VERSION;

  constructor(private meta:MetaDataProvider) { }

  ngOnInit(){
    this.meta.getSystemVersion().subscribe(version=>{
      this.versions = version;
    })
  }

}

<div class="about-container" *ngIf="versions">
    <h4 class="color-secondary"> Versions </h4>
    <div class="versions">
        <p> Engine Version: <span>{{ versions.engine_version }}</span> </p>
        <p> System Version: <span>{{ versions.local_version }}</span> </p>
        <p> Local Version: <span>{{ front_version }}</span> </p>
    </div>
</div>
