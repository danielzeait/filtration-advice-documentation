import { Component, OnInit } from '@angular/core';
import {FormControl, Validators, FormBuilder, FormGroup} from '@angular/forms';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { SuccessDialogComponent } from 'src/app/components/modal-dialogs/success-dialog/success-dialog.component';
import { MatDialog } from '@angular/material/dialog';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-register-page',
  templateUrl: './register-page.component.html',
  styleUrls: ['./register-page.component.scss']
})
export class RegisterPageComponent implements OnInit {
  
  // public password = new FormControl('', [Validators.required]);
  // public confirm_password = new FormControl('', [Validators.required]);
  // public email = new FormControl('', [Validators.required, Validators.email]);
  public hide:boolean = true;
  public hide_confirm:boolean = true;
  public registerData:any = {};
  public register_errors:Array<any> = [];
  public showTC: boolean = false;
  public registerForm:FormGroup;
  register_success: boolean;

  constructor(
    private authProvider:AuthProvider, 
    private dialog:MatDialog,
    private commonProvider:CommonProvider,
    private fb: FormBuilder,
    private navigation:NavigationProvider,
  ) { }

  ngOnInit() {
    this.registerData = {
      first_name:'',
      last_name:'',
      email:'',
      username:'',
      password:'',
      confirm_password:''
    }
    this.registerForm = this.fb.group({
      username: new FormControl(''),
      email: new FormControl('', Validators.compose([
        Validators.required,
        Validators.email
      ])),
      first_name: new FormControl('', [Validators.required]),
      last_name: new FormControl('', [Validators.required]),
      password: new FormControl('', [Validators.required,Validators.minLength(8)]),
      confirm_password: new FormControl('', [Validators.required,Validators.minLength(8)]),
    },{validator:this.checkPasswords});

  }

  checkPasswords(group: FormGroup) { // here we have the 'passwords' group
    let pass = group.controls.password.value;
    let confirmPass = group.controls.confirm_password.value;

    return pass === confirmPass ? null : { notSame: true }
  }

  _tryRegister(){
    this.showTC = true;
  }
  
  _continue(){
    this.registerData = this.registerForm.value;
    this.registerData.username = this.registerData.email;
    this.authProvider.registerUser(this.registerData).subscribe((response)=>{
      this.registerData.token = response.token;
      this.register_success = true;
    },error=>{
      this.showTC = false;
      this.register_errors = [];
      var obj = error.error;
      this.handleRegistErrors(obj);
    })
  }

  _resendEmail(){
    this.authProvider.registerUserConfirmation(this.registerData.token).subscribe((response)=>{
      this.commonProvider.showSnackBar('Confirmation email','We sent you an emial to confirm your acount','alert-success');
    });
  }

  _finishRegister(){
    this.navigation._land();
  }

  showModalSuccess(){
    let dialog_data = {
      title:'ACCOUNT SUCCESFULLY CREATED',
      message:'We sent you a confirmation email to validate your account',
      action:'Ok. I understand'
    }
    const dialogRef = this.dialog.open(SuccessDialogComponent, {
      width: '50%',
      height: 'auto',
      data: dialog_data,
      id:'confirmed'
    });
  }

  handleRegistErrors(obj){
    for (let key in obj){
      if(obj.hasOwnProperty(key)){
        this.register_errors.push(obj[key]);
      }
    }
  }

  _goLoginPage(){
    this.navigation._goLogin();
  }

  getErrorMessage() {
    return this.registerForm.controls.email.hasError('required') ? 'You must enter a value' :
        this.registerForm.controls.email.hasError('email') ? 'Not a valid email' :
            '';
  }

  getMissMatchPasswordErrorMessage(){
    return "Confirm password doesn't match"
  }

  getPasswordErrorMessage(){
    return "Password must have at least 8 characters"
  }

}
