import { Component, OnInit, Input, SimpleChanges } from '@angular/core';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-instrument-indicator',
  templateUrl: './instrument-indicator.component.html',
  styleUrls: ['./instrument-indicator.component.scss']
})
export class InstrumentIndicatorComponent implements OnInit {
  @Input('instruments') instruments : any;
  constructor( private utils:UtilsFunctions ) { }

  ngOnInit() {
    this.instruments.flow_inst.last_reading !== "" ? this.instruments.flow_inst.last_reading = parseFloat(this.instruments.flow_inst.last_reading) : '';
    this.instruments.drop_pressure_inst.last_reading !== "" ? this.instruments.drop_pressure_inst.last_reading = parseFloat(this.instruments.drop_pressure_inst.last_reading).toFixed(3) : '';
  }

  ngOnChanges(changes: SimpleChanges){
    if(!changes.instruments.firstChange){
      this.instruments = {...changes.instruments.currentValue};
      this.instruments.flow_inst.last_reading !== "" ? this.instruments.flow_inst.last_reading = parseFloat(this.instruments.flow_inst.last_reading) : '';
      this.instruments.drop_pressure_inst.last_reading !== "" ? this.instruments.drop_pressure_inst.last_reading = parseFloat(this.instruments.drop_pressure_inst.last_reading).toFixed(3) : '';
    };
  }

  getCheckDate(date){
    return this.utils.formatDateTimeInstrument(date);
  }

  showLastCheck(){
    console.log('show');
  }

  hideLastCheck(){
    console.log('hide');
  }

}
