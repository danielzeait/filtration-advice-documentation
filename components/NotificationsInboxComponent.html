import { Component, OnInit } from '@angular/core';
import { NotificationService } from 'src/app/providers/notifications.service';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-notifications-inbox',
  templateUrl: './notifications-inbox.component.html',
  styleUrls: ['./notifications-inbox.component.scss']
})
export class NotificationsInboxComponent implements OnInit {
  public notifications: { title: string; image: string; type: string; text: string; time: string; seen: boolean; priority: string; is_read:boolean, body:string, string_type:string}[];
  public read_all:boolean=false;
  constructor(
    private commonProvider:CommonProvider,
    private notificationProvider:NotificationService) { }

  ngOnInit(){
    this.getNotifications();
  }

  getNotifications(){
    this.notificationProvider.getNotifications().subscribe((notifications:any)=>{
      this.notifications = notifications;
      this.notifications.map(item => item.body.length > 112 ? item.body = item.body.slice(0, 112) + '...' : item.body);
      this.notifications.map(item => item.string_type = item.type.replace('_', ' '));
    })
  }

  readAll(){
    this.notifications.map(notification => {
      if(!notification.is_read){
        this.readNotification(notification)
      }
    });
  }

  _deleteNotification(notification,pos){
    this.notificationProvider.deleteNotification(notification).subscribe(
      ()=>{
        this.notificationProvider.getNotifications().subscribe((notifications:any)=>{
          this.notifications = notifications;
          this.commonProvider.updateNotifications(this.notifications);
        })
    })
  }

  readAllNotifications(){
    
  }
  
  readNotification(notification){
    if(!notification.is_read){
      notification.is_read = true;
      this.notificationProvider.readNotification(notification).subscribe(
        ()=>{
          this.commonProvider.updateNotifications(this.notifications);
      })
    }
  }

  readMore(notification){}


}
