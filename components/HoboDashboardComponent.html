import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { InstrumentProvider } from 'src/app/providers/instrument.service';

@Component({
  selector: 'app-hobo-dashboard',
  templateUrl: './hobo-dashboard.component.html',
  styleUrls: ['./hobo-dashboard.component.scss']
})
export class HoboDashboardComponent implements OnInit {
  @Input('is_for_case') is_for_case:boolean;
  @Output() select = new EventEmitter();
  constructor(private instrumentProvider:InstrumentProvider) { }

  public new_logger_serial:any;
  public hobo_loggers:any;

  ngOnInit(): void {
    this.getHoboLoggers();
  }

  getHoboLoggers(){
    this.instrumentProvider.getHoboInstrument().subscribe((loggers)=>{
      this.hobo_loggers = loggers;
    })
  }

  addHoboLoggers(){
    this.instrumentProvider.setHoboLogger(this.new_logger_serial).subscribe((response)=>{
      this.getHoboLoggers();
    })
  }

  useSensor(sensor){
    this.select.emit(sensor);
  }

}
