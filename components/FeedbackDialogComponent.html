import { Component, Inject, OnInit } from '@angular/core';
import { FormGroup, FormControl, Validators, FormBuilder } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-feedback-dialog',
  templateUrl: './feedback-dialog.component.html',
  styleUrls: ['./feedback-dialog.component.scss']
})
export class FeedbackDialogComponent implements OnInit {
  public stars:any = ['star_border','star_border','star_border','star_border','star_border'];
  public reviewForm:FormGroup;
  public message_sent:boolean=false;
  public sending:boolean=false;
  public button_class:string='';
  public starset:boolean = false;
  constructor(
    private fb: FormBuilder, 
    private authService:AuthProvider,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private dialogref:MatDialogRef<FeedbackDialogComponent>) { }

  ngOnInit(){
    this.setReviewForm();
  }

  setReviewForm(){
    this.reviewForm = this.fb.group({
      email: new FormControl({value:this.data.email,disabled:true}),
      rating:new FormControl({value:'',disabled:this.sending}),
      review:new FormControl({value:'',disabled:this.sending},Validators.required),
    });
  }

  setStars(pos){
    this.stars = ['star_border','star_border','star_border','star_border','star_border'];
    for (let i = 0; i < pos +1; i++) {
      this.stars[i] = 'star'
    }
    this.reviewForm.controls.rating = pos+ 1
    this.starset = true;
  }



  sendMessage(){
    //this.reviewForm.disable()
    this.sending = true;
    this.button_class = 'animated fadeIn infinite';
    let data = {
      comment:this.reviewForm.controls.review.value,
      rate:this.reviewForm.controls.rating
    }
    
    this.authService.sendUserReview(data).subscribe(()=>{
      this.sending = false;
      this.button_class = '';
      this.message_sent = true;
    })
  }

  close(){
    this.dialogref.close();
  }

}
