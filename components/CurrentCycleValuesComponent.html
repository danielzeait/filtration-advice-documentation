import { Component, Input, OnInit } from '@angular/core';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-current-cycle-values',
  templateUrl: './current-cycle-values.component.html',
  styleUrls: ['./current-cycle-values.component.scss']
})
export class CurrentCycleValuesComponent implements OnInit {
  @Input('case_data') case_data:any;
  @Input('snapshot') snapshot:any;
  @Input('current_limit') current_limit:any;
  @Input('operation_chart_data') operation_chart_data:any;

  constructor(private utils:UtilsFunctions) { }

  ngOnInit(): void {}

  formatDate(date){
    return this.utils.formatDateStringMonth(date);
  }

  getHours(){
    return this.utils.substractDates(null);
  }

  

}
