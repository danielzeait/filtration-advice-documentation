import { Component, OnInit } from '@angular/core';
import { MetaDataProvider } from 'src/app/providers/metadata.service';

@Component({
  selector: 'app-case-actions',
  templateUrl: './case-actions.component.html',
  styleUrls: ['./case-actions.component.scss']
})
export class CaseActionsComponent implements OnInit {
  actions:any;
  constructor(
    private metadata:MetaDataProvider
  ) { }

  ngOnInit(): void {
    this.actions = this.metadata.getCaseActions();
  }

}
