import { Component, OnInit, ChangeDetectorRef, EventEmitter, Output } from '@angular/core';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { Filter } from 'src/app/models/filter.model';
import { FormControl } from '@angular/forms';
import {map, startWith, filter} from 'rxjs/operators';
import { Observable } from 'rxjs';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { FilterProvider } from 'src/app/providers/filter.service';
import { NotificationService } from 'src/app/providers/notifications.service';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-model-filter-form',
  templateUrl: './model-filter-form.component.html',
  styleUrls: ['./model-filter-form.component.scss']
})
export class ModelFilterFormComponent implements OnInit {

  @Output() done = new EventEmitter<any>();
  
  public manufacturerControl = new FormControl();
  public options:any={};
  public currentCheckedValue:any=null;
  public currentValueFaceScreen:any=null;
  public isIncinerable: string;
  public hasFaceScreen: string;
  public filter_manufacturer:any;
  public new_filter:Filter = new(Filter);
  public manufacturersOptions: Observable<string[]>;
  public manufacturers: any;
  public new_manufactuer:any;
  public filter_img_file: File;
  public filter_img_name:string=null;
  public filter_img_display: string | ArrayBuffer;
  new_object_type: any;
  createNewObject: boolean;
;

  constructor(
    private commonProvider:CommonProvider,
    private metadata:MetaDataProvider,
    private metaDataProvider:MetaDataProvider, 
    private cdr: ChangeDetectorRef,
    private filterProvider:FilterProvider,
  ) { }

  ngOnInit() {
    this.metaDataProvider.getFilterTypes()
    .subscribe((filter_types)=>{
      this.options.FILTER_TYPE = filter_types;
    });

    this.metaDataProvider.getFrameMaterial()
    .subscribe((frame_materials)=>{
      this.options.FRAME_MATERIAL = frame_materials;
    });

    this.metaDataProvider.getFrameTypes()
    .subscribe((frame_types)=>{
      this.options.FRAME_TYPE = frame_types;
    });

    this.metaDataProvider.getMediaTypes()
    .subscribe((media_types)=>{
      this.options.MEDIA_MATERIAL = media_types;
    });

    this.metaDataProvider.getFilterTypes()
    .subscribe((filter_types)=>{
      this.options.FILTER_TYPE = filter_types;
    });

    this.getManufacturers();
    
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  private _filter(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      return option
    }
    
  }

  getManufacturers(){
    this.metaDataProvider.getAllManufacturers()
    .subscribe((manufacturers)=>{
      this.manufacturers = manufacturers;
      this.manufacturersOptions = this.manufacturerControl.valueChanges.pipe(
        startWith(''),
        map(value => this._filter(value,this.manufacturers)))
    });
  }

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      var type = files[0].file.type;
      if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
        this.setErrorFile();
      } else {
        this.filter_img_file = files[0].file;
        this.filter_img_name = files[0].file.name;
        this.filter_img_display = files[0].imgURL;
      }
    }
  }

  fileSelectedChange(ev) {
    var type = ev.target.files[0].type
    if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
      this.setErrorFile();
    } else {
      this.filter_img_file = ev.target.files[0];
      this.filter_img_name = ev.target.files[0].name;
      const reader = new FileReader();
      reader.readAsDataURL(ev.target.files[0]);
      reader.onload = () => {
        this.filter_img_display = reader.result as string;
      };
    }
  }

  setErrorFile(){

  }

  _setManufacturer(value){
    this.new_filter.manufacturer = {...value};
  }

  _enbaleCreationForm(type){
    this.new_object_type = type
    this.createNewObject = true;
  }

  _saveNewFilter(){
    this.new_filter.short_name = this.new_filter.name;
    this.new_filter.picture = this.filter_img_display;
    this.filterProvider.createFilters(this.new_filter, this.filter_img_file).subscribe(
      (response)=>{
        this._cancel(response);
      }
    ), error =>{
        console.log(error);
    }
  }

  _saveNewManufacturer(){
    this.metadata.addManufacturer(this.new_manufactuer).subscribe(
      (response:any)=>{
        this.commonProvider.showSnackBar('Success',"Manufacturer added correctly",'success-snackbar');
        this.createNewObject = false;
        this.getManufacturers();
      },error=>{
        console.log(error);        
      }
    )
  }

  _cancel(object){
    this.done.emit(object);
  }

  _done(filter){
    this.done.emit(filter);
  }


}
