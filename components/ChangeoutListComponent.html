import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { CaseSelectorComponent } from '../modal-dialogs/case-selector/case-selector.component';
import { MatDialog } from '@angular/material/dialog';
import { AuthProvider } from 'src/app/providers/auth.service';
import { Width } from 'ngx-owl-carousel-o/lib/services/carousel.service';
import { CaseProvider } from 'src/app/providers/case.service';
import { emit } from 'process';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-changeout-list',
  templateUrl: './changeout-list.component.html',
  styleUrls: ['./changeout-list.component.scss']
})
export class ChangeoutListComponent implements OnInit {
  @Input('data') input_data:any;
  @Output() change_list = new EventEmitter(); 
  @Output() remove = new EventEmitter(); 
  public change_out_data:any;
  public change_out_list:any;
  public date:any;
  constructor(
    public dialog:MatDialog, 
    private navigator:NavigationProvider,
    private authProvider:AuthProvider, 
    private caseProvider:CaseProvider) { }

  ngOnInit() {
    if(this.input_data){
      this.change_out_data = [...this.input_data];
      this.getChangeOutDates();
    }
  }

  openCaseSelector(){
    let dialogRef = this.dialog.open(CaseSelectorComponent, {
      width: '70%',
      height: '91%',
      data:this.change_out_data ? this.change_out_data : null,
      id:'case-selector'
    });

    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.authProvider.getUserDashboard().subscribe((dashboard:any)=>{
          dashboard.find(widget => widget.type == "changeout").data = [...result];
          this.authProvider.setUserDashboard(dashboard).subscribe(
            (updated_dashboard:any)=>{
              this.change_out_data = [...updated_dashboard.find(widget => widget.type == 'changeout').data]; 
              this.getChangeOutDates();
          })
        })
      }
    });
  }

  getChangeOutDates(){
    var cases = [...this.change_out_data].join();
    if(cases.length > 0){
      this.caseProvider.getChangeOutList(cases).subscribe((change_out_list) => {
        this.change_out_list = [...change_out_list];
        this.change_out_list.map( item => item.already_past = new Date(item.replace_date.replace(' at','')) < new Date() )
        this.change_list.emit(this.change_out_data);
      })
    }
  }

  removeFromDashboard(){
    this.remove.emit();
  }

  viewCaseDetails(_case){
    this.navigator._goOptimizationsResults(_case);
  }

}
