import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material';
import { AuthProvider } from 'src/app/providers/auth.service';
import { FeedbackDialogComponent } from '../../feedback-dialog/feedback-dialog/feedback-dialog.component';

@Component({
  selector: 'app-instrument-help',
  templateUrl: './instrument-help.component.html',
  styleUrls: ['./instrument-help.component.scss']
})
export class InstrumentHelpComponent implements OnInit {

  public helpForm:FormGroup;
  public message_sent:boolean=false;
  public sending:boolean=false;
  public button_class:string='';

  constructor(
    private fb: FormBuilder, 
    private authService:AuthProvider,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private dialogref:MatDialogRef<InstrumentHelpComponent>
  ) { }

  ngOnInit(): void {
    this.setReviewForm();
  }

  setReviewForm(){
    this.helpForm = this.fb.group({
      username:new FormControl({value:this.data.first_name + " " + this.data.last_name,disabled:true}),
      email: new FormControl({value:this.data.email,disabled:true}),
      message:new FormControl({value:'',disabled:this.sending},Validators.required),
    });
  }

  sendMessage(){
    //this.reviewForm.disable()
    this.sending = true;
    this.button_class = 'animated fadeIn infinite';
    let data = {
      message:this.helpForm.controls.message.value,
      username:this.helpForm.controls.username.value,
    }

    this.authService.sendUserInstrumentHelp(data).subscribe(()=>{
      this.sending = false;
      this.button_class = '';
      this.message_sent = true;
    })
  }

  close(){
    this.dialogref.close();
  }

}
