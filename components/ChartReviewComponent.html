import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { ReportsProvider } from 'src/app/providers/reports.service';
import { MatSnackBar } from '@angular/material';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { ExportFileProvider } from 'src/app/providers/export-file.service';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-chart-review',
  templateUrl: './chart-review.component.html',
  styleUrls: ['./chart-review.component.scss']
})
export class ChartReviewComponent implements OnInit {
  public file_to_upload: any;
  public confirm_view: boolean = false;
  public error_title: string;
  public snack_message: string;
  public error_file: boolean;
  public data:any={
    area:2,
    duct_constant:0.00000021,
    time_differential:2,
    end_date:new Date(),
    start_date:new Date('07-07-2019')
  };
  public dp_histogram_data: any;
  public airflow_histogram_data: any;
  constructor(
    private commonProvider:CommonProvider,
    private handlerError:CustomErrorHandler, 
    private utils:UtilsFunctions,
    private exportFileProvider:ExportFileProvider,
    private reportProvider: ReportsProvider, private _snackBar: MatSnackBar) { }

  ngOnInit() {
    window.addEventListener("dragover", e => {
      e && e.preventDefault();
    }, false);
    window.addEventListener("drop", e => {
      e && e.preventDefault();
    }, false);
  }

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      this.file_to_upload = files[0].file;
      this.confirm_view = true;    
    }
  }

  fileSelectedChange(ev) {
    this.file_to_upload = ev.target.files[0];
    this.confirm_view = true;
  }

  setErrorFile() {
    this.error_title = "File not supported"
    this.snack_message = "Make sure that the file you are trying to upload is a 'PDF' document"
    this.error_file = true;
    this.showSnackNotification(5000);
  }

  showSnackNotification(time) {
    this.commonProvider.showSnackBar('Message',this.snack_message,'alert-snackbar');
  }

  _clearFile() {
    this.file_to_upload = '';
    this.confirm_view = false;
  }

  _consult(){
    this.data.start_date = this.utils.formatDate(this.data.start_date);
    this.data.end_date = this.utils.formatDate(this.data.end_date);
    this.reportProvider.getFrankCalculations(this.data,this.file_to_upload).subscribe(
    (results:any)=>{
      this.airflow_histogram_data = results.airflow_histogram;
      this.dp_histogram_data = results.dp_histogram;
    })
  }

  exportFile(title,type){
    let data:any=[];
    type == 'airflow' ? data = this.airflow_histogram_data : data = this.dp_histogram_data
    this.exportFileProvider.exportData(data,title);
  }
}
