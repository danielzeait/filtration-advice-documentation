import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { isNgTemplate } from '@angular/compiler';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-notifications-list',
  templateUrl: './notifications-list.component.html',
  styleUrls: ['./notifications-list.component.scss']
})
export class NotificationsListComponent implements OnInit {
  @Input('notifications') notifications:any;
  @Output() close = new EventEmitter();
  constructor(private navigator:NavigationProvider) { }

  ngOnInit() {
    if(this.notifications.length > 0 ){
      if(this.notifications.length > 5){
        this.notifications = this.notifications.slice(0,4)
      }
      this.notifications.map(item => item.body.length > 85 ? item.body = item.body.slice(0, 85) + '...' : item.body);
    }
    
  }
