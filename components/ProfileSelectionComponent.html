import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { PaymentProvider } from 'src/app/providers/payment.service';
import { SubscriptionDialogComponent } from '../modal-dialogs/subscription-dialog/subscription-dialog/subscription-dialog.component';

@Component({
  selector: 'app-profile-selection',
  templateUrl: './profile-selection.component.html',
  styleUrls: ['./profile-selection.component.scss']
})
export class ProfileSelectionComponent implements OnInit {
  @Input('plan') plan:any;
  @Input('edition') edition:any;
  @Output() done = new EventEmitter();

  profiles:any=[];
  case_types:any=[];
  type_of_case:any=null;
  monitoring_permission:boolean = false;
  constructor(
    public dialog: MatDialog,
    private metadataService:MetaDataProvider, 
    private paymentProvider:PaymentProvider, 
    private navigator:NavigationProvider
  ) {}

  ngOnInit() {
    this.profiles = this.metadataService.getProfiles();
    this.case_types = this.metadataService.getCaseTypes();
    this._getUserSubscriptions();
  }

  _getUserSubscriptions(){
    this.paymentProvider.getUserSubscriptions().subscribe((response)=>{
      let user_subscriptions = response;
      this.monitoring_permission = user_subscriptions.find(subscription => subscription.plan.slug == "unlimited_instrumented_system") ? true : false;
    })
  }

  setProfile(profile){
    if(this.type_of_case && this.type_of_case.type == 'monitor') profile.is_monitored = true;
    this.done.emit(profile);
  }

  setTypeOfCase(type){
    if(type.type == 'monitor'){
      if(this.monitoring_permission){
        this.type_of_case = type;
      }else{
        this._openSubscriptionModal();
      }
    }else{
      this.type_of_case = type;
    }
  }

  _openSubscriptionModal(){
    const dialogRef = this.dialog.open(SubscriptionDialogComponent, {
      width: '50%',
      height: 'auto',
    });
    dialogRef.afterClosed().subscribe(result => {
      if(result){
        this._openPaymentModalDialog();
      }
    });
  }

  _openPaymentModalDialog(): void {
    this.paymentProvider.getAllSubscriptions().subscribe((response)=>{
      let plans = response;
      let plan = plans.find(plan => plan.slug == 'unlimited_instrumented_system');
      plan.action = 'plans';
      this.navigator._goCheckOut(plan);
    })
  }

  changeTypeOfCase(){
    this.type_of_case = null;
    this.profiles.map(profile => profile.is_monitored = false);
  }

}
