import { Component, OnInit, Inject } from '@angular/core';
import {MatDialog, MatDialogRef, MAT_DIALOG_DATA} from '@angular/material/dialog';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'airflow-dialog',
  styleUrls: ['./air-flow-dialog.component.scss'],
  templateUrl: 'air-flow-dialog.component.html',
})
export class AirFlowDialogComponent implements OnInit{
  public total_airflow_error:boolean=false;
  public negative_value_error: boolean;
  public airflow_value_error:boolean=false;
  public max_flow_limit:number;
  constructor(
    public dialogRef: MatDialogRef<AirFlowDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any, public utils : UtilsFunctions) {}

  onNoClick(): void {
    this.dialogRef.close();
  }

  
  ngOnInit(){
    if(this.data.airflows.length == 1){
      this.data.airflows[1] = {'flow':parseFloat('118').toFixed(2),'percent':0};
      this.data.airflows[2] = {'flow':parseFloat('118').toFixed(2),'percent':0};
      this.data.airflows[3] = {'flow':parseFloat('118').toFixed(2),'percent':0};
    }
    if(this.data.airflows.length == 2){
      this.data.airflows[2] = {'flow':parseFloat('118').toFixed(2),'percent':0};
      this.data.airflows[3] = {'flow':parseFloat('118').toFixed(2),'percent':0};
    }
    if(this.data.airflows.length == 3){
      this.data.airflows[3] = {'flow':parseFloat('118').toFixed(2),'percent':0};
    }
    this.max_flow_limit = this.data.airflow_limit;
  }   

  setAmountDecimal(ev){
    this.utils.transformAmount(ev);
  }
  
  done():void {
    let sum_airflow = this.data.airflows.reduce((a, b) => +a + +b.percent, 0);
    if(sum_airflow != 100){
      this.total_airflow_error = true;
    }else if(this.data.airflows[0].percent < 0 || this.data.airflows[1].percent < 0 || this.data.airflows[2].percent < 0 || this.data.airflows[3].percent < 0){
      this.negative_value_error = true;
    }else if(this.data.airflows[0].flow < 118 || this.data.airflows[1].flow < 118 || this.data.airflows[2].flow < 118 || this.data.airflows[3].flow < 118){
      this.airflow_value_error = true;
    }else if(this.data.airflows[0].flow > this.max_flow_limit || this.data.airflows[1].flow > this.max_flow_limit || this.data.airflows[2].flow > this.max_flow_limit || this.data.airflows[3].flow > this.max_flow_limit){
      this.airflow_value_error = true;
    }else{
      
      let average_airflow = this.data.airflows.reduce((a, b) => +a + +(b.flow * (b.percent/100)), 0).toFixed(2);
  
      let return_data = {
        flows:this.data.airflows,
        average:average_airflow
      }
      this.dialogRef.close(return_data);
    }
  }
  
}
