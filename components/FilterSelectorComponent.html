import { Component, OnInit, Output, EventEmitter, Input, HostListener } from '@angular/core';
import { FilterProvider } from 'src/app/providers/filter.service';
import { Filter } from 'src/app/models/filter.model';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { clear_search_data } from "../../mocks/mocks.objects";
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { RequestFilterComponent } from '../modal-dialogs/request-filter/request-filter/request-filter.component';
import { MatDialog } from '@angular/material/dialog';

@Component({
  selector: 'app-filter-selector',
  templateUrl: './filter-selector.component.html',
  styleUrls: ['./filter-selector.component.scss']
})
export class FilterSelectorComponent implements OnInit {
  @Input('filter') input_filter:any;
  @Input('review') review:any;
  @Output() done = new EventEmitter<any[]>();
  @Output() prevent_next = new EventEmitter<any[]>();
  @Output() allow_next = new EventEmitter<any[]>();
  @Output() report_set = new EventEmitter<any[]>();

  private advanced_search_data: any = {};

  public searchedWordsControl = new FormControl('');
  public show_advanced_filter: boolean;
  public url_search_data: any;
  public search_models: any;
  public clear_search: any;
  public num_cols: number = 3;
  


  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
    map((search: string) => search.trim().split(' '))
  )
  public isImgLoaded:any; 
  public showFilterDetails:boolean=false;
  public filters: Array<Filter> = new Array<Filter>();
  public filters_search_result: Array<Filter> = [];
  public original_filters:any;
  public search: string = "";
  public filter_selected: any;
  constructor(private filterService:FilterProvider,private responsiveProvider:ResponsiveProvider,public dialog: MatDialog) { }

  ngOnInit() {
    if(this.input_filter){
      this._openFilterDetails(this.input_filter); 
    }
    Object.assign(this.advanced_search_data, clear_search_data);
    this.filterService.getAllPublicFilters().subscribe(this.setFilters);
    //this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    //this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }


  setFilters = (filters: Array<Filter>) => {
    this.filters = [...filters];
    this.original_filters = [...filters];
  }

  ngAfterViewInit() {
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
      //this.filters = this.filterService.filterFilterListHard(this.original_filters, changes);
    })
  }

  _openSearchModal(): void {
    this.show_advanced_filter = !this.show_advanced_filter;
  }

  searchBywords(){
    if(this.filters_search_result.length > 0){
      this.filters = [...this.filterService.filterFilterListHard(this.filters_search_result, this.search)]
    }else if(this.original_filters){
      this.filters = [...this.filterService.filterFilterListHard(this.original_filters, this.search)]  
    }
  }

  refreshFilters(result){
    this.filters = [...result.filters];
    this.filters_search_result = [...result.filters];
    this.advanced_search_data = {...result.data};
    this.url_search_data = result.url;
    this.search_models = {...result.model};
    this.show_advanced_filter = false;
  }

  clearFiltersFilter() {
    this.filters = [...this.original_filters];
    this.advanced_search_data = {...clear_search_data};
    this.url_search_data = '';
    this.search_models = {
      filter_type:[],
      media_type:[],
      manufacturer:[],
      frame_type:[],
      frame_material:[],
      dust_type:[],
      laboratory:[],
      provider:[],
      standard:[],
    };
    this.filters_search_result = [];
    this.clear_search = {...{clear_me:true}};
    this.show_advanced_filter = false;
  }

  requestFilter(){
    const dialogRef = this.dialog.open(RequestFilterComponent, {
      width: '35%',
      height: '75%',
      id: 'request-filter',
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {}
    });
  }
  
  prevent(ev){
    if(ev){
      this.prevent_next.emit()
    }else{
      this.allow_next.emit();
    }
  }

  restrieveReports(event){
    this.report_set.emit(event);
  }

  _openFilterDetails(filter) {
    this.filter_selected = {...filter};
    this.showFilterDetails = true;
  }

  filterSet(event){
    this.done.emit(event);
  }

}
