import { Component, OnInit, HostListener, ViewChild, ElementRef } from '@angular/core';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { NavigationEnd, Router } from '@angular/router';

@Component({
  selector: 'app-product',
  templateUrl: './product.component.html',
  styleUrls: ['./product.component.scss']
})
export class ProductComponent implements OnInit {
  show_section_1: boolean;
  show_section_2: boolean;
  show_section_3: boolean;
  show_section_4: boolean;

  constructor(private navigation:NavigationProvider, private router:Router) { }
  
  @HostListener('window:scroll', ['$event']) 
  doSomething(event) {
      let e1 = document.getElementById('method1');
      let e2 = document.getElementById('method2');
      let e3 = document.getElementById('method3');
      let e4 = document.getElementById('method4');

      let section1 = e1.parentElement.offsetTop;  
      let section2 = e2.parentElement.offsetTop + e1.offsetHeight;
      let section3 = e3.parentElement.offsetTop + (e1.offsetHeight * 2) - 100;   
      let section4 = e4.parentElement.offsetTop + e1.offsetHeight + (e2.offsetHeight * 2 ) + 400;
      var position =  parseInt((window.innerHeight + window.pageYOffset/1.3).toFixed(0));
      if(position >= section1 - 20 && position <= section1 + 20){
        this.show_section_1 = true;
      }
      if(position >= section2 - 20 && position <= section2 + 20){
        this.show_section_2 = true;
      }
      if(position >= section3 - 20 && position <= section3 + 20){
        this.show_section_3 = true;
      }
      if(position >= section4 - 20 && position <= section4 + 20){
        this.show_section_4 = true;
      }
    }



  ngOnInit(){
    this.router.events.subscribe((evt) => {
      if (!(evt instanceof NavigationEnd)) {
          return;
      }
      window.scrollTo(0, 0)
  });
    // var viewportOffset = this.testElement.nativeElement;
    // console.log(viewportOffset);
  }


  _goLogin(){
    this.navigation._goLogin();
  }

  _goRegister(){
    this.navigation._goRegister();
  }

}
