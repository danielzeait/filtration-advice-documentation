import { Component, OnInit, AfterViewInit, Input, Output, EventEmitter, SimpleChanges } from '@angular/core';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-filter-details',
  templateUrl: './filter-details.component.html',
  styleUrls: ['./filter-details.component.scss']
})
export class FilterDetailsComponent implements OnInit {
  public searchedWordsControl = new FormControl('');
  @Input('filter') filter: any;
  @Input('report') report= [];
  @Input('review') review:boolean;
  @Input('hide_reports') hide_reports:boolean;
  @Input('can_edit_size') can_edit_size:boolean;
  @Input('is_for_case') is_for_case:boolean = false;
  @Output() onSelectReport = new EventEmitter<any[]>();
  @Output() onNormChange = new EventEmitter<any[]>();
  @Output() close = new EventEmitter<any[]>();
  @Output() done = new EventEmitter();
  @Output() prevent = new EventEmitter();
  @Output() set_report = new EventEmitter();
  @Output() changeFilter = new EventEmitter<any[]>();
  public showFilterReports = false;
  public showFilterReportsPrivate;
  public filter_case_size_width = 24;
  public filter_case_size_height = 24;
  public data: any;
  public show_more:boolean = false;
  public existing_case: any;
  public filter_detail: any;
  public norms;
  public owner;
  public std;
  public norm;
  public private;
  public show_sellers:boolean = false;
  public filter_reports:any=[];
  public set_size:boolean = false;

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl.valueChanges.pipe(map((showFilterReportsPrivate: string) => showFilterReportsPrivate.trim().split(' '))
    );


  constructor(private navigator: NavigationProvider, public metaDataProvider: MetaDataProvider, private handlerError: CustomErrorHandler) { }

  ngOnInit() {
    this.existing_case = localStorage.getItem('new_case_data');
    const data = {
      filter: this.filter ? this.filter.id : null,
      owner: null
    };
    
    this._getNormsFilter(data);
    this.filter.case_size_depth = this.filter.size_depth;
    this.filter.case_size_height ? this.filter_case_size_height = this.filter.case_size_height : this.filter_case_size_height = this.filter.size_height;
    this.filter.case_size_width ? this.filter_case_size_width = this.filter.case_size_width : this.filter_case_size_width = this.filter.size_width;
    this.attemptToFinishFilterSelection();
    this.filter_reports = this.filter.reports.private.concat(this.filter.reports.public);
  }

  ngAfterViewInit() {
    if(this.filter.reports.public.length > 0){
      this.showFilterReportsPrivate='1';
      const data = {
        filter: this.filter.id,
        owner: false
      };
      this._getNormsFilter(data);
    } else {
      this.showFilterReportsPrivate='2';
      const data = {
        filter: this.filter.id,
        owner: true
      };
      this._getNormsFilter(data);
    }
    this.searchedWordsControl.valueChanges.subscribe((changes) => {
      if (changes == 1) {
        this.private = false;
        this.data = {
          filter: this.filter.id,
          owner: false
        };
      this.metaDataProvider.getNormsFilter(this.data).subscribe((response) => {
        this.norms = response;
      });
      } else if(changes == 2){

        this.private = true;
        this.data = {
          filter: this.filter.id,
          owner: true
        };
        this.metaDataProvider.getNormsFilter(this.data).subscribe((response) => {
        this.norms = response;
      });
      } else {
     
      }
    });
  }

  onOwnerChange(owner){
    if(owner==1){
      this.private= false;
      this.metaDataProvider.setOwn(this.private);
      this.showFilterReportsPrivate='1';
      const data = {
        filter: this.filter.id,
        owner: false
      };
      this._getNormsFilter(data);
    } else if (owner==2){
      this.private= true;
      this.metaDataProvider.setOwn(this.private);
      this.showFilterReportsPrivate='2';
      const data = {
        filter: this.filter.id,
        owner: true
      };
      this._getNormsFilter(data);
    } else {
      this.private= null;
      this.metaDataProvider.setOwn(this.private);
    }
    const data = {
      filter: this.filter.id,
      owner: this.private
    };
    this._getNormsFilter(data);
  }

  getDisabledStart(showFilterReportsPrivate) {
    if(showFilterReportsPrivate == '2' && this.report.length < 1){
      return true;
    }else{
      return false;
    }
  }

  toggleInfo(){
    this.show_more = !this.show_more;
  }


  _getNormsFilter(data) {
    this.metaDataProvider.getNormsFilter(data).subscribe((response) => {
        this.norms = response;
        if(this.norms.length > 0){
          this.std= this.norms[0];
        } else {
          this.showFilterReportsPrivate='2';
          const data = {
            filter: this.filter.id,
            owner: true
          };
          this.metaDataProvider.getNormsFilter(data).subscribe((response) => {
            this.norms = response;
            this.std= this.norms[0];
            this.showFilterReportsPrivate='2';
          })
        }
      }, error => {
        this.handlerError.handleError(error);
      });
  }

  setReports(e) {
     this.report = e;
     this.set_report.emit(this.report);
     var no_report_selected = (this.report.length == 0 && this.filter.reports.public.length == 0);
     no_report_selected ? this.prevent.emit(true) : this.attemptToFinishFilterSelection();
  }

  onStandardChange(standard){
    this.std = standard;
    this.metaDataProvider.setStd(this.std);
  }

  setFilterSize(){
    this.set_size = true;
  }


  selectOwner(ow) {
    if (ow == 1) {
    this.owner = false;
    } else {
    this.owner = true;
    }
  }

  _startNewCase() {
    this.navigator._goNewCaseWithFilter(this.filter, this.report, this.std);
  }

  _backToFilterList(){
    this.close.emit();
  }

  contact(){
    this.navigator._contactUs();
  }

  changeFilterSelection(){
    this.changeFilter.emit();
  }

  attemptToFinishFilterSelection(){
    var no_report_selected = (this.report.length == 0 && this.filter.reports.public.length == 0);

    if((this.filter_case_size_width && (this.filter_case_size_width < 12 || this.filter_case_size_width > 24)) || 
       (this.filter_case_size_height && (this.filter_case_size_height < 12 || this.filter_case_size_height > 24)) ||
       !this.filter_case_size_width || !this.filter_case_size_height || (this.showFilterReportsPrivate == 2 && this.report.length == 0) || no_report_selected){
      this.prevent.emit(true);
    }else{
      this.prevent.emit(false);
      this.continue();
    }
  }

  continue(){
    this.filter.case_size_width = this.filter_case_size_width;
    this.filter.case_size_height = this.filter_case_size_height;
    this.done.emit(this.filter);
    if(this.report.length == 0){
      this.set_report.emit(this.filter.reports.public);
    }
  }

}
