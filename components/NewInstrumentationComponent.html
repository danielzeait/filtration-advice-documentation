import { Component, OnInit, ErrorHandler } from '@angular/core';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { CaseProvider } from 'src/app/providers/case.service';
import { System } from 'src/app/models/system.model';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { InstrumentProvider } from 'src/app/providers/instrument.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { PairedCaseInstrumentProvider } from 'src/app/providers/paired-case-instrument.service';
import { NotificationService } from 'src/app/providers/notifications.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { CommonProvider } from 'src/app/providers/common.service';
import { PaymentProvider } from 'src/app/providers/payment.service';

@Component({
  selector: 'app-new-instrumentation',
  templateUrl: './new-instrumentation.component.html',
  styleUrls: ['./new-instrumentation.component.scss']
})
export class NewInstrumentationComponent implements OnInit {
  public isLinear = false;
  public cases:any=[];
  public firstFormGroup: FormGroup;
  public secondFormGroup: FormGroup;
  public length:number;
  public pageSize = 20;
  public pageSizeOptions: number[] = [20,40,60,80,100];
  public jumpToPageOption:any;
  public list_url:any;
  public bottom_paginator_data:any = {
    previousPageIndex: 0,
    pageIndex: 0,
    pageSize: 20,
    length: 0
  }
  public original_cases: any;
  public original_case: any;
  public unavailable_slot: boolean;
  public data: any;
  public rectangle: any = true;
  public circle: boolean;
  public system_paired: any = {
    drop_pressure_inst:null,
    flow_inst:null,
    filterless_dp: 0.02,
    airflow_without_filter:118,
    airflow_instrument_coefficient:0.0000027,
    is_circular_pipe:false,
    duct_diameter:1,
    duct_height:24,
    duct_width:24,
    installation_date:'',
  };
  
  public outside_conected:boolean=false;
  public directly_connected:boolean=true;
  public maxDate:Date= new Date();
  public dp_instrument:any;
  public airflow_instrument:any;
  public instruments:any=[];
  public plan:any;
  step: any;

  constructor(
    private _formBuilder: FormBuilder, 
    private utils:UtilsFunctions,
    private caseProvider:CaseProvider, 
    private navigator:NavigationProvider,
    private pairCaseInstrumentProvider:PairedCaseInstrumentProvider,
    private notificationProvider:NotificationService,
    private commonProvider:CommonProvider,
    private paymentProvider:PaymentProvider,
    private instrumentProvider:InstrumentProvider,
    private handlerError:CustomErrorHandler) {}

  ngOnInit() {
    this.firstFormGroup = this._formBuilder.group({
      firstCtrl: ['', Validators.required]
    });
    this.secondFormGroup = this._formBuilder.group({
      secondCtrl: ['', Validators.required]
    });
    this.getInstrumentList();
    this.getCases();
    this.getPlans();
  }

  getPlans(){
    this.paymentProvider.getAllProducts().subscribe((response)=>{
    this.plan = response.find(plan => plan.slug == "optimization_update")
    })
  }

  getInstrumentList(){
    this.instrumentProvider.getAllInstrument().subscribe(
      (instrument_list) =>  {
        this.instruments = instrument_list.filter(instrument => !instrument.paired_with && instrument.units == 'inWC' );
      }
    );
  }

  getCases(){
    this.caseProvider.getPersonalCases().subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    });
  }

  setCases = (cases: Array<System>) => {
    this.cases = cases.filter(study_case => !study_case.is_instrumented);
    this.original_cases = cases;
  }

  setPaginatorOptions(cases){
    this.length = cases.count;
    this.bottom_paginator_data.length = cases.count;
  }

  _getCaseListByUrl(url) {
    this.caseProvider.getCasesByUrl(url).subscribe((cases)=>{
      this.setCases(cases.results);
      this.setPaginatorOptions(cases);
    },error=>{
      this.handlerError.handleError(error);
    });
  }

  setDefaultCaseSelected(){
    this.original_case = null;
    this.cases = this.cases.map(function(e) {
      e.selected ? e.selected = true : e.selected = false;
      return e;
    });
  }

  setOriginalCase(study_case, index){
    this.unavailable_slot = false;
    this.clearSelectedCase();
    this.cases[index].selected = true;
    this.original_case = study_case;
    if(study_case.is_instrumented){
      this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
        (case_details)=> {
          if(case_details.monitoring_module.drop_pressure_inst){
            this.setDpInstrumentData(case_details);
          }
          if(case_details.monitoring_module.flow_inst){
            this.setAirFlowIntrumentData(case_details);
          }
          if(case_details.monitoring_module.flow_inst && case_details.monitoring_module.drop_pressure_inst && this.data.units == 'inWC'){
            this.unavailable_slot = true;
          }
        }
      )
    }
  }

  setAirFlowIntrumentData(case_details){
    this.system_paired.duct_diameter = case_details.duct_diameter;
    this.system_paired.is_circular_pipe = case_details.is_circular_pipe;
    this.system_paired.duct_width = case_details.width;
    this.system_paired.duct_height = case_details.height;
    this.system_paired.airflow_without_filter = case_details.airflow_without_filter;
    this.system_paired.airflow_instrument_coefficient = case_details.monitoring_module.airflow_instrument_coefficient;
  }

  setDpInstrumentData(case_details){
    this.system_paired.filterless_dp = case_details.filterless_dp;
    this.system_paired.installation_date = case_details.installation_date;
  }

  setAmountDecimal(ev) {
    this.utils.transformAmount(ev);
  }

  setDuctShape(shape){
    if(shape == 'rectangle'){
      !this.rectangle ? this.circle = true : this.circle = false
      this.system_paired.is_circular_pipe = false;  
    }else{
      this.system_paired.is_circular_pipe = true;
      !this.circle ? this.rectangle = true : this.rectangle = false
    }
  }

  handlePage(event){
    this._getCaseListByUrl(event.url);
  }

  pairInstruments(event) {
    this.original_case.monitoring_module = event.data;
    if(this.original_case.current_filter.is_already_updated){
      delete this.original_case.current_filter;
      this.plan.action = 'update-optimization';
      localStorage.setItem('pending_optimization',JSON.stringify(this.original_case));
      localStorage.setItem('pending_instrumentation',JSON.stringify(event.data));
      localStorage.setItem('new_case_data',JSON.stringify(this.original_case));
      this.navigator._goCheckOut(this.plan);
    }else{
      this.original_case.monitoring_module = event.data;
      this.original_case.is_paid_with_credits = true;
      delete this.original_case.current_filter;
      this.caseProvider.patchCase(this.original_case).subscribe(
      (response) => {
        this.commonProvider.showSnackBar('Case monitored','Your case now is being monitored','success-snackbar');
        this.navigator._goInstrumentList();
      }, error => {
        this.handlerError.handleError(error);
      });
    }
  }

  setStep(step){
    this.step = step;
  }

  cancelInstrumentation(){
    this.clearSelectedCase();
    this.original_case = null;
    this.navigator._goInstrumentList();
  }


  clearSelectedCase(){
    this.cases = this.cases.map(function(e) {
      e.selected = false;
      return e;
    });
  }
}
