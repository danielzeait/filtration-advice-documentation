import { Component, OnInit, Input, SimpleChanges} from '@angular/core';
import { CaseProvider } from 'src/app/providers/case.service';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-table-cost-summary',
  templateUrl: './table-cost-summary.component.html',
  styleUrls: ['./table-cost-summary.component.scss']
})
export class TableCostSummaryComponent implements OnInit {
  @Input('filter_data') filter_data:any;
  
  @Input('case_id') case_id:any;

  @Input('id') table_id:any;

  @Input('filter_id') filter_id:any;

  @Input('is_proposed') is_proposed:boolean;

  @Input('is_snapshot') snapshot:boolean;

  @Input('replacement_date') replacement_date:any;

  @Input('selected_data') selected_data:any;

  @Input('installation_date') installation_date:any;

  @Input('consult') consult:any;

  public total_table_costs:any;

  public selected_costs:any;

  constructor(private utils:UtilsFunctions, private caseProvider:CaseProvider) { }

  ngOnInit(): void {
    this._calculateTotalCosts();
  }

  ngOnChanges(changes:SimpleChanges){
    if(changes.consult && !changes.consult.firstChange && changes.consult.currentValue){
      if (this.selected_data) this.getSelectedCosts(this.selected_data.change_out_time);
    }

    if(changes.selected_data && !changes.selected_data.firstChange){
      changes.selected_data.currentValue ? this.getSelectedCosts(changes.selected_data.currentValue.change_out_time) : this.selected_costs = [];
      //if (this.selected_data) this.getSelectedCosts(this.selected_data.change_out_time);
    }

    if(changes.filter_data && !changes.filter_data.firstChange && changes.filter_data.currentValue){
      this._calculateTotalCosts();
    }
  }


  _calculateTotalCosts(){
    this.total_table_costs = {
      total_current_cost_per_cycle:this.filter_data.costs.reduce((a, b) => +a + +b.per_cycle.current, 0).toFixed(2),
      total_optimized_cost_per_cycle:this.filter_data.costs.reduce((a, b) => +a + +b.per_cycle.optimized, 0).toFixed(2),
      total_current_cost_annualized:this.filter_data.costs.reduce((a, b) => +a + +b.per_year.current, 0).toFixed(2),
      total_optimized_cost_annualized:this.filter_data.costs.reduce((a, b) => +a + +b.per_year.optimized, 0).toFixed(2),
      total_selected_cost_per_cycle:null,
      total_selected_cost_annualized:null,
    };
  }

  getSelectedCosts(time?){
    this.caseProvider.getCostsByTime(this.is_proposed ? this.filter_id : this.case_id,time,this.is_proposed,this.snapshot).subscribe(
      selected_costs => {
        this.selected_costs = selected_costs;
        this.total_table_costs.total_selected_cost_per_cycle = this.selected_costs.reduce((a, b) => +a + +b.per_cycle.selected, 0).toFixed(2);
        this.total_table_costs.total_selected_cost_annualized = this.selected_costs.reduce((a, b) => +a + +b.per_year.selected, 0).toFixed(2);
      }
    )
  }

  getChangeOutDate(date){
    return this.utils.getFutureDateByMonths(this.installation_date,date);
  }

  formatTitle(string:String){
    return this.utils.formatStringTitle(string);
  }


}
