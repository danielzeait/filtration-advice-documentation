import { Component, OnInit, Input, SimpleChanges, AfterViewInit, Output, EventEmitter } from '@angular/core';
import Chart from 'chart.js';
import { max } from 'rxjs/operators';

@Component({
  selector: 'app-chart-cost-vs-time',
  templateUrl: './chart-cost-vs-time.component.html',
  styleUrls: ['./chart-cost-vs-time.component.scss']
})
export class ChartCostVsTimeComponent implements AfterViewInit {
  @Input('chart_data') chart_data : any;  
  @Input('limit_dp') limit_dp : any; 
  @Input('chart_id') chart_id : any;
  @Input('scale') scale : any;
  @Input('selected_values') selected_values : any; 
  @Input('current_limit') current_limit : any;
  @Output() has_limit = new EventEmitter<number>();
  private already_fixed_y_axis:boolean=false;
  private chartData: any = [];
  private chart: any;
  private verticalLimitChartData: any;
  private max_cost:any;
  private variableData: { 'x': any; 'y': any; }[];
  currentVerticalLimit: { x: any; y: any; }[];
  passData: any;
  constructor() { }

  ngAfterViewInit() {
    this.chartData = this.setChartData();
    if(this.selected_values){
      this.variableData = [{
        'x':Number.parseFloat(this.selected_values.change_out_time).toFixed(2),
        'y':Number.parseFloat(this.selected_values.annual_cost).toFixed(2)
      }]
    }
    this.initChart();
  }
  
  ngOnChanges(changes: SimpleChanges){
    var update = false;
    if(changes.scale && !changes.scale.firstChange){
      if(changes.scale.currentValue.use){
        this.chart.options.scales.yAxes[0].ticks.max = changes.scale.currentValue.limit_y ? changes.scale.currentValue.limit_y : this.max_cost ;
        this.chart.options.scales.xAxes[0].ticks.max = changes.scale.currentValue.limit_x;
        update = true
      }else{
        this.chart.options.scales.yAxes[0].ticks.max = this.max_cost;
        this.chart.options.scales.xAxes[0].ticks.max = Math.ceil(this.chartData[this.chartData.length-1].x);
        update = true
      }
    }

    if(changes.selected_values && !changes.selected_values.firstChange){
      this.selected_values = changes.selected_values.currentValue;
      this.variableData = [{
        'x':Number.parseFloat(this.selected_values.change_out_time).toFixed(2),
        'y':Number.parseFloat(this.selected_values.annual_cost).toFixed(2)
      }];
      this.chart.data.datasets[0].data[0] = this.variableData[0];
      update = true
    }
    
    if(changes.chart_data && !changes.chart_data.firstChange){
      this.chart_data = changes.chart_data.currentValue;
      this.chartData = this.setChartData();
      this.chart.data.datasets[2].data[0] = this.chartData;
      update = true
    }

    if(update){
      this.chart.update();
    }
  }

  setChartData(){
    var chartData = [];
    for (let i = 0; i < this.chart_data.length; i++) {
      var point = {
        'x':Number.parseFloat(this.chart_data[i].change_out_time).toFixed(2),
        'y':Number.parseFloat(this.chart_data[i].annual_cost).toFixed(2)
      }
      if(this.chart_data[i].is_optimal){
        this.max_cost = Math.floor(this.chart_data[i].annual_cost * 5);
      }
      if(this.chart_data[i].is_at_max_operational){
        if(!this.max_cost){
          this.max_cost = Math.floor(this.chart_data[i].annual_cost * 5);
        }
        this.verticalLimitChartData = [
          {x:Number.parseFloat(this.chart_data[i].change_out_time).toFixed(2),y:this.max_cost},
          {x:Number.parseFloat(this.chart_data[i].change_out_time).toFixed(2),y:0},
        ]
      }

      chartData.push(point); 

      if(this.current_limit && this.chart_data[i].is_current){
        this.passData = chartData.slice(0,i+1);
      }
      if(this.current_limit){
        this.currentVerticalLimit = [
          { x:Number.parseFloat(this.current_limit.change_out_time).toFixed(2),y:0},
          { x:Number.parseFloat(this.current_limit.change_out_time).toFixed(2),y:this.max_cost},
        ]
      }
    }
    this.has_limit.emit(this.max_cost);
    return chartData
  }

  updateChartTicks(scale) {
    setTimeout(() => {
      if(!this.already_fixed_y_axis){
        this.already_fixed_y_axis = true;
        this.chart.options.scales.yAxes[0].ticks.max = scale[0];
        this.chart.update();
      }
    }, 250); 
  };

  initChart(){
    var id = this.chart_id;
    var ctx:HTMLElement = <HTMLElement> document.getElementById(id);
    ctx.style.backgroundColor = '#FFF';
    ctx.style.padding = '10px';

    var canvas : any = document.getElementById(id);
    var ctx2 = canvas.getContext("2d");
    let y_limit:number;
    // var originalController = Chart.controllers.line;
    // Chart.controllers.line = Chart.controllers.line.extend({
    //   draw: function() {
    //     originalController.prototype.draw.call(this, arguments);
    //       var chartInstance = this.chart;
    //       var meta = chartInstance.controller.getDatasetMeta(2);
    //       var len = meta.data.length-1;
    //       var xOffset = meta.data[len]._model.x - 110;
    //       ctx2.font = "10px Verdana";
    //       ctx2.fillStyle = "black";
    //       ctx2.fillText("Operational Dp Limit", xOffset, 50);
    //   }
    // });


    this.chart = new Chart(ctx2, {
        type: 'line',
        pointBackgroundColor: "#55bae7",
        pointBorderColor: "#55bae7",
        data: {
            datasets: [
              {
                data:this.variableData,
                pointRadius:'6',
                borderColor:'#1C036F',
                borderWidth:2,
                backgroundColor:'#c7d42d'
              },
              {
                label: 'Cost vs Time',
                data: this.passData,
                backgroundColor:'#d5d1de',
                borderWidth:1,
                lineTension:0,
                borderColor:'#1C036F',
                pointBackgroundColor: "#222",
                pointBorderColor: "#222",
                pointRadius:'1'
              },
              {
                label: 'Cost vs Time',
                data: this.chartData,
                backgroundColor:'#f2eff9',
                borderWidth:1,
                lineTension:0,
                borderColor:'#1C036F',
                pointBackgroundColor: "#222",
                pointBorderColor: "#222",
                pointRadius:'1'
              },
              {
                data:this.verticalLimitChartData,
                pointRadius:'0',
                borderColor:'#ffa629',
                borderWidth:5,
                fill:false,
              },
              // {
              //   data:this.currentVerticalLimit,
              //   pointRadius:'0',
              //   borderColor:'#003A70',
              //   borderWidth:5,
              //   fill:false,
              // }
            ]
        },
        options: {
          elements: {
            line: {
              tension: 0.4
            }
          },
          legend:{
            display: false,
          },
          tooltips:{
            callbacks:{
                label:function(tooltipItem, data) {
                    var value = Number.parseFloat(tooltipItem.yLabel).toFixed(2);
                    var parts = value.toString().split(".");
                    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                    var final_value = (parts.join("."))
                    var label = 'Cost: $ ' + final_value;
                    return label;
                },
                title: function(tooltipItem){
                  var value = Number.parseFloat(tooltipItem[0].xLabel).toFixed(0);
                  return value + ' months';
                }
            },
          },
          scales: {
              xAxes: [{
                  type: 'linear',
                  position: 'bottom',
                  scaleLabel: {
                    display: true,
                    labelString: 'Time in months',
                    fontColor:'#222',
                    fontSize:10,
                  },
                  ticks: {
                    callback: function(value, index, values) {
                      return Number.parseFloat(value).toFixed(1);
                    },
                    //stepSize: 2,
                    fontSize:10,
                  },
                },
              ],
              yAxes: [{
                scaleLabel: {
                  display: true,
                  labelString: 'Cost per Year (USD)',
                  fontColor:'#222',
                  fontSize:10,
                },
                ticks: {
                  beginAtZero: true,
                  fontSize:10,
                  //stepSize:0.5
                  callback: function(value, index, values) {
                    var parts = Number.parseFloat(value).toFixed(2).toString().split(".");
                    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                    return parts[0];
                  },
                  max:this.max_cost
                },
                afterBuildTicks: (scale) => {
                  var val_1 = (scale.ticks[scale.ticks.length-1]);
                  var val_2 = (scale.ticks[scale.ticks.length-2]);
                  var step = val_2 - val_1;
                  var max_value = scale.ticks[1] + step;
                  var new_scale  = [...scale.ticks];
                  new_scale[0] = max_value;
                  this.updateChartTicks(new_scale);
                  return;
                },
                beforeUpdate: function(oScale) {
                  return;
                },
              },
            ],
          }
        }
    });

  }

}
