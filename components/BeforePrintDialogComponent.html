import { Component, Inject, OnInit } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material';
import { concat } from 'rxjs-compat/operator/concat';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { Company } from 'src/app/models/company.model';
import { User } from 'src/app/models/user_data.model';
import { AuthProvider } from 'src/app/providers/auth.service';
import { CommonProvider } from 'src/app/providers/common.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { SettingsProvider } from 'src/app/providers/settings.service';
import { RedirectDialogComponent } from '../../redirect-dialog/redirect-dialog.component';

@Component({
  selector: 'app-before-print-dialog',
  templateUrl: './before-print-dialog.component.html',
  styleUrls: ['./before-print-dialog.component.scss']
})
export class BeforePrintDialogComponent implements OnInit {

  public auto_fill:boolean = false;
  public include_reports:boolean = false;
  public date:any;
  public confirm_client_image:boolean=false;
  public client_file:any;
  public client_image:any;
  public options:any = {
    report_date:new Date(),
    include_company_info:false,
    company_info_type:null,
    include_client_data:false,
    include_generated_data:true,
    include_data_sensors:true,
    include_input_data:true,
    include_filter_details:true,
    include_reports:true,
    include_filters_channels:false,
    include_comments:true,
    report_type:'current',
    client_data:{},
    comments:'',
  };
  constructor(
    @Inject(MAT_DIALOG_DATA) public data: any,
    public dialogRef: MatDialogRef<BeforePrintDialogComponent>,
    private commonProvider:CommonProvider,
    private settingsProvider:SettingsProvider,
    private authProvider:AuthProvider,
    private handlerError:CustomErrorHandler,
  ) { }

  async ngOnInit(): Promise<void> {
    this.date = new Date();
    this.setEmptyClient();
    if(this.data.case.current_filter.model_filter.references){
      for (let i = 0; i < this.data.case.current_filter.model_filter.references.length; i++) {
        let image = await this.getSellerImage(this.data.case.current_filter.model_filter.references[i].image);
        this.data.case.current_filter.model_filter.references[i].pdf_image = image;
      }
    }

    if(this.data.proposed_filter && this.data.proposed_filter.model_filter.references && this.data.proposed_filter.model_filter.references.length > 0){
      for (let i = 0; i < this.data.proposed_filter.model_filter.references.length; i++) {
        let image = await this.getSellerImage(this.data.proposed_filter.model_filter.references[i].image);
        this.data.proposed_filter.model_filter.references[i].pdf_image = image;
      }
    }
  }

  removeCompanyOptions(){
    if(!this.options.include_company_info){
      this.options.company_info_type = null;
    }
  }

  setEmptyClient(){
    this.options.client_data = {
      name:'',
      email:'',
      phone:'',
      address:'',
    }
  }

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      var type = files[0].file.type;
      if (type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") {
        this.setErrorFile();
      } else {
        this.options.client_data.client_file = files[0].file;
        this.options.client_data.client_image = files[0].imgURL;
        this.confirm_client_image = true;
      }
    }
  }

  fileSelectedChange(ev) {
    var type = ev.target.files[0].type;
    var size = ev.target.files[0].size;
    if ((type !== "image/png" && type !== "image/jpeg" && type !== "image/jpg") || (size > 200000)) {
      this.setErrorFile();
    } else {
      this.options.client_data.client_file = ev.target.files[0];
      const reader = new FileReader();
      reader.readAsDataURL(ev.target.files[0]);
      reader.onload = () => {
        this.options.client_data.client_image = reader.result as string;
        this.confirm_client_image = true;
      };
    }
  }

  setErrorFile() {
    var message = "Make sure that the file you are trying to upload is an image and the size in bellow 2MB"
    this.showSnackNotification(message);
  }
  
  showSnackNotification(message) {
    this.commonProvider.showSnackBar('Message',message,'alert-snackbar');
  }

  
  _closeDialog(){
    this.dialogRef.close('exit');
  }
  
  fillInfo(){
    this.auto_fill = true;
    if(this.data.case.client_name && this.data.case.client_address_1){
      this.options.client_data.name = this.data.case.client_name;
      this.options.client_data.email = this.data.case.client_email ? this.data.case.client_email : '';
      this.options.client_data.address = this.data.case.client_address_1;
    }else{
      this.setEmptyClient();
    }
  }

  continue(){
    this.dialogRef.close(this.options);
  }

  attemptToPrint(){
    let company = new Company();
    this.settingsProvider.getPdfSettings().subscribe((settings)=>{
      company = settings;
      this.authProvider.getUserInfo().subscribe((response)=>{
        let personal_data:User = response;
        var complete_data_company = ((company.name !== "") &&  (company.contact_email !== "") && (company.contact_name !== "") && (company.contact_phone !== "") && (company.logo !== ""));
        var complete_personal_data = ((personal_data.first_name !== "") &&  (personal_data.last_name !== "") && (personal_data.phone !== ""));
        if(!(complete_data_company || complete_personal_data)){
          this.dialogRef.close(false);
        }else{
          this.options.company_information = company;
          this.dialogRef.close(this.options)
        }
      },error=>{  
        this.handlerError.handleError(error)
      })
    })
  }

  async toDataUrl(url, callback) {
    var xhr = new XMLHttpRequest();
    xhr.onload = function() {
        var reader = new FileReader();
        reader.onloadend = function() {
            callback(reader.result);
        }
        reader.readAsDataURL(xhr.response);
    };
    xhr.open('GET', url);
    xhr.responseType = 'blob';
    xhr.send();
}

  async getSellerImage(url){
    return new Promise(async (resolve) => {
        await this.toDataUrl(url, function(myBase64) {
            resolve(myBase64);
        });
    });
  }

}
