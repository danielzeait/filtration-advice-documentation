import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { CommonProvider } from 'src/app/providers/common.service';

@Component({
  selector: 'app-dust-vs-dp',
  templateUrl: './dust-vs-dp.component.html',
  styleUrls: ['./dust-vs-dp.component.scss']
})
export class DustVsDpComponent implements OnInit {
  @Input('points') points:any;
  @Output() change_points =new EventEmitter<any>();
  dust_held:boolean=true;
  dust_fed:boolean=false;
  constructor(private commonProvider:CommonProvider) { }

  ngOnInit() {}

  _changePoint(pos){
    this.calculateDustTypes(pos); 
    this.commonProvider.pointHasChanged(this.points);
  }

  calculateDustTypes(pos){
    if(this.dust_held){
      if(this.points[pos].arrestance && this.points[pos].dust_held){
        this.points[pos].dust_fed = ((((100 - parseFloat(this.points[pos].arrestance)) * parseFloat(this.points[pos].dust_held)) / 100) + parseFloat(this.points[pos].dust_held)).toFixed(2);
      }
    }
    if(this.dust_fed){
      if(this.points[pos].dust_fed && this.points[pos].arrestance){
        this.points[pos].dust_held = ((this.points[pos].dust_fed * this.points[pos].arrestance) / 100).toFixed(2);
      }
    }
    this.points[pos].dust = this.points[pos].dust_fed;
  }

}
