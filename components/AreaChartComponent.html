import { Component, Input, OnInit } from '@angular/core';
import Chart from 'chart.js';
import { UtilsFunctions } from 'src/app/providers/utils.functions';

@Component({
  selector: 'app-area-chart',
  templateUrl: './area-chart.component.html',
  styleUrls: ['./area-chart.component.scss']
})
export class AreaChartComponent implements OnInit {
  @Input('id') chart_id:any;
  @Input('ylabel') y_label:any;
  @Input('point') point:any;
  @Input('color') color:any;
  @Input('data') chart_data:any;
  @Input('snapshot') snapshot:any;
  data:any = [];
  chart:any;
  snapshot_data:any = [];
  constructor(private utils:UtilsFunctions) { }

  ngOnInit(): void {
    this.data = [];

    for (let i = 0; i < this.chart_data.length; i++) {

      if(this.snapshot){
        var timestamp = new Date( this.chart_data[i].date).getTime() / 1000;
        if(timestamp == this.snapshot){
          this.snapshot_data.push(
            { 
              x:i,
              y:this.chart_id == 'energy' ? this.chart_data[i].energy_consumed : this.chart_data[i].operation_hours
            }
          )
        }
      }
      this.data.push(
        { 
          x:i,
          y:this.chart_id == 'energy' ? this.chart_data[i].energy_consumed : this.chart_data[i].operation_hours
        }
      )
    };
  }

  ngAfterViewInit() {
    this.initChart();
  }

  initChart(){
    var id = this.chart_id;
    var ctx:HTMLElement = <HTMLElement> document.getElementById(id);
    ctx.style.backgroundColor = '#FFF';
    ctx.style.padding = '10px';

    var canvas : any = document.getElementById(id);
    var ctx2 = canvas.getContext("2d");

    var datasets =[
      {
        data: this.data,
        backgroundColor:this.color,
        borderWidth:0,
        pointBackgroundColor: this.point,
        pointBorderColor: this.point,
        pointRadius:'1.5'
      }
    ]

    if(this.snapshot){
      datasets.push({
        data: this.snapshot_data,
        backgroundColor:"#25345B",
        borderWidth:1,
        pointBackgroundColor: "#fff",
        pointBorderColor: "#25345B",
        pointRadius:'8'
      })
    }

    this.chart = new Chart(ctx2, {
        type: 'line',
        data: {
            datasets: datasets
        },
        options: {
          elements: {
            line: {
              tension: 0.4
            }
          },
          legend:{
            display:false,
          },
          tooltips:{
            callbacks:{
                label:(tooltipItem, data) => {
                    var value = Number.parseFloat(tooltipItem.yLabel).toFixed(2);
                    var parts = value.toString().split(".");
                    var sufix = 'hours';
                    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                    var final_value = (parts.join("."));
                    if(this.chart_id == 'energy'){
                      sufix = 'wh'
                    }
                    var label = final_value + ' ' + sufix;
                    return label;
                },
                title: (tooltipItem) =>{
                  var value = Number.parseInt(tooltipItem[0].xLabel);
                  var date = this.utils.formatDateStringMonth(this.chart_data[value].date);
                  return date;
                }
            },
          },
          scales: {
              xAxes: [{
                  type: 'linear',
                  position: 'bottom',
                  scaleLabel: {
                    display: true,
                    labelString: 'Date',
                    fontColor:'#222',
                    fontSize:10,
                  },
                  max: this.chart_data.length,
                  ticks: {
                    callback: (value, index, values) => {
                      var pos = Number.parseInt(value);
                      var date = this.chart_data[pos] ? this.utils.formatDateChart(this.chart_data[pos].date) : ''
                      return date;
                    },
                    fontSize:10,
                  }
                },
              ],
              yAxes: [{
                scaleLabel: {
                  display: true,
                  labelString: this.y_label,
                  fontColor:'#222',
                  fontSize:10,
                },
                ticks: {
                  beginAtZero: true,
                  fontSize:10,
                },
              },
            ],
          }
        }
    });
  }


}
