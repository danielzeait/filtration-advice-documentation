import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { NewCardFormProvider } from 'src/app/forms/new-card.form';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { PaymentProvider } from 'src/app/providers/payment.service';
import { FormControl } from '@angular/forms';
import * as _moment from 'moment';
import { Moment } from 'moment';
import { MatDatepicker, MatSnackBar } from '@angular/material';
import {MomentDateAdapter, MAT_MOMENT_DATE_ADAPTER_OPTIONS} from '@angular/material-moment-adapter';
import {DateAdapter, MAT_DATE_FORMATS, MAT_DATE_LOCALE} from '@angular/material/core';
import { CommonProvider } from 'src/app/providers/common.service';
const moment =  _moment;

export const MY_FORMATS = {
  parse: {
    dateInput: 'MM/YYYY',
  },
  display: {
    dateInput: 'MM/YYYY',

@Component({
  selector: 'app-new-card',
  templateUrl: './new-card.component.html',
  styleUrls: ['./new-card.component.scss'],
  providers: [
    // `MomentDateAdapter` can be automatically provided by importing `MomentDateModule` in your
    // application's root module. We provide it at the component level here, due to limitations of
    // our example generation script.
    {
      provide: DateAdapter,
      useClass: MomentDateAdapter,
      deps: [MAT_DATE_LOCALE, MAT_MOMENT_DATE_ADAPTER_OPTIONS]
    },

    {provide: MAT_DATE_FORMATS, useValue: MY_FORMATS},
  ],
})
export class NewCardComponent implements OnInit {
  @Input('checkout') is_checkout:boolean;
  @Output() cancel = new EventEmitter<boolean>();
  @Output() created = new EventEmitter<boolean>();
  public card_brand: string;
  public date = new FormControl(moment());
  public minDate:any;
  public newCardForm:any;
  public card_error: boolean;
  public error_message: any;

  constructor(
    private commonProvider: CommonProvider,
    private formsProvider:NewCardFormProvider,
    private handlerError:CustomErrorHandler, 
    private paymentProvider:PaymentProvider
  ) { }

  ngOnInit() {
    this.newCardForm = this.formsProvider.newCardForm();
    this.newCardForm.controls['number'].valueChanges.subscribe((value)=>{
      if(value.length > 3){
        this.card_brand = this.getCreditCardImage(value);
        this.newCardForm.controls['brand'].setValue(this.card_brand);
      }else if(value.length < 4){
        this.card_brand = '';
      }
    })
    this.setMinDate();
  }

  setMinDate(){
    var date = new Date();
    var month = date.getMonth() + 1;
    var year = date.getFullYear();
    var day = date.getDate();
    this.minDate = new Date(year , month, day );
    this.initNewCardData();
  }

  initNewCardData(){
    this.newCardForm.controls['exp_month'].setValue(this.minDate.getMonth()+1);
    this.newCardForm.controls['exp_year'].setValue(this.minDate.getFullYear());
    var date = moment(this.minDate);
    this.date.setValue(date);
  }

  getCreditCardImage(value){
    if(value[0] == 4){
      return 'visa'
    }else if (value[0] == 5){
      return 'master'
    }else if (value[0] == 3){
      if(value[1] == 4 || value[1] == 7 ){
        return 'amex'
      }
      if(value[1] == 0 || value[1] == 6 || value[1] == 8 ){
        return 'diners'
      }
      if(value[1] == 5 ){
        return 'jcb'
      }
    }else if(value[0] == 6){
      if(value[1] == 2){
        return 'upay'
      }else{
        return 'discover'
      }
    }else if(value[0] == 2 && value[1] == 2 && value[2] == 2 && value[3] == 3){
      return 'master'
    }
  }

  chosenYearHandler(normalizedYear: Moment) {
    const ctrlValue = this.date.value;
    ctrlValue.year(normalizedYear.year());
    this.date.setValue(ctrlValue)
  }

  chosenMonthHandler(normalizedMonth: Moment, datepicker: MatDatepicker<Moment>) {
    const ctrlValue = this.date.value;
    ctrlValue.month(normalizedMonth.month());
    this.date.setValue(ctrlValue);
    this.newCardForm.controls['exp_month'].setValue(ctrlValue._d.getMonth() +1);
    this.newCardForm.controls['exp_year'].setValue(ctrlValue._d.getFullYear());
    datepicker.close();
  }

  _saveNewCard(){
    this.paymentProvider.createNewPaymentMethod(this.newCardForm.value).subscribe((response)=>{
      this.newCardForm = this.formsProvider.newCardForm();
      this.created.emit(true);
      this.commonProvider.showSnackBar('Payment created.','Your new payment method has been created correctly','success-snackbar');
    },error=>{
      this.card_error = true;
      this.error_message = error.error[0];
      this.handlerError.handleError(error)
    })
  }

  _cancel(){
    this.cancel.emit(true);
  }

}
