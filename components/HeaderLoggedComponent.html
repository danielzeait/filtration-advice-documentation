import { Component, OnInit, Input, SimpleChanges, HostListener } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { MatSnackBar } from '@angular/material/snack-bar';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { ResponsiveProvider } from 'src/app/providers/responsive-provider';
import { NotificationService } from 'src/app/providers/notifications.service';
import { CommonProvider } from 'src/app/providers/common.service';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { environment } from 'src/environments/environment';

@Component({
  selector: 'app-header-logged',
  templateUrl: './header-logged.component.html',
  styleUrls: ['./header-logged.component.scss']
})
export class HeaderLoggedComponent implements OnInit {
  @Input('active') link_active :any;
  public num_cols: number = 12;
  public user_confirmed:any;
  public user_data: any={};
  public show_notifications:boolean=false;
  public show_credits:boolean=false;
  public notifications:any;
  public front_version:any = environment.VERSION;
  public credits:any={

  };
  public expand_menu:boolean=true;
  swap_image: boolean=true;
  public transition_class1:string = 'fadeIn';
  public transition_class2:string = 'fadeIn';
  unread_notifications: any;
  
  constructor(
    private notificationProvider:NotificationService,
    private handlerError:CustomErrorHandler,
    private metadata:MetaDataProvider,
    private authProvider:AuthProvider, 
    private commonProvider:CommonProvider,
    private responsiveProvider:ResponsiveProvider,
    private navigator:NavigationProvider, 
    private _snackBar: MatSnackBar) { }
  
  @HostListener('window:resize', ['$event'])
    onResize(event) {
    this.num_cols = this.responsiveProvider.getNumCols(window.innerWidth);
  }
  
  ngOnInit() {
    
    this.user_confirmed = localStorage.getItem('confirmed_user');
    this.authProvider.getUserInfo().subscribe((response)=>{
      this.user_data = response;
      localStorage.setItem('user_data', JSON.stringify(response));
    });
    this.getNotifications();
    this.getCredits();
    this.commonProvider.updateNotificationList.subscribe((notifications)=>{
      this.notifications = [...notifications];
      this.unread_notifications = this.notifications.filter(notification => !notification.is_read);
    })
    this.commonProvider.updateUserCredits.subscribe(()=>{
      this.refreshCredits();
    });

    this.commonProvider.updateUserInfo.subscribe((data)=>{
     this.user_data = data;
    });

    this.metadata.getSystemVersion().subscribe((version)=>{
      this.commonProvider.setSystemVersion(version);
    })

  }

  _logOut(){
    this.authProvider.userLogut().subscribe((response)=>{
      this.authProvider.destroySession();
     },error=>{
      this.handlerError.handleError(error);
    })
  }

  getNotifications(){
    this.notificationProvider.getNotifications().subscribe((notifications:any)=>{
      this.notifications = [...notifications];
      this.unread_notifications = this.notifications.filter(notification => !notification.is_read);
    })
  }

  getCredits(){
    this.authProvider.getUserCredits().subscribe((credits:any)=>{
      if(this.credits.total_credits != credits.total_credits){
        this.credits = {...credits};
      }
    })
  }

  refreshCredits(){
    this.authProvider.getUserCredits().subscribe((credits:any)=>{
      if(this.credits.total_credits != credits.total_credits){
        this.credits = {...credits};
      }
    })
  }

  _confirmEmail(){
    this.authProvider.registerUserConfirmation('').subscribe((response)=>{
      if(response.detail == 'mail sent'){
        this.commonProvider.showSnackBar('Email sent.','A confirmation email was sent to confirm your account','success-snackbar');
      }
    },error=>{
      this.handlerError.handleError(error)
    })
  }

  ngOnChanges(changes: SimpleChanges){}

  _goHome(){
    this.navigator._goHome();
  }
  _goCases(){
    this.navigator._goHomeCases(null);
  }

  _goCompares(){
    this.navigator._goHomeComparisons();
  }

  _goReports(){
    this.navigator._goHome();
  }

  _goFilters(){
    this.navigator._goFilterList();
  }

  _openProfileSettings(){
    this.navigator._goProfile();
  }

  _goLoginSonicu(){
    this.navigator._goLoginSonicu();
  }

  _goInstrument(){
    this.navigator._goInstrumentList();
  }

  _toggleNotifications(){
    this.show_credits ? this.show_credits = false : this.show_credits;
    this.show_notifications = !this.show_notifications;
  }

  _toggleCredits(){
    this.show_notifications ? this.show_notifications = false : this.show_notifications;
    this.show_credits = !this.show_credits;
  }

  _expandMenu(){
    if(this.expand_menu){
      this.transition_class1 = "fadeIn";
      this.transition_class2 = "fadeOut"
    }else{
      this.transition_class1= "fadeOut";
      this.transition_class2 = "fadeIn"
    }
    this.expand_menu = !this.expand_menu;
    this.commonProvider.menuExpand(this.expand_menu);
    // this.swap_image = false;
    // setTimeout(() => {
    //   this.swap_image = true;
    // }, 500);
  }

}
