  entered($event: CdkDragEnter) {
    moveItemInArray(this.widgets_cards, $event.item.data, $event.container.data);
  }
  entered2($event: CdkDragEnter) {
    moveItemInArray(this.widgets_cards, $event.item.data, $event.container.data);
  }
  
  swap($event: CdkDragEnd<any>){
    this.setUserWidgets();
  }

  constructor(
    private dashboardProvider:DashboardProvider,
    private commonProvider:CommonProvider,
    private responsiveProvider:ResponsiveProvider,
    private navigator:NavigationProvider,
    private caseProvider:CaseProvider, 
    private cdr:ChangeDetectorRef, 
    private authProvider:AuthProvider) { }

  ngOnInit() {
    this.getRowspan();
  }

  ngAfterViewInit() {
    this.dropsQuery.changes.subscribe(() => {
      this.drops = this.dropsQuery.toArray()
    })
    Promise.resolve().then(() => {
      this.drops = this.dropsQuery.toArray();
    })
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  ngOnChanges(changes:SimpleChanges){
    // if(changes.widgets_cards.currentValue){
    //   let widgets = [...changes.widgets_cards.currentValue];
    //   widgets.forEach(element => {

    //         card.in_use = element.in_use;
    //       }
    //     });
    //   });
    // }
  }

  getRowspan(){
    this.merv_rowspan = this.responsiveProvider.getRowspan(window.innerWidth);
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.merv_rowspan = this.responsiveProvider.getRowspan(window.innerWidth);
  }


  setUserWidgets(){
    this.authProvider.setUserDashboard(this.widgets_cards).subscribe((dashboard)=>{
      //this.getWidgetCaseData();
    });
  }

  setChangeOutData(data){
    this.widgets_cards.find(widget => widget.type == "changeout").data = [...data];
  }

  removeChangeOutWidget(){
    this.removewWidget('changeout')
  }

  removewWidget(type){
    this.widgets_cards.find(widget => widget.type == type).in_use = false;
    this.setUserWidgets();
  }

  getWidgetCaseData(){
    this.widgets_cards.forEach(widget => {
      if(widget.type == "case" && widget.in_use && widget.data){
        this.caseProvider.getCaseById(widget.data).subscribe(
          ( case_info ) => {
          widget.case_data = {...case_info.results[0]}
        })
      }
    });
  }

  removeCaseFromDashboard(case_study){
    this.dashboardProvider.removeCaseFromDashboard(case_study);
    var pos = this.widgets_cards.findIndex(widget => widget.type == 'case' && widget.data == case_study.alter_autoincrement_id);
    this.widgets_cards.splice(pos,1);
  }

  goCases(){
    this.navigator._goHomeCases(null);
  }
  
  _editCase(study_case) {
    this.caseProvider.getCaseDetails(study_case.alter_autoincrement_id).subscribe(
      (case_details)=>{
        this.navigator._goNewCase(case_details);
      }
    )
  }

  _viewOptimization(case_data: System) {
    this.navigator._goOptimizationsResults(case_data);
  }

}
