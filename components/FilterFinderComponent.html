import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
import { startWith, map } from 'rxjs/operators';
import { FilterProvider } from 'src/app/providers/filter.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-filter-finder',
  templateUrl: './filter-finder.component.html',
  styleUrls: ['./filter-finder.component.scss']
})
export class FilterFinderComponent implements OnInit {
  public merv_rows:any;
  public model_filters: any;
  public filterControl = new FormControl();
  public modelFilterOptions: Observable<string[]>;
  public not_found:boolean=false;
  public filter_selected:any=null;

  constructor( private filterService:FilterProvider , private navigator:NavigationProvider) { }

  ngOnInit(){
    this.getFilters();
  }

  getFilters(){
    this.filterService.getLandingFilters().subscribe(
      (filters)=>{
        this.model_filters = filters;
        this.modelFilterOptions = this.filterControl.valueChanges
          .pipe(
          startWith(''),
          map(value => this._filterFilters(value,this.model_filters))
        );
      }
    )
  }

  private _filterFilters(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      if(option.length == 0 ){
        this.not_found = true;
      }else{
        this.not_found = false;
      }
      return option
    }
  }

  _setFilter(filter){
    if(filter){
      this.filter_selected = {...filter};
    }
  }

  _goRegister(){
    this.navigator._goRegister();
  }
  
  _goLogin(){
    this.navigator._goLogin();
  }

  _goContact(){
    this.navigator._goContact();
  }


}
