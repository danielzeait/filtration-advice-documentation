import { Component, OnInit } from '@angular/core';
import { MatDialogRef } from '@angular/material';

@Component({
  selector: 'app-filter-terms-conditions',
  templateUrl: './filter-terms-conditions.component.html',
  styleUrls: ['./filter-terms-conditions.component.scss']
})
export class FilterTermsConditionsComponent implements OnInit {

  constructor(
    public dialogRef: MatDialogRef<FilterTermsConditionsComponent>,
  ) { }

  ngOnInit(): void {
  }

  _closeDialog(){
    this.dialogRef.close(false);
  }

  _agree(){
    this.dialogRef.close(true)
  }

}
