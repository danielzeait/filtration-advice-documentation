  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.resize = this.responsiveProvider.getScreenSize(window.outerWidth);
  }
    
  ngOnInit() {
    this.loaderProvider.presentLoader('Wait a moment please','case_details');
    localStorage.removeItem('case_reports_id');
    //this.case_data.is_processing = true;
    
    this.routeActivated.params.subscribe((parameters) => {
      this.url_id = parameters.id;
      this.snapshot_date = parameters.date;
      this.snapshot_date ? this.is_snapshot = true : this.is_snapshot = false;
      this.caseProvider.getCaseDetails(this.url_id,this.snapshot_date ? this.snapshot_date : null).subscribe((response)=>{
        this.loaderProvider.leaveLoaders();
        this.case_data = response;
        // if(this.case_data.monitoring_module && this.case_data.monitoring_module.op_hours_so_far < 400){
        //   this.case_data.calibration = true;
        // };
        if(this.case_data.is_processing){
          this.case_data.display_status = this.case_data.status.status[this.case_data.status.status.length -1].status; 
          this._refreshCaseStatus(this.case_data.status.status[this.case_data.status.status.length -1].status);
        }
        this._calculateSavings();
        this.warning_error = response.monitoring_module ? response.monitoring_module.warnings.find(warning => warning.code == 'OPT-001') : null;
        if(!this.warning_error && !response.is_processing && !response.update_error){
          this.hide_charts = false;
          this.getCharts();
        }else{
          this.hide_charts = true;
        }
        this.circle_change = false;
        this.initSystemValues();
        this.initInstrumentPanel()
        this.resize = this.responsiveProvider.getScreenSize(window.outerWidth);
      })
      this.unitsProvider.getUnitsK().then(lbs => {
        this.lbs = lbs;
      });
      this.unitsProvider.getUnitsI().then(inc => {
