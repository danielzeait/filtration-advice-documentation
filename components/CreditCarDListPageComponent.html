export class CreditCarDListPageComponent implements OnInit {
  date = new FormControl(moment());
  cards_list:any=[];
  create_card:boolean = false;
  card_details:boolean = false;
  new_card:any={};
  toggleOverlayClass:any = [];
  card_error: boolean;
  error_message: any;
  minDate:any;
  newCardForm:any;
  card_brand: string;
  constructor(
    private commonProvider:CommonProvider,
    private handlerError:CustomErrorHandler, 
    private paymentProvider:PaymentProvider,
    private _snackBar: MatSnackBar,public dialog: MatDialog,) { }

  ngOnInit() {    
    this.getCreditCards();
  }

  getCreditCards(){
    this.paymentProvider.getPaymentMethods().subscribe((response)=>{
      this.cards_list = response;
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  newCard(){
    this.create_card = true;
  }

  newCardCreated(){
    this.create_card = false;
    this.getCreditCards();
  }

  _setDefaultCard(card){
    this.paymentProvider.setDefaultPaymentMethod(card).subscribe(()=>{
      this.getCreditCards();
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  _openModalConfirmation(card): void {
    let dialog_data = {
      title:'credit card',
      description:"You WON'T be able to recover this card in the future"
    };
    const dialogRef = this.dialog.open(ConfirmDialogComponent, {
      width: '50%',
      height: '30%',
      data: dialog_data
    });

    dialogRef.afterClosed().subscribe(result => {
      if(result){
        this.paymentProvider.deletePaymentMethod(card).subscribe(()=>{
          this.getCreditCards();
          this.commonProvider.showSnackBar('Card deleted.','Your credit card has been deleted correctly.' ,'success-snackbar');
          this._snackBar
        },error=>{
          this.handlerError.handleError(error);
        })
      }
    });
    
  }
}
