import { ChangeDetectorRef, Component, Inject, OnInit } from '@angular/core';
import { FileHandle } from 'src/app/directives/dragDrop.directive';
import { MatSnackBar } from '@angular/material/snack-bar';
import { ReportsProvider } from 'src/app/providers/reports.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { MatDialogRef } from '@angular/material';
import { CommonProvider } from 'src/app/providers/common.service';
import { FilterTermsConditionsComponent } from '../../filter-terms-and-conditions/filter-terms-conditions/filter-terms-conditions.component';
import { MatDialog, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { Observable } from 'rxjs';
import { startWith, map } from 'rxjs/operators';
import { FormControl } from '@angular/forms';
import { SuccessDialogComponent } from '../../success-dialog/success-dialog.component';

@Component({
  selector: 'app-upload-report',
  templateUrl: './upload-report.component.html',
  styleUrls: ['./upload-report.component.scss']
})
export class UploadReportComponent implements OnInit {
  public modelFilterOptions: Observable<string[]>;
  public filterControl = new FormControl();
  public filter_selected:any={};
  public model_filters: any;
  public original_filters: any;
  public file_to_upload: any;
  public confirm_view: boolean;
  public new_report: any;
  public fileChanged: any;
  public error_title: string;
  public snack_message: string;
  public error_file: boolean;
  public agree:boolean = false;
  public not_found:boolean = false;
  public request:any = {
    model_filter_name:'',
    manufacturer:'',
    is_public:true,
  }
  constructor(
    public dialog: MatDialog,
    public dialogRef: MatDialogRef<UploadReportComponent>,
    private _snackBar:MatSnackBar, 
    private commonProvider:CommonProvider,
    private navigator:NavigationProvider,
    private cdr:ChangeDetectorRef,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private reportProvider:ReportsProvider, 
    private handlerError:CustomErrorHandler) { }

  ngOnInit(): void {
    this.setFilters();
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }

  setFilters(){
    this.model_filters = this.data.filters;
    this.original_filters = [...this.data.filters];
    this.setFilterOptions();
  }
  
  setFilterOptions(){
    this.modelFilterOptions = this.filterControl.valueChanges
    .pipe(
      startWith(''),
      map(value => this._filterFilters(value,this.model_filters))
      );
  }

  private _filterFilters(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      if(option.length == 0){
        this.not_found = true;
        this.request.model_filter_name = value;
      }else{
        this.not_found = false;
      } 
      return option
    }
  }

  setRequestFilter(filter){
    this.filter_selected = {...filter};
    this.request.model_filter_name = filter.name;
  }
  

  _filesDropped(files: FileHandle[]): void {
    if (files[0]) {
      if (files[0].file.type != "application/pdf") {
        this.setErrorFile();
      } else {
        this.file_to_upload = files[0].file;
        this.confirm_view = true;
      }
    }
  }

  fileSelectedChange(ev) {
    if (ev.target.files[0].type != "application/pdf") {
      this.setErrorFile();
    } else {
      this.file_to_upload = ev.target.files[0];
      this.confirm_view = true;
    }
  }

  _viewTermsConditions(){
    const dialogRef = this.dialog.open(FilterTermsConditionsComponent, {
      width: '80%',
      height: '80%',
      id: 'terms-filter',
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.agree = true;
      }
    });
  }

  setErrorFile() {
    this.error_file = true;
    this.commonProvider.showSnackBar('Alert','File selected is not a PDF file','alert-snackbar');
  }
  
  _sendReport() {
    this.reportProvider.sendPDFReport(this.request,this.file_to_upload).subscribe((response) => {
      this.file_to_upload = '';
      this._requestSucceed();
    }, error => {
      this.handlerError.handleError(error);
    })
  }

  _requestSucceed(){
    let dialog_data = {
      image:"../../../../../assets/imgs/petición enviada.png",
      title:'Thank you! Your request has been sent.',
      message:'We will endeavor to procure this filter test report and email you with any updates on it.',
      dialog_id:'report_sent',
      action:"Ok. I'll be waiting"
    }
    const dialogRef = this.dialog.open(SuccessDialogComponent, {
      width: '30%',
      height: 'auto',
      data: dialog_data,
      id:'confirmed'
    });
    this.dialogRef.close();
  }

  _clearFile() {
    this.file_to_upload = '';
    this.confirm_view = false;
  }

  _closeDialog(bool?){
    this.dialogRef.close(bool ? bool : false);
  }

  contact(){
    this.dialogRef.close();
    this.navigator._contactUs();
  }


}
