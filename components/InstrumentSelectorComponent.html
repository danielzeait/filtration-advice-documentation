import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { InstrumentProvider } from 'src/app/providers/instrument.service';

@Component({
  selector: 'app-instrument-selector',
  templateUrl: './instrument-selector.component.html',
  styleUrls: ['./instrument-selector.component.scss']
})
export class InstrumentSelectorComponent implements OnInit {
  @Input('type') instrument_type:any;
  @Output() close = new EventEmitter();
  @Output() done = new EventEmitter();
  public instrument_cards:any;
  public instrument_providers:any;
  public instruments:any=[];
  public slots:any=[];
  public unlogged:any=false;
  public provider:any;
  provider_to_log: any;
  hobo_dashboard: boolean;
  constructor(private instrumentsProvider:InstrumentProvider) { }

  ngOnInit() {
    this.instrument_cards = this.instrumentsProvider.getInstrumentsCards();
    let provider = JSON.parse(localStorage.getItem('provider_selected'));
    if(provider){
      this._checkForInstruments(provider);
    }else{
      this.provider = null;
    }
    this.instrumentsProvider.getAllInstrumentProviders().subscribe((providers)=>{
      this.instrument_providers =  providers;
    })

    if(this.instrument_type == 'air_velocity'){
      this.instrument_cards.map(card=>card.highlight = false)
      this.instrument_cards[0].highlight = true;
      this.instrument_type = "Airflow";
    }
    if(this.instrument_type == 'pressure_drop'){
      this.instrument_cards.map(card=>card.highlight = false)
      this.instrument_cards[1].highlight = true;
      this.instrument_type = "Pressure Drop";
    }
    if(this.instrument_type == 'temperature'){
      this.instrument_cards.map(card=>card.highlight = false)
      this.instrument_cards[2].highlight = true;
      this.instrument_type = "Temperature";
    }
    if(this.instrument_type == 'humidity'){
      this.instrument_cards.map(card=>card.highlight = false)
      this.instrument_cards[3].highlight = true;
      this.instrument_type = "Humidity";
    }
  }

  _checkForInstruments(provider){
    if(!provider.is_already_logged){
      this.provider_to_log = provider;
    }else{
      this.provider = provider;
      localStorage.setItem('provider_selected',JSON.stringify(this.provider));
      if(provider.slug == 'sonicu'){
        this.getSonicuInstruments();
      }else{
        this.hobo_dashboard = true;
      }
    }

  }
  
  getSonicuInstruments(){
    this.instrumentsProvider.getSonicuInstruments().subscribe(
      (instrument_list) => { 
        this.instruments = instrument_list;
      },(error) => {
        error.status === 403 ? this.unlogged = true : '';
      }
    );
  }

  goBack(){
    let provider = localStorage.getItem('provider_selected');
    if(provider){
      this.close.emit();
    }else{
      if(this.instruments && this.instruments.length > 0 ){
        this.instruments = null;
        this.provider = null;
      }else{
        if(this.provider_to_log){
          this.provider_to_log = null;
        }else{
          if(this.provider){
            this.provider = null;
          }else{
            this.close.emit()
          }
        }
      }
    }
  }

  setInstrument(event){
    let instrument = event;
    instrument.provider = this.provider;
    this.done.emit(instrument);
  }

  

}
