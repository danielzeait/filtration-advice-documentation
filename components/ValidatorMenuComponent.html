import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { Router } from '@angular/router';
import { AuthProvider } from 'src/app/providers/auth.service';
import { Report } from 'src/app/models/report.model';
import { ReportsProvider } from 'src/app/providers/reports.service';
import { FormGroup, FormBuilder, FormControl } from '@angular/forms';
import {MatPaginator} from '@angular/material/paginator';
import {MatTableDataSource} from '@angular/material/table';
import {MatSort} from '@angular/material/sort';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { ConfirmDialogComponent } from 'src/app/components/modal-dialogs/confirm-dialog/confirm-dialog.component';
import { MatDialog } from '@angular/material/dialog';
import { CommonProvider } from 'src/app/providers/common.service';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { report } from 'process';

@Component({
  selector: 'app-validator-menu',
  templateUrl: './validator-menu.component.html',
  styleUrls: ['./validator-menu.component.scss']
})


export class ValidatorMenuComponent implements OnInit {
  @ViewChild(MatPaginator, {static: true}) paginator: MatPaginator;
  @ViewChild(MatSort, {static: true}) sort: MatSort;

  public searchedWordsControl = new FormControl('');

  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
  )
  
  public active:Array<any> = [
    {active:true,border_left:'#199AD8'},
    {active:false,border_left:'#d2a700'},
    {active:false,border_left:'#249e24'},
    {active:false,border_left:'#bb4102'},
  ];
  public displayedColumns: string[] = ['number','norm', 'filter','manufacturer','laboratory', 'date', 'user','status','actions'];
  public dataSource:any;
  public options: FormGroup;
  public reports:Array<Report>=[];
  public original_reports:Array<Report>=[];
  public all_reports:Array<Report>=[];
  public _searchTerm:string='';
  public pending_reports: Report[];
  public approved_reports: Report[];
  public rejected_reports: Report[];
  
  constructor(
    private elementRef: ElementRef,
    public fb: FormBuilder,
    private userProvider:AuthProvider,
    private reportProvider:ReportsProvider,
    private commonProvider:CommonProvider,
    public dialog: MatDialog,
    private router:Router,
    private navigator:NavigationProvider,
    private authProvider:AuthProvider,
  ) { 
    this.options = fb.group({
      bottom: 0,
      fixed: false,
      top: 0
    });
  }


  ngOnInit() {
    this.getValidatorReports();
  }

  setActiveBtn(pos) {
    this.active.map(active=>active.active = false);
    this.active[pos].active = true;
    this.elementRef.nativeElement.style.setProperty('--color', this.active[pos].border_left);
    if(pos == 0){
      this.setDataSource(this.original_reports);
    }
    if(pos == 1){
      this.setDataSource(this.pending_reports);
    }
    if(pos == 2){
      this.setDataSource(this.approved_reports);
    }
    if(pos == 3){
      this.setDataSource(this.rejected_reports);
    }
  }

  getValidatorReports(){
    this.reportProvider.getUserReports().subscribe((response)=>{
      this.reports = response.reverse();
      this.original_reports = response.reverse();
      // this.reports.map(report => !report.user ? report.user = {email:'Public',id:''} : report.user);
      // this.original_reports.map(report => !report.user ? report.user = {email:'Public',id:''} : report.user);
      this.pending_reports = this.reports.filter(reports => reports.status == 'pending');
      this.approved_reports = this.reports.filter(reports => reports.status == 'approved');
      this.rejected_reports = this.reports.filter(reports => reports.status == 'rejected');
      this.elementRef.nativeElement.style.setProperty('--color', this.active[0].border_left);
      this.setDataSource(this.reports);
    },error=>{
      console.log(error);
    })
  }

  setDataSource(reports){
    this.dataSource =  new MatTableDataSource<any>(reports);
    this.dataSource.paginator = this.paginator;
    this.dataSource.sort = this.sort;
  }

  filterReportsByStatus(value){
  }

  get searchTerm():string{
    return this._searchTerm
  }

  set searchTerm(value:string){
    this._searchTerm = value
    this.reports = this.searchReportList(value);
  }

  searchReportList(value:string){
    return this.original_reports.filter(report=>
      (report.id.toString().toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1)  ||
      (report.created.toString().toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
      (report.status.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1)
      //(report.laboratory.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1)  
    )
  }

  searchBywords(){
    this.reports = [...this.original_reports.filter(report=>
      (report.id.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1)  ||
      (report.laboratory ? report.laboratory.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1 : false)  ||
      (report.user ? report.user.email.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1 : false)  ||
      (report.manufacturer.name ? report.manufacturer.name.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1 : false)  ||
      (report.created.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1) ||
      (report.status.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1) ||
      (report.number.toString().toLowerCase().indexOf(this.searchTerm.toLocaleLowerCase()) !== -1))]
    this.setDataSource(this.reports);
  }

  clearSearch(){
    this.reports = this.original_reports;
    this._searchTerm = '';
  }

  refresh(){
    this.getValidatorReports();
  }

  logOut(){
    this.authProvider.userLogut().subscribe((response)=>{
      this.authProvider.destroySession();
    },error =>{
      console.log(error);
    })
  }

  _attemptDelete(report){
    let dialog_data = {
      title:'lab test report',
      description:"You WON'T be able to recover this report in the future"
    };
    const dialogRef = this.dialog.open(ConfirmDialogComponent, {
      width: '50%',
      height: '30%',
      data: dialog_data
    });

    dialogRef.afterClosed().subscribe(result => {
      if(result){
        this.reportProvider.deleteReport(report).subscribe(()=>{
          this.commonProvider.showSnackBar('Report deleted','The report has been deleted correctly','success-snackbar');
          this.refresh();
        })
      }
    });
  }

  _reviewReport(report){
    this.navigator._goReportReview(report);
  }

  _newReport(){
    this.navigator._goReportReview(null);
  }
}
