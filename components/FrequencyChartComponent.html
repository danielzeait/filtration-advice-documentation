import { Component, Input, OnInit } from '@angular/core';
import Chart from 'chart.js';

@Component({
  selector: 'app-frequency-chart',
  templateUrl: './frequency-chart.component.html',
  styleUrls: ['./frequency-chart.component.scss']
})
export class FrequencyChartComponent implements OnInit {
  @Input('chart_data') chart_data : any;  
  @Input('chart_id') chart_id : any;  
  @Input('label') label : any;  
  @Input('color') color : any;  

  private chart: any;
  private chartData:any=[];
  public verticalData:any;

  constructor() { }

  ngOnInit() {}
  
  setCurrentChartData(){
    var data = [...this.chart_data];
    var y_max:number;
    data.sort(function(a, b) {  return b.label - a.label });

    if(this.chart_id == 'airflow_frequency'){
      y_max = Math.abs((data[0].label % 100) - 100) + data[0].label 
    }else{
      y_max = Math.abs((data[0].label % 0.1) - 0.1) + data[0].label 
    }
    for (let i = 0; i < this.chart_data.length; i++) {
      var point = {
        'x':parseFloat(this.chart_data[i].frequency).toFixed(3),
        'y':this.chart_id == "airflow_frequency" ? parseFloat(this.chart_data[i].label).toFixed(2) : parseFloat(this.chart_data[i].label).toFixed(3),
      }
      
      if(this.chart_data[i].is_cut){
        var x = this.chart_data[i].frequency;
      }
      
      this.chartData.push(point);
      this.verticalData = [
        {x:x,y:0},
        {x:x,y:y_max.toFixed(3)}
      ]
      if(i == this.chart_data.length - 1){
        setTimeout(() => {
          this.initChart();
        }, 500);
      }
    }
  }
  
  ngAfterViewInit(){
    this.setCurrentChartData();
  }

  initChart(){
    var id = this.chart_id;
    var ctx:HTMLElement = <HTMLElement> document.getElementById(id);
    var base_label = this.label;
    ctx.style.backgroundColor = '#FFF';
    ctx.style.padding = '10px';

    var canvas : any = document.getElementById(id);
    var ctx2 = canvas.getContext("2d");
    
    Chart.Legend.prototype.afterFit = function() {
      this.height = this.height + 10;
    };
    this.chart = new Chart(ctx2, {
      type: 'line',
      data: {
        datasets:[
          {
            label:this.label,
            data:this.chartData,
            pointRadius:'3',
            pointBackgroundColor:'#fff',
            pointBorderWidth:'1',
            borderColor:this.color,
            borderWidth:2,
            fill:false,
            backgroundColor:'#f3f3f3'
          },
          {
            label:'0',
            data:this.verticalData,
            pointRadius:'0',
            borderColor:this.chart_id == 'dp_frequency' ? '#199AD8' : '#ffa629',
            borderWidth:3,
            fill:false,
            backgroundColor:'#f3f3f3'
          }
        ]
      },
      options: {
        legend: {
          labels: {
            filter: function(item, chart) {
              // Logic to remove a particular legend item goes here
              return !item.text.includes('0');
            },
            fontFamily:'lato',
            fontSize:11.5
          },
        },
        elements: {
          line: {
            tension: 0.1
          }
        },
        tooltips:{
          callbacks:{
              label:function(tooltipItem, data) {
                  let that = this;
                  var value:any;
                  if(id == 'airflow_frequency'){
                    value = Number.parseFloat(tooltipItem.yLabel).toFixed(2);
                  }else{
                    value = Number.parseFloat(tooltipItem.yLabel).toFixed(3);
                  }
                  var parts = value.toString().split(".");
                  parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                  var final_value = (parts.join("."))
                  var label = base_label + ' ' + final_value;
                  return label;
              },
              title: function(tooltipItem){
                var value = Number.parseFloat(tooltipItem[0].xLabel).toFixed(2);
                return value + ' %';
              }
          },
        },
        scales: {
            xAxes: [{
                type: 'linear',
                position: 'bottom',
                scaleLabel: {
                  display: true,
                  fontColor:'#222',
                  fontSize:10,
                },
                ticks: {
                  //stepSize: 2,
                  fontSize:10,
                  callback: function(value, index, values) {
                    var parts = value.toString().split(".");
                    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                    var label = (parts.join("."))
                    return label + ' %';
                  }    
                }
              },
            ],
            yAxes: [{
              scaleLabel: {
                labelString:this.label,
                display: true,
                fontColor:'#222',
                fontSize:10,
              },
              ticks: {
                beginAtZero: true,
                fontSize:10,
                callback: function(value, index, values) {
                  var parts = Number.parseFloat(value).toFixed(3).toString().split(".");
                  var final_value:any;
                  if(id == "dp_frequency"){
                    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                    final_value = (parts.join("."))
                  }else{
                    final_value = parts[0];
                  }
                  return final_value;;
                },

              },
            },
          ],
        }
      }
    });
  }

}
