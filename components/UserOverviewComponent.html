import { Component, OnInit, Input, EventEmitter, Output } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-user-overview',
  templateUrl: './user-overview.component.html',
  styleUrls: ['./user-overview.component.scss']
})
export class UserOverviewComponent implements OnInit {
  @Input('data') overview_data:any;
  @Output() remove = new EventEmitter(); 
  public credits:any;
  public user_data:any;
  constructor(private navigator:NavigationProvider,private authProvider:AuthProvider) { }

  ngOnInit() {
    this.user_data = JSON.parse(localStorage.getItem('user_data'));
    this.getCredits();
  }

  removeFromDashboard(){
    this.remove.emit();
  }

  getCredits(){
    this.authProvider.getUserCredits().subscribe((credits:any)=>{
      this.credits = {...credits};
    })
  }

  goProfile(){
    this.navigator._goProfile();
  }

  goCases(){
    this.navigator._goHomeCases(null);
  }

  goFilters(){
    this.navigator._goFilterList();
  }

  goComparisons(){
    this.navigator._goHomeComparisons();
  }

  goInstruments(){
    this.navigator._goInstrumentList();
  }

}
