import { Component, OnInit, Output, EventEmitter, Input, SimpleChanges } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { SettingsProvider } from 'src/app/providers/settings.service';
import { CaseProvider } from 'src/app/providers/case.service';
import { ThrowStmt } from '@angular/compiler';
import { MatDialog } from '@angular/material/dialog';
import { ChangeCaseDialogComponent } from '../modal-dialogs/change-case-dialog/change-case-dialog/change-case-dialog.component';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-contact-component',
  templateUrl: './contact-component.component.html',
  styleUrls: ['./contact-component.component.scss']
})
export class ContactComponentComponent implements OnInit {
  @Output() set_data = new EventEmitter();
  @Output() done = new EventEmitter();
  @Output() leave_case = new EventEmitter();
  @Input('review') review:boolean;
  @Input('profile') profile:any;
  @Input('data') input_data:any;
  @Input('case_name') case_name:any;
  public industries:any;
  public zip_code_not_found:boolean = false;
  public user_data:any;
  public case_number:any;
  public company_zip_code:any;
  public user_zip_code:any;
  public personal_data:any={
    
  };
  constructor(
    public navigator:NavigationProvider,
    public dialog: MatDialog,
    private userProvider:AuthProvider, 
    private companyProvider:SettingsProvider, 
    private caseProvider:CaseProvider) { }

  ngOnInit() {
    this.case_number = parseInt(localStorage.getItem('next_case_number')) + 1;
    this.industries = [
      {id:1,name:'Industrial'},
      {id:2,name:'Commercial'},
      {id:3,name:'Residential'},
    ]
    this.getUserData();
    if(this.input_data){
      this.personal_data = {...this.input_data};
      this.personal_data.industry_type = this.industries.find(i => i.id == this.personal_data.industry_type.id);
      if(this.input_data.name) this.personal_data.case_name = this.input_data.name;
    }

  }
  
  ngOnChanges(changes: SimpleChanges){
    if(changes.input_data && !changes.input_data.firstChange){
      this.personal_data = {...changes.input_data.currentValue};
      this.personal_data.industry_type = this.industries.find(i => i.id == this.personal_data.industry_type.id);
    }
    if(changes.profile && !changes.profile.firstChange){
      this.profile = {...changes.profile.currentValue};
      if(this.profile.type == 'personal'){
        this.personal_data.company_name = null;
        this.personal_data.company_address = null;
        this.personal_data.client_name = null;
        this.personal_data.client_email = null;
        this.personal_data.client_address_1 = null;
        this.personal_data.client_address_2 = null;
        this.personal_data.client_zip_code = null;
      }
    } 
  }

  setData(){
    this.set_data.emit(this.personal_data)
  }

  continue(){
    scroll(0,0);
    this.done.emit();
  }

  getUserData(){
    this.userProvider.getUserInfo().subscribe((personal_data)=>{
      this.user_zip_code = personal_data.zip_code;
      this.companyProvider.getPdfSettings().subscribe((company_data)=>{
        this.company_zip_code = company_data.zip_code;
        this.user_data = { ...personal_data, ...company_data }
      })
    })
  }

  _calculateEnergyCosts() {
    this.setData();
  }  

  setPersonalInfo(){
    this.personal_data.contact_name = this.user_data.first_name + ' ' + this.user_data.last_name;
    this.personal_data.phone = this.user_data.phone;
    this.personal_data.email = this.user_data.email;
    this.personal_data.job_information = this.user_data.job_information ? this.user_data.job_information : '';
    this.personal_data.personal_zip_code = this.user_zip_code ? this.user_zip_code : '';
  }

  setCompanyInfo(){
    this.personal_data.company_name = this.user_data.name;
    this.personal_data.company_address = this.user_data.address;
    this.personal_data.company_zip_code = this.company_zip_code ? this.company_zip_code : '';
  }

  _attemptChangeCase(){
    const dialogRef = this.dialog.open(ChangeCaseDialogComponent, {
      width: '50%',
      height: 'auto',
    });

    dialogRef.afterClosed().subscribe((result)=>{
      if(result){
        this.leave_case.emit(true)
      }
    })
    
  }

}
