import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';

@Component({
  selector: 'app-instuments-list',
  templateUrl: './instuments-list.component.html',
  styleUrls: ['./instuments-list.component.scss']
})
export class InstumentsListComponent implements OnInit {
  @Input('instruments') instruments:any;
  @Input('type') type:any;
  @Input('for_case') for_case:boolean;
  @Output() done = new EventEmitter();
  public searchedWordsControl = new FormControl('');
  public show_me:Array<boolean> = []; 
  public original_instruments:any;
  public search:any;
  public displayedColumns: string[] = ['serial_number', 'name', 'unit', 'status', 'type','actions'];
  
  public searchedWords$: Observable<string[]> = this
    .searchedWordsControl
    .valueChanges
    .pipe(
      map((search: string) => search.trim().split(' '))
  )
  
  constructor() { }

  ngOnInit() {
    this.setInstruments()
  }

  setInstruments = () => {
    this.original_instruments = this.instruments;
    this.instruments.sort((a,b) => 
      (a['units'] > b['units']) ? 1 : ((b['units'] > a['units']) ? -1 : 0)
    )
    this.instruments.reverse();
    this.instruments.map(i => i.paired_with ? i.case_name = i.paired_with.case_name : i.case_name = 'Unpaired' );
    this.instruments.map(i => i.slot ? '' : i.slot = '-');
  }

  selectInstrument(ev){
    ev.selected=true;
    this.done.emit(ev);
  }

}
