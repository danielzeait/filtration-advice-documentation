import { Component, OnInit, Input, OnChanges, SimpleChanges, Output, EventEmitter } from '@angular/core';
import { Report } from 'src/app/models/report.model';
import { SharedReportsComponent } from 'src/app/components/modal-dialogs/shared-reports/shared-reports.component';
import { UserSharedReportsComponent } from 'src/app/components/modal-dialogs/users-shared-reports/users-shared-reports.component';
import { PdfViewerComponent } from 'src/app/components/modal-dialogs/pdf-viewer/pdf-viewer.component';
import { MatDialog } from '@angular/material/dialog';
import { MetaDataProvider } from 'src/app/providers/metadata.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';
import { PaymentProvider } from 'src/app/providers/payment.service';

const ELEMENT_DATA: Report[] = [];

@Component({
  selector: 'app-filter-test-reports',
  templateUrl: './filter-test-reports.component.html',
  styleUrls: ['./filter-test-reports.component.scss']
})
export class FilterTestReportsComponent implements OnInit, OnChanges {
  @Input('reports') reports: any;
  @Input('std') std: any;
  @Input('filter') filter: any;
  @Input('norm') norm: any;
  @Input('to_case') for_case: any;
  @Input('to_compare') to_compare: any;
  @Input('can_check') can_check: any;
  @Input('review') review:boolean;
  @Output() onReportChange = new EventEmitter<Number[]>();
  @Output() onReportsPrivate = new EventEmitter<Number[]>();
  @Output() onNormsChange = new EventEmitter<Number[]>();
  displayedColumns: string[] = ['check', 'number', 'created', 'laboratory', 'efficiency', 'owner', 'sharedw', 'actions'];
  dataSource = ELEMENT_DATA;
  checkedReports: Array<any> = [];
  public case_edition:any;
  public standardSelect;
  public norms;
  public owner;
  public otra;
  public stdDesdeService;
  public ownDesdeService;
  //public standard = this.standardSelect;
  public showFilterReports: any;
  public showFilterReportsPrivate;
  public view;
  public selected;
  public check;
  purchase_plan: any;
  constructor(
    public dialog: MatDialog, 
    public metaDataProvider: MetaDataProvider, 
    private handlerError:CustomErrorHandler,
    private navigator:NavigationProvider,
    private paymentProvider:PaymentProvider,
  ) { }



  ngOnInit() {
    this.dataSource = this.reports;
    this.case_edition = JSON.parse(localStorage.getItem('case_in_edition'));
    if(this.case_edition){
      let reports = JSON.parse(localStorage.getItem('reports_for_case'));
      var reports_ids = [];
      reports.forEach(report => {
        reports_ids.push(report.id);
      });
      this.initCheckedReports(reports_ids);
    }
    let ids = JSON.parse(localStorage.getItem('case_reports_id'));
    if(ids){
      this.initCheckedReports(ids);
    }
    if(!this.dataSource[0].user){
      this.dataSource[0].user= 'PUBLIC';
    } else {
      this.dataSource[0].user= this.dataSource[0].user;
    }
    this.showFilterReportsPrivate = 1;
    if(this.to_compare){
      if(this.can_check){
        this.displayedColumns = ['check', 'number', 'created' , 'laboratory','efficiency']
      }else{
        this.displayedColumns = ['number', 'created' , 'laboratory', 'efficiency']  
      }
    }else{
      // if(this.dataSource[0].is_public){
      // }
      this.displayedColumns = ['number', 'created' , 'laboratory', 'efficiency','actions']
    }
  }

  ngOnChanges(changes: SimpleChanges) {

    this.dataSource = this.reports;
    this.stdDesdeService = this.metaDataProvider.getStd();
    this.ownDesdeService = this.metaDataProvider.getOwn();
    
  }

  findReportId(id){
    let report_ids = JSON.parse(localStorage.getItem('case_reports_id'));
    if(report_ids){
      let id_found = report_ids.find(report_id => report_id == id);
      return id_found;
    }
  }

  initCheckedReports(ids){
    ids.forEach(report_id => {
      this.checkedReports.push(this.reports.find(report => report.id == report_id));
      this.onReportChange.emit(this.checkedReports);
    });
  }

  reportCheckChange(e, report) {
    if (e.checked) {
      this.norm = this.standardSelect;
      this.checkedReports.push(report);
    } else if(!(e.checked)){
      const index = this.checkedReports.findIndex(r => report.id == r.id);
      this.checkedReports.splice(index, 1);
    } else {
      this.checkedReports=[];
    }
    let reports_id = [];
    this.checkedReports.map(report=> reports_id.push(report.id));
    localStorage.setItem('case_reports_id',JSON.stringify(reports_id));
    this.onReportChange.emit(this.checkedReports);

  }
 
  setNorm(n) {
    this.std = n;
  }


  selectOwner(o) {
    this.owner = o;
    if (this.owner == 2) {
      this.onReportsPrivate.emit(this.owner);
    }
  }

  _openPdfReport(row) {
     this.dialog.open(PdfViewerComponent, {
      width: '65%',
      height: '98%',
      data: row.pdf_file_limited
    });
  }

  _openSharedModalDialog(reports): void {

    const dialogRef = this.dialog.open(SharedReportsComponent, {
      width: '50%',
      height: 'auto',
      data: reports
    });
  }

  _openUserSharedReport(reports): void {

    const dialogRef = this.dialog.open(UserSharedReportsComponent, {
      width: '50%',
      height: 'auto',
      id:'shared_report_dialog',
      data: reports
    });
  }

  _buyReport(element){
    let report = element;
    let report_filter = this.filter;
    this.paymentProvider.getAllProducts().subscribe((response)=>{
      localStorage.setItem('report_filter',JSON.stringify(report_filter));
      localStorage.setItem('report_pending',JSON.stringify(report));
      this.purchase_plan = response.find(plan => plan.slug == "report"),
      this.purchase_plan.action = 'report';
      this.navigator._goCheckOut(this.purchase_plan);
    },error=>{
      this.handlerError.handleError(error);
    })
  }

  _downloadReport(report){
    const url = report.pdf_file_limited;
    window.open(url, '_blank');
  }

}
