import { ChangeDetectorRef, Component, Inject, OnInit } from '@angular/core';
import { MatDialogRef } from '@angular/material';
import { FilterTermsConditionsComponent } from '../../filter-terms-and-conditions/filter-terms-conditions/filter-terms-conditions.component';
import { MatDialog, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { SuccessDialogComponent } from '../../success-dialog/success-dialog.component';
import { FilterProvider } from 'src/app/providers/filter.service';
import { Observable } from 'rxjs';
import { startWith, map } from 'rxjs/operators';
import { FormControl } from '@angular/forms';

@Component({
  selector: 'app-request-filter',
  templateUrl: './request-filter.component.html',
  styleUrls: ['./request-filter.component.scss']
})
export class RequestFilterComponent implements OnInit {
  public modelFilterOptions: Observable<string[]>;
  public filterControl = new FormControl();
  public filter_selected:any={};
  public model_filters: any;
  public original_filters: any;
  public agree:boolean=false;
  public request:any = {
    filter_name:'',
    filter_manufacturer_name:'',
    measurements:'',
    shop_link:'',
  }

  constructor(
    private filterService:FilterProvider,
    public filterProvider:FilterProvider,
    public dialog:MatDialog,
    private cdr:ChangeDetectorRef,
    @Inject(MAT_DIALOG_DATA) public data: any,
    public dialogRef: MatDialogRef<RequestFilterComponent>,
  ) { }

  ngOnInit(): void {
    this.setFilters();
  }

  ngAfterViewChecked() {
    this.cdr.detectChanges();
  }


  _closeDialog(){
    this.dialogRef.close();
  }

  setFilters(){
    this.model_filters = this.data.filters;
    this.original_filters = [...this.data.filters];
    this.setFilterOptions();
  }
  
  _sendRequest(){
    this.filterProvider.requestFilter(this.request).subscribe(()=>{
      this._requestSucceed();
    })
  }
  
  setFilterOptions(){
    this.modelFilterOptions = this.filterControl.valueChanges
    .pipe(
      startWith(''),
      map(value => this._filterFilters(value,this.model_filters))
      );
  }

  private _filterFilters(value: any,array:any): string[] {
    if (typeof value === 'string') {
      const filterValue = value ? value.toLowerCase() : '';
      let option = array.filter(option => option ? option.name.toLowerCase().includes(filterValue) : '');
      return option
    }
  }

  setRequestFilter(filter){
    this.filter_selected = {...filter};
    this.request.filter_name = filter.name;
  }
  
  _requestSucceed(){
    let dialog_data = {
      image:"../../../../../assets/imgs/petición enviada.png",
      title:'Thank you! Your request has been sent.',
      message:'We will endeavor to procure this filter test report and email you with any updates on it.',
      action:"Ok. I'll be waiting"
    }
    const dialogRef = this.dialog.open(SuccessDialogComponent, {
      width: '30%',
      height: 'auto',
      data: dialog_data,
      id:'confirmed'
    });
    this.dialogRef.close();
  }

  _viewTermsConditions(){
    const dialogRef = this.dialog.open(FilterTermsConditionsComponent, {
      width: '80%',
      height: '80%',
      id: 'terms-filter',
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.agree = true;
      }
    });
  }

}
