import { Component, OnInit } from '@angular/core';
import { FormGroup, FormControl, Validators, FormBuilder } from '@angular/forms';
import { AuthProvider } from 'src/app/providers/auth.service';
import { NavigationProvider } from 'src/app/providers/navigation.provider';

@Component({
  selector: 'app-contact',
  templateUrl: './contact.component.html',
  styleUrls: ['./contact.component.scss']
})
export class ContactComponent implements OnInit {
  public contactForm:FormGroup;
  public message_sent:boolean=false;
  public sending:boolean=false;
  public button_class:string='';
  constructor(private fb: FormBuilder, private authService:AuthProvider, private navigator:NavigationProvider) { }

  ngOnInit(){
    this.setContactForm();
  }

  setContactForm(){

    this.contactForm = this.fb.group({
      name: new FormControl({value:'',disabled:this.sending}),
      email: new FormControl({value:'',disabled:this.sending}, Validators.compose([
        Validators.required,
        Validators.email
      ])),
      message:new FormControl({value:'',disabled:this.sending},Validators.required),
      validation:new FormControl({value:null,disabled:this.sending}, Validators.required),
    },
    {
      validator:this.checkValidation
    });
  }

  checkValidation(group: FormGroup) { // here we have the 'passwords' group
    let validation = group.controls.validation.value;
    return validation ? null : { notValidation: true }
  }

  sendMessage(){
    this.contactForm.disable()
    this.sending = true;
    this.button_class = 'animated fadeIn infinite';
    this.authService.sendUserMessage(this.contactForm.value).subscribe(()=>{
      this.sending = false;
      this.button_class = '';
      this.message_sent = true;
    })
  }

  back(){
    this.navigator._land();
  }

}
