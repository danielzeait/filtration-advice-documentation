import { Component, OnInit, Input, SimpleChanges, Output, EventEmitter, ViewChild } from '@angular/core';
import { UtilsFunctions } from 'src/app/providers/utils.functions';
import { report } from 'process';
import { CaseProvider } from 'src/app/providers/case.service';

@Component({
  selector: 'app-system-information',
  templateUrl: './system-information.component.html',
  styleUrls: ['./system-information.component.scss']
})
export class SystemInformationComponent implements OnInit {
  @ViewChild("date_time_picker", {static:false}) date_time_picker: any;
  
  @Input('filters') num_filters:number;
  @Input('filter_case') filter_case:any;
  @Input('review') review:boolean;
  @Input('data') input_data:any;
  @Input('reports') reports:any;
  @Input('instrumentation') instrumentation:any;

  @Output() set_data = new EventEmitter();
  @Output() input_has_error = new EventEmitter();
  @Output() next_step = new EventEmitter();
  public list_height:string="520px";
  public exist_case:string;
  public days: any = Array.from(Array(30), (x, index) => index + 1);
  public hours: any = Array.from(Array(24), (x, index) => index + 1);
  public months: any = Array.from(Array(12), (x, index) => index + 1);
  public maxDate:Date = new Date();
  public original_airflow_values:any = {};
  public airflow_max_limit:any;
  public dateTime: Date;
  public airflow_min_limit:any;
  public system_information:any={
    drive_efficiency:100,
    motor_efficiency:90,
    fan_efficiency:67,
    installation_date:null,
    months_per_year:12,
    days_per_month:30,
    hours_per_day:24,
    hours_per_year:null,
    replacement_cycle:null,
    airflow_per_filter:null,
    total_airflow:null,
    average_pressure_drop:null,
    variable_airflow:1,
    maximum_operational_dp:null,
  }
  public max_airflow:any;
  public variable_airflows = [
    {'flow':null,'percent':null},
  ]
  public total_airflow_error: boolean;
  public negative_value_error: boolean;
  public airflow_value_error: boolean;
  dp_max: number;
  dp_min: number;

  constructor(private caseProvider:CaseProvider, private utils:UtilsFunctions) { }

  ngOnInit() {
    if(this.input_data){
      this.system_information = {...this.input_data};
      this.system_information.avg_hour_per_day = Math.ceil(this.system_information.hours_per_year / 365);
      if(this.system_information.airflows_per_filter){
        this.variable_airflows = [...this.system_information.airflows_per_filter];
      }
      //this.getEditionDpLimits();
    }else{
      this._updateHoursOfUse();
      this._updateAirflowPerFilter();
      if(this.filter_case){
        this.airflow_min_limit = 118 * this.num_filters;
        this._setAirflowLimits();
      }else{
        this.airflow_max_limit = 1968 * this.num_filters;
      }
    }
    this.getDpLimits();

  }
  
  ngOnChanges(changes: SimpleChanges){
    if(changes.input_data && !changes.input_data.firstChange){
      this.system_information = {...changes.input_data.currentValue};
      this.getDpLimits();
      if(this.system_information.airflows_per_filter){
        this.variable_airflows = [...this.system_information.airflows_per_filter];
        this._updateAirflowPerFilter();
        this._setAirflowLimits();
      }
    }
    if(changes.num_filters){
      this.num_filters = changes.num_filters.currentValue;
      this.getDpLimits();
      this._updateAirflowPerFilter();
      this._setAirflowLimits();
    };
    if(changes.filter_case){
      this._setAirflowLimits();
      this.getDpLimits();
    };
    if(changes.instrumentation && !changes.instrumentation.firstChange){
      this.instrumentation = {...changes.instrumentation.currentValue};
    }
  }
  /* Once the months days or hours change, new system operation hours 
  must be calculated */
  _updateHoursOfUse() {
    this.system_information.hours_per_year = 
    this.system_information.months_per_year * this.system_information.days_per_month * this.system_information.hours_per_day;
    this.system_information.avg_hour_per_day = Math.ceil(this.system_information.hours_per_year / 365);
  }

  // Every time a number change fix input with decimals
  setAmountDecimal(ev) {
    this.utils.transformAmount(ev);
  }

  _updateAirflowPerFilter(){
    this.system_information.airflow_per_filter = (this.system_information.total_airflow / this.num_filters);
  }

  _updateTotalAirflow(){
    this.system_information.total_airflow = (this.system_information.airflow_per_filter * this.num_filters);
  }

  _setAirflowType(){
    if(this.system_information.variable_airflow == '2'){
      this.variable_airflows = [
        {'flow':null,'percent':100},
      ]
    }
    this.original_airflow_values = { ...this.system_information }
    this.system_information.airflow_per_filter = null;
    this.system_information.total_airflow = null;
    
  }

  _setAirflowLimits(){
    var airflow_max_limit;
    if(this.filter_case && this.filter_case.case_size_height && this.filter_case.case_size_width){
      this.airflow_max_limit = parseFloat(((1968*this.filter_case.case_size_height * this.filter_case.case_size_width)/(24*24)).toFixed(2));
    }else{
      this.airflow_max_limit = 1968;
    }
    this.airflow_max_limit = this.airflow_max_limit * this.num_filters;
    // this.system_information.total_airflow = this.airflow_max_limit;
  }

  _addNewAirflow(){
    this.variable_airflows.push(
      {'flow':null,'percent':null},
    )
    this.list_height = this.getRowHeight();
  }

  getRowHeight(){ 
    if(this.variable_airflows.length > 2){
      return (520 + (80 * (this.variable_airflows.length-2))).toString() + 'px';
    }else{
      return '520px' 
    }
  }

  removeAirflowValue(pos){
    this.variable_airflows.splice(pos,1);
    if(this.variable_airflows.length == 1){
      this.variable_airflows[0].percent = 100;
      this.total_airflow_error = false;
      this.max_airflow = this.variable_airflows[0].flow;
    }
    this.list_height = this.getRowHeight();
    this._calculateAvgAirflow();
    this.getMaxAirflow();
  }

  _calculateAvgAirflow(){
    let sum_airflow = this.variable_airflows.reduce((a, b) => +a + +b.percent, 0);
    // this.total_airflow_error = false;
    // this.negative_value_error = false;
    // this.airflow_value_error = false;
    if((sum_airflow != 100)){
      this.total_airflow_error = true;
    }else{
      this.total_airflow_error = false;
    }
    if((this.variable_airflows.filter(value => value.percent < 0 || value.percent == 0).length > 0)){
      this.negative_value_error = true;
    }else{
      this.negative_value_error = false;
    }
    if((this.variable_airflows.find(value => value.flow < 0 || value.flow == 0 || value.flow < (118 * this.num_filters) || value.flow > this.airflow_max_limit))){
      this.airflow_value_error = true;
    }else{
      this.airflow_value_error = false;
    }
    if(!(this.total_airflow_error) && !(this.negative_value_error) && !(this.airflow_value_error)){
      this.system_information.total_airflow = this.variable_airflows.reduce((a, b) => +a + +(b.flow * (b.percent/100)), 0).toFixed(2);
      this._updateAirflowPerFilter();
      this.getMaxAirflow();
    }
  }

  getMaxAirflow(){
    let values = [...this.variable_airflows]; 
    values.sort(function(a, b) {  return b.flow - a.flow; });
    this.max_airflow = values[0].flow;
  }

  getDpLimits() {
    if(this.reports && this.system_information.airflows_per_filter){
      var can_consult:boolean;
      if(this.system_information.airflows_per_filter.find(airflow => airflow.flow > 0 && airflow.percent > 0 && airflow.flow !== "0" && airflow.flow !== '')){
        can_consult = true;
      }

      if(this.system_information.variable_airflow == 1){
        this.system_information.airflows_per_filter = [{
          flow:this.system_information.airflow_per_filter,
          percent:100
        }];
        if(this.system_information.airflows_per_filter[0].flow == "0" || this.system_information.airflows_per_filter[0].flow == 0 || !this.system_information.airflows_per_filter[0].flow){
          can_consult = false
        }else{
          can_consult = true;
        }
      }else{
        if(this.variable_airflows.find(airflow => !airflow.flow || !airflow.percent || airflow.flow == 0 || airflow.percent == 0)){
          can_consult = false;
        }else{
          can_consult = true;
        }
      }
      var data = {
        "height": this.filter_case.case_size_height,
        "width": this.filter_case.case_size_width,
        "reports":this.reports.map(r => r.id),
        "is_variable":this.system_information.variable_airflow == "2" ? true : false, 
        "num_filters":this.num_filters,
        "airflows_per_filter":this.system_information.variable_airflow == 1 ? this.system_information.airflows_per_filter : this.variable_airflows,
      }
      if (can_consult) {
        this.caseProvider.getSystemDpValues(data).subscribe(
          (dp_values) => {
            this.dp_max = parseFloat(dp_values.max_dp.toFixed(2));
            this.dp_min = parseFloat(dp_values.min_dp_current.toFixed(2));
          }, error => {
        })
      }
    }

    // if (this.report && this.report.length > 0) {
    //   var data = {
    //     "height": this.filter_case.case_size_height,
    //     "width": this.filter_case.case_size_width,
    //     "reports": this.report.map(r => r.id),
    //     "airflows_per_filter": [{
    //       flow: this.system.airflow_per_filter,
    //       percent: 100
    //     }]
    //   }
      
    // }
  }

  getEditionDpLimits(){
    var data = {
      "height": this.system_information.height,
      "width": this.system_information.width,
      "reports":this.system_information.current_filter.reports.map(r => r.id),
      "is_variable":this.system_information.variable_airflow == "2" ? true : false, 
      "num_filters":this.system_information.filters_number,
      "airflows_per_filter":this.system_information.variable_airflow == 1 ? this.system_information.airflows_per_filter : this.variable_airflows,
    }
    
    this.caseProvider.getSystemDpValues(data).subscribe(
      (dp_values) => {
        this.dp_max = parseFloat(dp_values.max_dp.toFixed(2));
        this.dp_min = parseFloat(dp_values.min_dp_current.toFixed(2));
      }, error => {
    })
  }

  continue(){
    if(this.system_information.variable_airflow == 1){
      this.variable_airflows[0].flow = this.system_information.airflow_per_filter;
      this.variable_airflows[0].percent = 100;
    }
    this.system_information.airflows_per_filter = this.variable_airflows;
    this.set_data.emit(this.system_information);
    this.next();
  }

  emitData(){
    var input_error_1 = false;
    var input_error_2 = false;
    this.system_information.airflows_per_filter = this.variable_airflows;
    if((this.system_information.average_pressure_drop && this.system_information.average_pressure_drop !== '' && (this.system_information.average_pressure_drop > this.dp_max || this.dp_min > this.system_information.average_pressure_drop) || this.system_information.average_pressure_drop == 0)){
      input_error_1 = true;
    }
    if(this.system_information.maximum_operational_dp && this.system_information.maximum_operational_dp !== '' && (this.system_information.maximum_operational_dp > this.dp_max || this.dp_min > this.system_information.maximum_operational_dp) || this.system_information.maximum_operational_dp == 0){
      input_error_2 = true;
    }
    if(input_error_1 || input_error_2){
      this.input_has_error.emit(true);
    }else{
      this.input_has_error.emit(false);
    }
    this.set_data.emit(this.system_information);
  }

  next(){
    this.next_step.emit()
  }

}
