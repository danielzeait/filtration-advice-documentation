import { Component, OnInit, Inject } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';

@Component({
  selector: 'app-customize-dashboard',
  templateUrl: './customize-dashboard.component.html',
  styleUrls: ['./customize-dashboard.component.scss']
})
export class CustomizeDashboardComponent implements OnInit {
  public widgets:any[];
  constructor(public dialogRef: MatDialogRef<CustomizeDashboardComponent>, @Inject(MAT_DIALOG_DATA) public data: any) { }

  ngOnInit() {
    this.widgets = [
      { name:'User Overview', cols: 1, rows: 6, image:'../../../../assets/imgs/overview_widget.png', type:'overview',data:null,in_use:true},
      { name:'Favorite Cases', cols: 1, rows: 4, image:'../../../../assets/imgs/case_widget.png', type:'case',data:null,in_use:true},
      { name:'Merv Efficiency', cols: 3, rows: 8, image:'../../../../assets/imgs/merv_widget.png', type:'merv',data:null,in_use:false},
      { name:'Optimal Date of Change-out', cols: 2, rows: 4, image:'../../../../assets/imgs/changeout_widget.png', type:'changeout',data:null,in_use:false},
    ]
    if(this.data.length > 0){
      this.setCheckedWidgets();
    }
  }
  
  close(){
    this.dialogRef.close(null);
  }

  applyDashboardSettings(){
    this.dialogRef.close(this.widgets);
  }

  setCheckedWidgets(){
    this.data.find(widget => widget.type == 'overview' && widget.in_use) ? this.widgets[0].in_use = true : this.widgets[0].in_use = false;
    this.data.find(widget => widget.type == 'case' && widget.in_use) ? this.widgets[1].in_use = true : this.widgets[1].in_use = false;
    this.data.find(widget => widget.type == 'merv' && widget.in_use) ? this.widgets[2].in_use = true : this.widgets[2].in_use = false;
    this.data.find(widget => widget.type == 'changeout' && widget.in_use) ? this.widgets[3].in_use = true : this.widgets[3].in_use = false;
  }

}
