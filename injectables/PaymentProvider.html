import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';

@Injectable()
export class PaymentProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }


    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public getPaymentMethods():Observable<any>{
        localStorage.setItem('loader_message','Getting your payment methods.');
        return this.http.get<any>(global.API_PAYMENT_METHODS,this.getRequestHeaders()).pipe();
    }

    public createNewPaymentMethod(data):Observable<any>{
        localStorage.setItem('loader_message','Creating new payment method.');
        return this.http.post<any>(
            global.API_PAYMENT_METHODS,data,this.getRequestHeaders()).pipe();
    }

    public deletePaymentMethod(data):Observable<any>{
        let url = global.API_PAYMENT_METHOD_DELETE;
        url = url.replace('__payment_id__',data.id);
        return this.http.delete<any>(url,this.getRequestHeaders()).pipe();
    }

    public setDefaultPaymentMethod(data):Observable<any>{
        let url = global.API_SET_DEFAULT_PAYMENT;
        url = url.replace('__payment-method__',data.id);
        return this.http.post<any>(url,data,this.getRequestHeaders()).pipe();
    }

    public getAllSubscriptions():Observable<any>{
        localStorage.setItem('loader_message','Getting subscriptions list.');
        return this.http.get<Subscription>(global.API_SUBSCRIPTION_PLANS,this.getRequestHeaders()).pipe();
    }

    public getAllProducts():Observable<any>{
        localStorage.setItem('loader_message','Getting subscriptions list.');
        return this.http.get<Subscription>(global.API_PRODUCTS,this.getRequestHeaders()).pipe();
    }

    public getUserSubscriptions():Observable<any>{
        localStorage.setItem('loader_message','Getting your subscriptions list.');
        return this.http.get<Subscription>(global.API_USER_SUBSCRIPTIONS,this.getRequestHeaders()).pipe();
    }

    public getUserBillings():Observable<any>{
        localStorage.setItem('loader_message','Getting your billings list.');
        return this.http.get<any>(global.API_USER_BILLINGS,this.getRequestHeaders()).pipe();
    }

    public createNewSubscription(data):Observable<any>{
        localStorage.setItem('loader_message','Creating New Subscription.');
        return this.http.post<Subscription>(
            global.API_USER_SUBSCRIPTIONS,data,this.getRequestHeaders()).pipe();
    }

    public deleteUserSubscription(data):Observable<any>{
        let url = global.API_DELETE_USER_SUBSCRIPTIONS;
        url = url.replace('__plan_id__',data.id);
        return this.http.delete<Subscription>(url,this.getRequestHeaders()).pipe();
    }

    public getPaymentInvoice(data):Observable<any>{
        let url = global.API_USER_PAYMENT_INVOICE;
        url = url.replace('__invoice_id__',data);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public filterBillingList(billingList:Array<any>, value:string){
        return billingList.filter(_billing =>
            (_billing.meta_data.status.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_billing.meta_data.description.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_billing.amount.toString().toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_billing.created.toString().toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1))
    }

}
