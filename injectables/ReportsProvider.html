import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map, retry } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { TypeScriptEmitter } from '@angular/compiler';
import { UtilsFunctions } from './utils.functions';

@Injectable()
export class ReportsProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router, private utils:UtilsFunctions) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
            .set('Content-Type', 'application/json')
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }

    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public getReports():Observable<any>{
        localStorage.setItem('loader_message','Getting filter test reports.');
        return this.http.get<any>(global.API_GET_REPORTS_LIST,this.getRequestHeaders()).pipe();
    }

    public getReportDetails(id):Observable<any>{
        localStorage.setItem('loader_message','Getting report details.');
        let url = global.API_GET_REPORT_DETAILS;
        url = url.replace('__id_report__',id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public buyTestReport(id_report,payment_id):Observable<any>{
        localStorage.setItem('loader_message','Completing your lab test report purchase.');
        let url = global.API_BUY_REPORT;
        url = url.replace('__id_report__',id_report);
        url = url.replace('__payment_id__',payment_id);
        let data = {
            is_paid_with_credits:false
        }
        return this.http.post<any>(url,data,this.getRequestHeaders()).pipe();
    }

    public getReportPoints(id):Observable<any>{
        localStorage.setItem('loader_message','Getting report details.');
        let url = global.API_REPORT_POINTS;
        url = url.replace('__id_report__',id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getReportAirflowPoints(id):Observable<any>{
        localStorage.setItem('loader_message','Getting report details.');
        let url = global.API_REPORT_AIRFLOW_POINTS;
        url = url.replace('__id_report__',id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getUserReports():Observable<any>{
        localStorage.setItem('loader_message','Getting filter test reports.');
        return this.http.get<any>(global.API_GET_USER_REPORTS,this.getRequestHeaders()).pipe();
    }

    public updateReport(data){
        const report_data = Object.assign({}, data);
        report_data.model_filter = report_data.model_filter.id;
        report_data.laboratory = report_data.laboratory_id;
        if(data.standard.id){
            report_data.standard = report_data.standard.id;
        }
        let url = global.API_UPDATE_REPORT;
        url = url.replace('__id_report__',report_data.id);
        return this.http.patch(url,JSON.stringify(report_data),this.getRequestHeaders()).pipe();
    }

    public editFullReport(data,file,file_restricted){
        const report_data = {...data};
        if(data.model_filter.id){
            report_data.model_filter = report_data.model_filter.id;
        }
        report_data.laboratory = parseInt(report_data.laboratory_id);
        report_data.pdf_date = this.utils.formatDate(report_data.pdf_date) + ' 00:00:00';
        delete report_data.exponent_n;

        if(data.standard.id){
            report_data.standard = report_data.standard.id;
        }
        let token = this.getSessionToken();
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token);
        let options = {
            headers: requestHeaders
        }
        var formData = new FormData();
        if(file){
            formData.append('pdf_file', file);
        }else{
            delete report_data.pdf_file
        }
        if(file_restricted.name !== ''){
            formData.append('pdf_file_limited', file_restricted);
        }else{
            delete report_data.pdf_file_limited
        }
        if(!report_data.user){
            delete report_data.user
        }else{
            report_data.user = data.user.id
        }
        for ( var key in report_data ) {
            formData.append(key, report_data[key]);
        }
        localStorage.setItem('loader_message','Updating changes for lab test report.');
        let url = global.API_UPDATE_REPORT;
        url = url.replace('__id_report__',report_data.id);
        return this.http.patch(
            url,
            formData,
            options,
        ).pipe();
    }
    
    public rejectReport(report_data){
        let url = global.API_REJECT_REPORTS;
        url = url.replace('__id_report__',report_data.id);
        var data = {
            status:report_data.status
        }
        return this.http.put(url,JSON.stringify(data),this.getRequestHeaders()).pipe();
    }

    public deleteReport(report_data){
        let url = global.API_GET_REPORT_DETAILS;
        url = url.replace('__id_report__',report_data.id);
        return this.http.delete(url,this.getRequestHeaders()).pipe();
    }

    public setReportPoints(point){
        let url = global.API_SET_REPORT_POINTS;
        return this.http.post(url,JSON.stringify(point),this.getRequestHeaders()).pipe();
    }

    public editReportPoints(point){
        let url = global.API_EDIT_REPORT_POINTS;
        url = url.replace('__id_point__',point.id);
        return this.http.put(url,JSON.stringify(point),this.getRequestHeaders()).pipe();
    }

    public setReportAirflowPoints(point){
        let url = global.API_SET_REPORT_AIRFLOW_POINTS;
        return this.http.post(url,JSON.stringify(point),this.getRequestHeaders()).pipe();
    }

    public editReportAirflowPoints(point){
        let url = global.API_EDIT_REPORT_AIRFLOW_POINTS;
        url = url.replace('__id_point__',point.id);
        return this.http.put(url,JSON.stringify(point),this.getRequestHeaders()).pipe();
    }


    public sendPDF(file){
        var formData = new FormData();
        formData.append('pdf_file', file);
        localStorage.setItem('loader_message','Sending your PDF report.');
        return this.http.post(
            global.API_UPLOAD_PDF_REPORT,
            formData,
            this.getRequestHeaders(),
        ).pipe();
    }

    public createFullReport(data,file,file_restricted){
        const report_data = {...data};
        report_data.model_filter = report_data.model_filter.id;
        report_data.laboratory = parseInt(report_data.laboratory_id);
        report_data.pdf_date = this.utils.formatDate(report_data.pdf_date) + ' 00:00:00';

        if(data.standard.id){
            report_data.standard = report_data.standard.id;
        }
        let token = this.getSessionToken();
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token);
        let options = {
            headers: requestHeaders
        }
        var formData = new FormData();
        for ( var key in report_data ) {
            formData.append(key, report_data[key]);
        }
        formData.append('pdf_file', file);
        formData.append('pdf_file_limited', file_restricted);
        localStorage.setItem('loader_message','Creating new report.');
        return this.http.post(
            global.API_UPLOAD_PDF_REPORT,
            formData,
            options,
        ).pipe();
    }

    public sendPDFReport(request,file){
        let token = this.getSessionToken();
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        let options = {
            headers: requestHeaders
        }
        var formData = new FormData();
        if(request){
            for ( var key in request ) {
                formData.append(key, request[key]);
            }
        }
        formData.append('pdf_file', file);
        localStorage.setItem('loader_message','Sending your PDF report.');
        return this.http.post(
            global.API_UPLOAD_PDF_REPORT,
            formData,
            options
        ).pipe();
    }

    public getReportsByFilterName(reports_array:Array<any>,filter_name:string){
        var array = [];
        for(var i=0; i<reports_array.length;i++){
            if((reports_array[i].model_filter) && (reports_array[i].model_filter === filter_name)){
                array.push(reports_array[i])
            }
        }
        return array;
    }

    public downloadSingleReport(url):Observable<Blob>{
        const httpOptions = {
            headers: new HttpHeaders({
            'Content-Type': 'application/json'
            }),
            observe: 'response' as 'body',
            responseType: 'blob' as 'blob'
        };

        return this.http.get(url, httpOptions);
    }

    downloadPDF(url): Observable<any> {
        return this.http.get<Blob>(url, { responseType: 'blob' as 'json' }).pipe();
    };

    public extractReports(data){
        return new Promise((resolve) => {
            let extracted_files = [];
            let requestHeaders = new  HttpHeaders()
                .set('Accept', 'application/json')
            let options = {
                headers: requestHeaders
            }
            for (let index = 1; index < data.length; index++) {
                var fd = new FormData();
                fd.append('File', data[index]);
                fd.append('PageRange', '1-1');
                fd.append('StoreFile', 'true');
                this.http.post<any>(global.API_EXTRACT_REPORTS,fd,options).pipe().subscribe(
                    (report)=>{
                        this.downloadPDF(report.Files[0].Url).subscribe(
                            async (response) => {
                                var file = new File(
                                    [response],
                                    'extracted_report_' + index + '.pdf',
                                    { type: "aplication/pdf" }
                                );
                                extracted_files.push(file);
                                if(index == data.length - 1 ){
                                    resolve(extracted_files);
                                }
                            }
                        )
                    }
                );
            }
        })
    }

    public cutPageReports(file,i){
        return new Promise((resolve) => {
            let requestHeaders = new  HttpHeaders()
                .set('Accept', 'application/json')
            let options = {
                headers: requestHeaders
            }
            var fd = new FormData();
            fd.append('File', file);
            fd.append('PageRange', '1-1');
            this.http.post<any>(global.API_EXTRACT_REPORTS,fd,options).pipe().subscribe(
                (report)=>{
                    var file = new File(
                        [report.Files[0].FileData],
                        'extracted_report_' + i + '.pdf',
                        { type: "aplication/pdf" }
                    );
                    resolve(file);
                }
            );
        })
    }

    public concatReports(data){
        let requestHeaders = new  HttpHeaders()
        .set('Accept', 'application/json')
        let options = {
            headers: requestHeaders
        }
        var fd = new FormData();
        for (let index = 0; index < data.length; index++) {
            fd.append('FILES[' + index + ']', data[index]);
        }
        fd.append('StoreFile', 'true');
        return this.http.post<any>(
            global.API_MERGE_REPORTS,fd,options).pipe();
    }

    public getFrankCalculations(data,file){
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json');
        var formData = new FormData();
        formData.append('csv_file', file);
        for ( var key in data ) {
            formData.append(key, data[key]);
        }
        let options = {
            headers: requestHeaders
        }
        localStorage.setItem('loader_message','Getting chart information.');
        return this.http.post(
            'https://engine.turpial.dev/frank/',
            formData,
            options,
        ).pipe();
    }

}
