import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { Company } from '../models/company.model';
import { User } from '../models/user_data.model';

@Injectable()
export class SettingsProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }


    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public createUserSettings(file,data){
        localStorage.setItem('loader_message','Saving settings.');
        var formData = new FormData();
        for ( var key in data ) {
            formData.append(key, data[key]);
        }
        formData.append('logo', file, file.name);
        return this.http.post<any>(global.API_PDF_SETTINGS, formData, this.getRequestHeaders()).pipe();
    }

    public updateUserProfile(file,data){
        delete data.birth_date;
        delete data.gender;
        var formData = new FormData();
        for ( var key in data ) {
            formData.append(key, data[key]);
        }
        formData.append('profile_image', file, file.name);
        localStorage.setItem('loader_message','Saving your profile information.');
        return this.http.put<any>(global.API_GET_USER_INFO, formData, this.getRequestHeaders()).pipe();
    }

    public patchUserProfile(data:any):Observable<any>{
        var user_data = {...data};
        delete user_data.profile_image;
        delete user_data.birth_date;
        delete user_data.gender;
        localStorage.setItem('loader_message','Saving your profile information.');
        return this.http.patch<User>(
            global.API_GET_USER_INFO,user_data,this.getRequestHeaders()).pipe();
    }

    public patchUserProfileUnit(data:any):Observable<any>{
      //  localStorage.setItem('loader_message','Saving your profile information.');
        return this.http.patch<User>(
            global.API_GET_USER_INFO,data,this.getRequestHeaders()).pipe();
    }

    public updatePdfSettings(data:Company):Observable<any>{
        localStorage.setItem('loader_message','Saving your company information.');
        return this.http.patch<Company>(
            global.API_PDF_SETTINGS,data,this.getRequestHeaders()).pipe();
    }

    public createNewPdfSettings(data:Company):Observable<any>{
        localStorage.setItem('loader_message','Saving your company information.');
        return this.http.post<Company>(
            global.API_PDF_SETTINGS,data,this.getRequestHeaders()).pipe();
    }

    public getPdfSettings():Observable<any>{
        return this.http.get<Company>(global.API_PDF_SETTINGS,this.getRequestHeaders()).pipe();
    }

    public getPdfImages(id,comparison?):Observable<any>{
        let url = global.API_PDF_IMAGES;
        url = url.replace('__case_id__',id);
        if(comparison){
            url = url + '?comparison='+ comparison
        }
        return this.http.get<Company>(url,this.getRequestHeaders()).pipe();
    }

    public getDummyCompany(){
        let company_information = new Company('Turpial Development CA','RIF-123456-7','Daniel Zeait','daniel@turpialdev.com','+57 (311) 262 57 74','logo_url');
        return company_information;
    }

   
}
