import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';

@Injectable()
export class SharedReportProvider {
    header: HttpHeaders;
    options: any;
    token: any;
    requestHeaders: any;
    constructor(private http: HttpClient, private router: Router) {}

    public getRequestHeaders(){
      let token = this.getSessionToken();
      this.requestHeaders = new  HttpHeaders()
          .set('Accept', 'application/json')
          .set("Authorization","token "+ token)
      this.options = {
          headers: this.requestHeaders
      }
      return this.options;
  }


    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public createNewSharedReport(data): Observable<any>{
      localStorage.setItem('loader_message','Sharing report with ' + data.count + ' emails');
      let url = global.API_POST_SHARE_REPORTS;
      url = url.replace('_report_id_', data.id);
      return this.http.post<any>(url, data, this.getRequestHeaders()).pipe();
  }

    public getUserSharedReports(report_id): Observable<any>{
    let url = global.API_GET_USERS_SHARE_REPORTS;
    url = url.replace('_report_id_', report_id);
    return this.http.get<any>(url, this.getRequestHeaders()).pipe();
   }

    public deleteUserSharedReport(data){
      let token = this.getSessionToken();
      const options = {
          headers: new HttpHeaders({
            'Content-Type': 'application/json',
            'Authorization': 'token ' + token,
          }),
          body: {
            emails: data.email,
          },
      };

      let url = global.API_REVOKE_USERS_SHARE_REPORTS;
      url = url.replace('_report_id_', data.id);
      return this.http.delete<any>(url, options).pipe();
     }




}
