import { Injectable, HostListener, Output,EventEmitter } from '@angular/core';
import { table } from 'console';
import { CommonProvider } from './common.service';

@Injectable()
export class ResponsiveProvider {

  
  @Output() setContentWidth = new EventEmitter<any>();
  @Output() setContentMargin = new EventEmitter<any>();
  main_content_width: string;
  main_content_margin: string;
  menu_expanded:boolean = true;
  constructor(private commonProvider:CommonProvider) {
    this.getNumCols(window.innerWidth);
    this.commonProvider.menuIsExpanded.subscribe((menu_expanded)=>{
      this.menu_expanded = menu_expanded;
      this.getMainContentBaseWidh(this.menu_expanded);
    })
  }

  @HostListener('window:resize', ['$event'])
  onResize(event) {
    this.getMainContentBaseWidh(this.menu_expanded);
  }

  public getNumCols(width){
    let num_cols:number;
    if(width > 820 && width < 1240) {
        num_cols = 2
      }
      if(width < 820){
        num_cols  = 1
      }
      if(width < 1460 && width > 1240){
        num_cols  = 3
      }
      if(width > 1460){
        num_cols  = 4
      }
    return num_cols
  }

  public getNumColsHalf(width){
    let num_cols:number;
    if(width < 860){
      num_cols  = 1
    }else{
      num_cols  = 2
    }
    return num_cols
  }

  public getHeaderLogoSize(width){
    let size:string;
    if(width < 940){
      size  = '30%'
    }else{
      size = '40%'
    }
    return size
  }

  public getHeaderActionsSize(width){
    let size:string;
    if(width < 940){
      size  = '70%'
    }else{
      size = '60%'
    }
    return size
  }

  public getRowspan(width){
    let rowspan:number;
    if(width > 1090 && width < 1240) {
        rowspan = 9
      }
      if(width > 979 && width < 1089){
        rowspan  = 10
      }
      if(width < 1440 && width > 1239){
        rowspan  = 8
      }
      if(width > 1440 || width == 1440){
        rowspan  = 7
      }
      if(width < 980){
        rowspan  = 11
      }
    return rowspan
  }

    public getRowHeight(width){
      let height:string;
        if(width < 460){
          height  = '400px'
        }
        if(width > 460 && width < 512) {
          height = '280px'
        }
        if(width > 512 && width < 640) {
          height = '320px'
        }
        if(width > 640 && width < 860) {
          height = '360px'
        }
        if(width < 960 && width > 860){
          height  = '320px'
        }
        if(width < 1024 && width > 960){
          height  = '350px'
        }
        if(width < 1200 && width > 1024){
          height  = '320px'
        }
        if(width < 1340 && width > 1200){
          height  = '340px'
        }
        if(width > 1340){
          height  = '380px'
        }
      return height
  }

  public getScreenSize(width){
    let tablet_mode:boolean = false;
    width < 1025 ? tablet_mode = true : false;
    return tablet_mode;
  }

  public getMainContentBaseWidh(menu_expanded){
    this.main_content_width = '';
    this.main_content_margin = '';
    if(menu_expanded){
      if(window.outerWidth < 1025){
        this.main_content_width = '69%'
      }else{
        this.main_content_width = '77%';
      }
      this.main_content_margin = '250px';
    }else{
      if(window.outerWidth < 1025){
        this.main_content_width = '75%';
      }else{
        this.main_content_width = '80%';
      }
      this.main_content_margin = '190px';
    }
    this.setContentWidth.emit(this.main_content_width);
    this.setContentMargin.emit(this.main_content_margin);
  }

}
