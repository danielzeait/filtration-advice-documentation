import { Injectable } from '@angular/core';
import { MatSnackBar } from '@angular/material';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { global } from  '../global';
 
@Injectable()
  export class NotificationService {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private _snackBar: MatSnackBar,private http: HttpClient) {}
    
    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }
    
    showNotification(message:string,action:string,duration:number){
        this._snackBar.open(message, action, {
            duration: duration,
        });
    }

    getNotifications(){
        localStorage.setItem('hide_loader','true');
        return this.http.get<any>(global.API_GET_ALL_NOTIFICATIONS,this.getRequestHeaders()).pipe();
    }

    deleteNotification(notification){
        localStorage.setItem('loader_message','Deleting notification.');
        let url = global.API_NOTIFICATION;
        url = url.replace('__id__',notification.id);
        return this.http.delete<any>(url,this.getRequestHeaders()).pipe();
    }

    readNotification(notification){
        localStorage.setItem('hide_loader','true');
        let url = global.API_NOTIFICATION;
        url = url.replace('__id__',notification.id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }
  
}
