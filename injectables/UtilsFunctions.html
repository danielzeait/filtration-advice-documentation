import { Injectable } from '@angular/core';

@Injectable()
  export class UtilsFunctions {
    header:any;
    options:any;
    requestHeaders:any;
    constructor() {}

    transformAmount(element: any){
        element.target.value = parseFloat(element.target.value).toFixed(2);
        return element;
    }

    numberWithCommas(x) {
        var parts = x.toString().split(".");
        parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ",");
        return parts.join(".");
    }

    replaceNumber(element:any){
        element.target.value = '';
        element.target.placeholder='0.00';
        return element
    }

    substractDates(date2){
        const oneDay = 24 * 60 * 60 * 1000; // hours*minutes*seconds*milliseconds
        const firstDate:any = new Date();
        const secondDate:any = date2 ? new Date(date2) : new Date();
        const diffDays = Math.round(Math.abs((firstDate - secondDate) / oneDay));
        return diffDays;
    }

    getFutureDateByMonths(date,months){
        var d = new Date(date);
        var totalsecs = 3600*24*30;
        var month_to_sec = 2.628e+6 * months;
        d.setSeconds(d.getSeconds() + month_to_sec);
        return d;
    }

    formatDate(date) {
    if(date){
        var d = new Date(date),
        month = '' + (d.getMonth() + 1),
        day = '' + d.getDate(),
        year = d.getFullYear();

        if (month.length < 2)
            month = '0' + month;
        if (day.length < 2)
            day = '0' + day;

        return [year, month, day].join('-');
        }else{
            return null
        }
    }

    formatDateISO(date){
        var d = new Date(date);
        return d.toISOString()
    }

    formatDateChart(date) {
        if(date){
            var d = new Date(date),
            month = d.getMonth() + 1,
            day = '' + d.getDate();
            return month + '/' + day
        }
    }

    formatDateStringMonth(date) {
        const months = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dic"];
        var count = 'th';
        if(date){
            var d = new Date(date),
            month = months[d.getMonth()],
            day = '' + d.getDate(),
            year = d.getFullYear();
            if(day[day.length - 1] === '1'){
                count = 'st'
            }
            if(day[day.length - 1] === '2'){
                count = 'nd'
            }
            if(day[day.length - 1] === '3' && day[day.length - 1] != '13'){
                count = 'rd'
            }
            return month + ' ' + day + count + ', ' + year
        }
    }

    formatDateTime(date) {
        const months = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dic"];
        var count = 'th';
        if(date){
            var d = new Date(date),
            month = months[d.getMonth()],
            day = '' + d.getDate(),
            year = d.getFullYear();
            var hours = d.getHours();
            hours = (hours+24-2)%24; 
            var mid='am';
            if(hours==0){ //At 00 hours we need to show 12 am
                hours=12;
            }
            else if(hours>12){
                hours=hours%12;
                mid='pm';
            }
            var min = d.getMinutes().toString();
            if (parseInt(min) < 10){
                min = '0' + min;
            }
            if(day[day.length - 1] === '1' && day !== '11'){
                count = 'st'
            }
            if(day[day.length - 1] === '2'){
                count = 'nd'
            }
            if(day[day.length - 1] === '3' && day[day.length - 1] != '13'){
                count = 'rd'
            }
            return month + ' ' + day + count + ', ' + year + ' - ' + hours+':'+min+' '+ mid; 
        }
    }

    formatDateTimeInstrument(date) {
        const months = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dic"];
        var count = 'th';
        if(date){
            var 
                mid='am', 
                d = new Date(date),
                month = months[d.getMonth()],
                day = '' + d.getDate(),
                year = d.getFullYear(),
                server_time = date.split('T')[1].split(':'),
                display_hour = server_time[0] + ':' + server_time[1];
            
            if(parseInt(server_time[0]) > 12){
                mid='pm';
            }

            if(day[day.length - 1] === '1' && day !== '11'){
                count = 'st'
            }
            if(day[day.length - 1] === '2'){
                count = 'nd'
            }
            if(day[day.length - 1] === '3' && day[day.length - 1] != '13'){
                count = 'rd'
            }
            return month + ' ' + day + count + ', ' + year + ' - ' + display_hour ; 
        }
    }

    numberThousandSeparator(number) {
        if(number){
            return number.toLocaleString(undefined, {maximumFractionDigits:2})
        }
    }

    isNumber(n) { return !isNaN(parseFloat(n)) && !isNaN(n - 0) }

    formatStringTitle(string){
        string = string.replace('_', ' ');
        string = string[0].toUpperCase() + string.slice(1);
        return string
    }
}
