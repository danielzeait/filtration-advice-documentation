import { Injectable } from '@angular/core';
import { AuthProvider } from 'src/app/providers/auth.service';
import { CustomErrorHandler } from 'src/app/providers/error.service';


@Injectable()
export class SupplierUnitsProvider {
    public lbs;
    public inc;
    public units;
    public cm;
    public user_data:any={};
    personal_data;


    constructor(    
        private handlerError:CustomErrorHandler, 
        private authProvider:AuthProvider
        ) {

    }

   async getUnitsActive() {
      /*  if(localStorage.getItem('metrics_units')){
            return this.units= 1;
           } else {
            return this.units= 0;
           }*/
     this.user_data = await this.authProvider.getUserInfo().toPromise();
     this.units = (this.user_data.unit_system == 'm') ? 1 : 0;
     return this.units;
    }

    async getUnitsK() {
    this.user_data = await this.authProvider.getUserInfo().toPromise();
    this.lbs = (this.user_data.unit_system == 'm') ? 'kgr' : 'lbs';
    return this.lbs;
    
   }

    async getUnitsI() {
    this.user_data = await this.authProvider.getUserInfo().toPromise();
    this.inc = (this.user_data.unit_system == 'm') ? 'mm' : 'inch';
    return this.inc;
    }

    async getUnitsc() {
    this.user_data = await this.authProvider.getUserInfo().toPromise();
    this.cm = (this.user_data.unit_system == 'm') ? 'pa' : 'cm';
    return this.cm;
   }






    

   


}
