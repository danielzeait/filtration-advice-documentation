import { Injectable } from '@angular/core';
import { Router, NavigationExtras } from '@angular/router';
import { SystemFilter } from '../models/system-filter.model';
import { System } from '../models/system.model';
import { Report } from '../models/report.model';
import { Location } from '@angular/common';



@Injectable()
export class NavigationProvider {

    constructor(private route: Router, private router:Router, private location:Location) {

    }

    _land(){
        this.router.navigate(["/landing"],{replaceUrl:true});
        this.location.replaceState('');
    }

    _goLogin(){
        this.router.navigate(["/login"],{replaceUrl:true});
    }
    _goLoginSonicu(){
      this.router.navigate(["/loginsonicu"],{replaceUrl:true});
    }

    _goProduct(){
        this.router.navigate(["/product"],{replaceUrl:true});
    }

    _goFaqs(){
        this.router.navigate(["/faqs"],{replaceUrl:true});
    }

    _goContact(){
        this.router.navigate(["/contact"],{replaceUrl:true});
    }

    _contactUs(){
        this.router.navigate(["/contact-us"]);
    }

    _goNewReport(){
        this.router.navigate(["/upload-report"],{replaceUrl:true});
    }

    _goHomeValidator(){
        this.router.navigate(["/validator"],{replaceUrl:true});
    }

    _goRegister(){
        this.router.navigate(["/register"],{replaceUrl:true});
    }

    _goRecoverPassword(){
        this.router.navigate(["/recover-password"],{replaceUrl:true});
    }

    _viewInvoice(bill){
        const url = this.router.serializeUrl(this.router.createUrlTree(['/invoice',bill.id]));
        window.open(url, '_blank');
    }

    _goNewPassword(token){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "token": token
            }
        };
        this.router.navigate(["/new-password"],navigationExtras);
    }

    _goTermsAndConditions(){
        this.router.navigate(["/t&c"],{replaceUrl:true});
    }

    _goHome(){
        this.router.navigate(["/home"],{replaceUrl:true});
    }

    _confirmEmail(token){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "token": token
            }
        };
        this.router.navigate(["/landing"],navigationExtras);
    }

    _goOptimizationsResults(data){
        window.scrollTo(0,0);
        this.router.navigate(["/optimization-results",data.alter_autoincrement_id]);
    }

    _goCaseSnapshot(case_id,timestamp){
        window.scrollTo(0,0);
        this.router.navigate(["/optimization-results",case_id,timestamp]);
    }

    _goFilterDetailsId(filter_id){
        window.scrollTo(0,0);
        this.router.navigate(["/filter-list",filter_id]);
    }

    _goNewInstrumentation(){
        window.scrollTo(0,0);
        this.router.navigate(["/new-instrumentation"]);
    }

    _goReportReview(data){
        window.scrollTo(0,0);
        if(data){
            this.router.navigate(["/report-review",data.id]);
        }else{
            this.router.navigate(["/report-review"]);
        }
    }

    _goComparisonResults(data){
        localStorage.setItem('comparison_data',JSON.stringify(data));
        localStorage.removeItem('case_to_compare');
        localStorage.removeItem('comparison_reports');
        localStorage.removeItem('cases');
        localStorage.removeItem('proposed_filter');
        this.router.navigate(["/comparison-results"],{replaceUrl:true});
    }

    _goCheckOut(pay_object){
        localStorage.setItem('subscription',JSON.stringify(pay_object));
        this.router.navigate(["/payment-checkout"]);
    }

    _goHomeCases(filter, report?, norm?){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "filter": JSON.stringify(filter),
              "report": JSON.stringify(report),
              "norm":   norm
            }
        };
        this.router.navigate(["/case-list"],navigationExtras);
    }

    _goNewCaseWithFilter(filter, report?, norm?){
        // let navigationExtras: NavigationExtras = {
        //     queryParams: {
        //         "filter":true,
        //     }
        // };
        // report ?  localStorage.setItem('report_for_case',JSON.stringify(report)) 
        //         : localStorage.removeItem('filter_for_case') 
        localStorage.setItem('filter_for_case',JSON.stringify(filter));
        this.router.navigate(["/new-case"],{replaceUrl:true});
    }

    _goNewCase(study_case:System){
        if(study_case){
            let navigationExtras: NavigationExtras = {
                queryParams: {
                  "edit_case": true,
                }
            };
            localStorage.setItem('case_to_edit', JSON.stringify(study_case))
            this.router.navigate(["/new-case"],navigationExtras);
        }else{
            let navigationExtras: NavigationExtras = {
                queryParams: {
                  "new_case": true,
                  "study_case":null
                }
            };
            this.router.navigate(["/new-case"],navigationExtras);
        }
    }

    // _goEditComparison(study_case,comparison){

    //     let navigationExtras: NavigationExtras = {
    //         queryParams: {
    //           "new_comparison": true,
    //           "edit": true
    //         }
    //     };
    //     localStorage.setItem('study_case',JSON.stringify(study_case));
    //     localStorage.setItem('comparison',JSON.stringify(comparison));
    //     this.router.navigate(["/comparisons-list"],navigationExtras);

    // }

    _goEditComparison(study_case,comparison){
        localStorage.setItem('case_to_compare',JSON.stringify(study_case));
        localStorage.setItem('proposed_filter',JSON.stringify(comparison));
        this.router.navigate(["/new-comparison"],{replaceUrl:true});
    }

    _goNewComparisonClear(){
        this.router.navigate(["/new-comparison"],{replaceUrl:true});
    }

    _goNewComparisonCaseSelected(study_case){
        localStorage.setItem('case_to_compare',JSON.stringify(study_case));
        this.router.navigate(["/new-comparison"],{replaceUrl:true});
    }

    _goToEditCase(study_case){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "edit": true,
              "study_case":JSON.stringify(study_case)
            }
        };
        this.router.navigate(["/case-list"],navigationExtras);
    }

    _uploadNewReport(){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "new_report": true
            }
        };
        this.router.navigate(["/filter-list"],navigationExtras);
    }

    _goHomeCasesClear(){
        this.router.navigate(["/case-list"],{replaceUrl:true , skipLocationChange:true});
    }

    _goHomeComparisons(){
        this.router.navigate(["/comparisons-list"],{replaceUrl:true});
    }

    _goNewComparison(){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "new_comparison": true
            }
        };
        this.router.navigate(["/comparisons-list"],navigationExtras);
    }   

    _goFilterList(){
        this.router.navigate(["/filter-list"]);
    }

    _goInstrumentList(){
      this.router.navigate(["/instrument-list"]);
    }

    _goProfile(){
        this.router.navigate(['/profile']);
    }

    _goFilterDetails(filter){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "filter": filter
            }
          };
        this.router.navigate(["/filter-details"],navigationExtras);
    }

    _goFilterListSelecect(filter){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "filter": filter.id,
            }
          };
        this.router.navigate(["/filter-list"],navigationExtras);
    }

    _goPdfPreview(data:System,chart_data:any){
        this.router.navigate(["/optimization-pdf"], {
            queryParams:{
                'pdf_data':JSON.stringify(data),
                'chart_data':JSON.stringify(chart_data)
            },
            skipLocationChange:true
        });
    }

    _goRegisterConfirmation(email){
        let navigationExtras: NavigationExtras = {
            queryParams: {
              "email": email
            }
          };
        this.router.navigate(['/confirm-register'],navigationExtras);
    }

    _goNotificationInbox(){
        this.router.navigate(['/notifications'],{replaceUrl:true})
    }


}
