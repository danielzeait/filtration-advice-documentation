import { Injectable, Output, EventEmitter } from '@angular/core';
import { CustomSnackBarComponent } from '../components/custom-snackbar/custom-snack-bar.component';
import { MatSnackBar } from '@angular/material';
import { BehaviorSubject } from 'rxjs';
import { emit } from 'process';

@Injectable()
export class CommonProvider {
    @Output() userLogin = new EventEmitter<any>();
    @Output() updateDustVsDpChart = new EventEmitter<any>();
    @Output() updateAirflowVsDpChart = new EventEmitter<any>();
    @Output() userValidator = new EventEmitter<any>();
    @Output() updateUserCredits = new EventEmitter<any>();
    @Output() updateNotificationList = new EventEmitter<any>();
    @Output() menuIsExpanded = new EventEmitter<any>();
    @Output() updateSnapshots = new EventEmitter<any>();
    @Output() updatePaginator = new EventEmitter<any>();
    @Output() updateUserInfo = new EventEmitter<any>();
    @Output() updatePlanList = new EventEmitter<any>();
    @Output() updateVersion = new EventEmitter<any>();

    private setdataEmmit = new BehaviorSubject('initial info');
    public getdataEmmit$ = this.setdataEmmit.asObservable();
    public system_version:any;

    user_dashboard:any;
    inovice:any;
    constructor(private _snackBar:MatSnackBar) {}

    userDidLogin(user: any) {
        this.userLogin.emit(user);
    }

    pointHasChanged(points: any) {
        this.updateDustVsDpChart.emit(points);
    }

    pointsHasChanged(points: any) {
        this.updateAirflowVsDpChart.emit(points);
    }

    caseHasFinished() {
        this.updateSnapshots.emit();
    }

    updateUserData(data) {
        this.updateUserInfo.emit(data);
    }

    showSnackBar(title,message,type){
        this._snackBar.openFromComponent( CustomSnackBarComponent, {
            data:{
              title:title,
              message:message,
            },
            panelClass:[type],
            horizontalPosition:'start',
            duration: 10000,
        });
    }

    setInvoiceData(invoice){
        this.inovice = invoice;
    }

    getInvoice(){
        return this.inovice;
    }

    sendData(data: any) {
        this.setdataEmmit.next(data)
    }

    setUserValidator(is_validator){
        this.userValidator.emit(is_validator);
    }

    updateNotifications(notifications){
        this.updateNotificationList.emit(notifications);
    }

    updatePlans(){
        this.updatePlanList.emit();
    }

    refreshCredits(){
        this.updateUserCredits.emit()
    }

    preservePaginator(){
        this.updatePaginator.emit()
    }

    menuExpand(expanded){
        this.menuIsExpanded.emit(expanded);
    }

    setSystemVersion(version){
        this.updateVersion.emit(version);
    }

    getSystemVersion(){
        return this.system_version;
    }

  
}
