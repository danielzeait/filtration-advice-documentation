import { Injectable } from '@angular/core';
import { System } from '../models/system.model';
import { UtilsFunctions } from './utils.functions';
import { Company } from '../models/company.model';
import { LoaderProvider } from './loader.service';
import { MetaDataProvider } from './metadata.service';
import { PdfConstructorProvider } from './pdf-contstructor.provider';

import pdfMake from 'pdfmake/build/pdfmake';
import pdfFonts from 'pdfmake/build/vfs_fonts';

import { global } from  '../global';

pdfMake.vfs = pdfFonts.pdfMake.vfs;

@Injectable()
export class PdfCreator {
    pdf_image: any = '';
    append_reports: any = false;
    company:Company;
    pdf_images: any = {};
    constructor(
        private metaData:MetaDataProvider,
        private loaderService:LoaderProvider, 
        private pdfConstructor:PdfConstructorProvider,
        private utils:UtilsFunctions) { 
        
    }

    async getDocumentDefinition(
                system: System, 
                cost_vs_time_chart_image, 
                dp_vs_time_chart_image,
                table_image,
                selected_data,
                provider_data,
                is_company,
                doc_images,
                conditions,
                concat_reports?,
                max_operational_point?,
                current_limit_point?,
                display_selected?,
                monitoriredImages?,
                snapshot?,
                warnings?,
                type?,
                pdf_options?
            ) 
        {
        var that = this;
        let docDefinition = {
            header: function(currentPage, pageCount, pageSize) {
                if(currentPage == 1){
                    return [
                        {
                            columns: 
                                [
                                    { text: 'Print Date: ' + that.utils.formatDateStringMonth(new Date()) , alignment: 'left', fontSize:7,color:'#333',marginLeft:20,marginTop:7.5},
                                    // { text: 'This report was produced by', alignment: 'left', fontSize:7,color:'#333', marginLeft:20,marginTop:7.5},
                                ]
                        },
                        {
                        style:'marginBottomSmall',
                        canvas: [{ lineColor:'#ccc', type: 'line', x1:20, y1: 5, x2: 575, y2: 5, lineWidth: 0.85}],
                        }, 
                    ]
                }else{
                    return [
                        {
                            columns: 
                                [
                                    { width:'auto',text: system.name, fontSize:7,color:'#333',marginRight:20, marginLeft:20,marginTop:15},
                                    { width:'auto',text: 'Case #' + system.alter_autoincrement_id, alignment: 'right', fontSize:7,color:'#333',marginRight:20,marginTop:15},
                                    { width:'auto',text: conditions ? 'Client: ' + conditions.name : '',  fontSize:7,color:'#333',marginTop:15,marginRight:20},
                                    system.monitoring_module ? { width:'auto',text: 'Estimation Date: ' + snapshot, alignment: 'left', fontSize:7,color:'#333',marginRight:20,marginTop:15} : 
                                                               { width:'auto',text: '' ,marginRight:20,marginTop:15},
                                    { width:'auto',text: 'Print Date: ' + that.utils.formatDateStringMonth(new Date()) , fontSize:7,color:'#333',marginRight:20, marginTop:15},
                                    {
                                        image:'data:image/png;base64,'+ global.APP_FILTRATION_REGISTERED,
                                        width:100,
                                        height:10,
                                        absolutePosition: {x:470, y:12},
                                    },
                                ]
                        },
                        {
                        style:'marginBottomSmall',
                        canvas: [{ lineColor:'#ccc', type: 'line', x1:20, y1: 5, x2: 575, y2: 5, lineWidth: 0.85}],
                        }, 
                    ]
                }
            },
            
            footer: function(currentPage, pageCount)  {
                return [
                    {
                        canvas: [{ lineColor:'#53BDF0', type: 'line', x1: 17.5, y1: 5, x2: 575, y2: 5, lineWidth: 0.85}],
                    },
                    {
                        style:'marginBottom',
                        columns: [
                            { text: 'Patents: US 9,546,952 B2 - US 10,527,537 B2', alignment: 'left', fontSize:7,color:'#333', marginLeft:20,marginTop:7.5},
                            { text: 'Page ' + currentPage.toString() + ' of ' + pageCount, alignment: 'right', fontSize:7,color:currentPage == 1 ? '#fff' : '#333',marginRight:20,marginTop:7.5},
                        ]
                    }
                ]
            },

            pageMargins: [20, 40, 20, 30],
            
            permissions: {
                printing: 'highResolution',
                modifying: true,
                copying: false,
                annotating: true,
                fillingForms: true,
                contentAccessibility: true,
                documentAssembly: true
            },

            info: {
                title: system.name,
            },
            
            content: [

                this.pdfConstructor.getIdenfiyerRow(system,doc_images,pdf_options),

                this.pdfConstructor.getPdfCover(system,doc_images,selected_data,max_operational_point),
                
                { text: '', pageBreak: "after"},

                {text:'Total Cost of Ownership',style:['blue','bold']},
        
                this.pdfConstructor.getTCOTable(table_image),
                
                { text: this.metaData.getPdfPremises(),alignment: 'left', fontSize:6,color:'#999', marginLeft:0,marginTop:25},
                
                { text: this.metaData.getPdfDisclaimer(),alignment: 'left', fontSize:6,color:'#999', marginLeft:0,marginTop:15},
                
                !system.monitoring_module || pdf_options.include_generated_data ? { text: '', pageBreak: "after"} : '',

                pdf_options.include_generated_data ? {text:'Collected and Generated Data (Since Installation of Current Filter)',style:['blue','bold']} : '',
        
                pdf_options.include_generated_data ? this.pdfConstructor.getOptimalCharts(cost_vs_time_chart_image,dp_vs_time_chart_image,selected_data,display_selected,system.current_filter.name) : '',

                pdf_options.include_generated_data ? this.pdfConstructor.getDpVsTimeCharts(dp_vs_time_chart_image,warnings,type) : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text: '', pageBreak: "after",} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text:'Data Meassured by Sensors',style:['blue','bold']} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? this.pdfConstructor.getEnergyCharts(monitoriredImages) : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text: '', pageBreak: "after",} : '',
                
                (system.monitoring_module && pdf_options.include_data_sensors) ? {text:'Data Meassured by Sensors',style:['blue','bold']} : '',
                
                (system.monitoring_module && pdf_options.include_data_sensors) ? this.pdfConstructor.getHistograms(monitoriredImages) : '',
                
                (system.monitoring_module && pdf_options.include_data_sensors) ? {text: '', pageBreak: "after",} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? this.pdfConstructor.getClassTables(system) : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text: '', pageBreak: "after",} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text:'Data Meassured by Sensors',style:['blue','bold']} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? {text:'',marginBottom:10} : '',

                (system.monitoring_module && pdf_options.include_data_sensors) ? this.pdfConstructor.getMonitoredAirflow(system,current_limit_point) : '',
                
                pdf_options.include_generated_data ? { text: '', pageBreak: "after",} : '',

                pdf_options.include_input_data ? {text:'Input Data',style:['blue','bold']} : '',

                {text:'',marginBottom:10},

                pdf_options.include_input_data ? this.pdfConstructor.getSystemInputData(system) : '',

                system.monitoring_module && pdf_options.include_data_sensors ? this.pdfConstructor.getSystemInstrumentData(system) : '',

                system.monitoring_module && pdf_options.include_data_sensors ? '' :  system.variable_airflow ? this.pdfConstructor.getVariableAirflowTable(system) : 
                pdf_options.include_input_data ? this.pdfConstructor.getConstantAirflowTable(system) : '',

                system.monitoring_module && pdf_options.include_data_sensors ? '' : pdf_options.include_input_data ? this.pdfConstructor.getSystemOperatingTimeTable(system) : '',

                pdf_options.include_input_data ? this.pdfConstructor.getSystemEfficienciesTable(system) : '',

                pdf_options.include_input_data ? { text: '', pageBreak: "after"} : '',

                pdf_options.include_filter_details ? {text:'Current Filter Information',style:['blue','bold']} : '',

                {text:'',marginBottom:10},

                pdf_options.include_filter_details ? this.pdfConstructor.getFilterTable(system,doc_images,false,null) : '',
        
                //this.pdfConstructor.getCostTable(system,false,null),

                pdf_options.include_filter_details ? { text: '', pageBreak: "after"} : '',

                {text:'Filter Efficiency Information',style:['blue','bold'], marginBottom:10},
        
                this.pdfConstructor.getMERVTable(system,null),

                pdf_options.include_comments ? this.pdfConstructor.getComments(pdf_options.comments) : '',

                pdf_options.include_filters_channels ? { text: '', pageBreak: "after"} : '',

                pdf_options.include_filters_channels ? {text:'Filter Sales Channels',style:['blue','bold']} : '',

                pdf_options.include_filters_channels ? {text:'',marginBottom:10} : '',
        
                pdf_options.include_filters_channels ? this.pdfConstructor.getSellersTable(system.current_filter.model_filter.references) : '',
                
            ],

            styles: this.pdfConstructor.getPdfStyles(), 
        }

        const pdfDocGenerator = pdfMake.createPdf(docDefinition);
        if (concat_reports) {
            this.pdfConstructor.appendReportsPdf(system, pdfDocGenerator);
            this.loaderService.leaveLoaders();
        } else {
            pdfDocGenerator.open();
            this.loaderService.leaveLoaders();
        }
    }

}
