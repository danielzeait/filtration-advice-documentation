import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { Filter } from '../models/filter.model';

@Injectable()
export class FilterProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }


    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public getAllPublicFilters():Observable<any>{
        localStorage.setItem('loader_message','Getting Filter List.');
        return this.http.get<Filter>(global.API_GET_ALL_FILTERS,this.getRequestHeaders()).pipe();
    }

    public getLandingFilters():Observable<any>{
        localStorage.setItem('hide_loader','true');
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
        let options = {
            headers: requestHeaders
        }
        return this.http.get<Filter>(global.API_GET_PUBLIC_FILTERS,options).pipe();
    }

    public createFilters(data,file):Observable<any>{
        var manufacturer:any = [];
        manufacturer.push(data.manufacturer.id);
        var formData = new FormData();
        for ( var key in data ) {
            if(key == 'manufacturer'){
                formData.append('manufacturer',manufacturer);
            }else if(key == 'picture'){
                formData.append('picture',file);
            }else{
                formData.append(key, data[key]);
            }
        }
        formData.append('model_filter_image', file, file.name);
        localStorage.setItem('loader_message','Creating New Filter Model.');
        return this.http.post<Filter>(global.API_GET_ALL_FILTERS,formData,this.getRequestHeaders()).pipe();
    }

    public requestFilter(data):Observable<any>{
        localStorage.setItem('loader_message','Requesting Filter Test Report.');
        return this.http.post<any>(global.API_REQUEST_FILTER,data,this.getRequestHeaders()).pipe();
    }

    public getFilterDetails(id):Observable<any>{
        localStorage.setItem('loader_message','Getting Filter Details.');
        let url = global.API_GET_FILTER_DETAILS;
        url = url.replace('__filter_id__',id);
        return this.http.get<Filter>(url,this.getRequestHeaders()).pipe();
    }

    public getPersonalFilters():Observable<any>{
        localStorage.setItem('loader_message','Getting Filter List.');
        return this.http.get<Filter>(global.API_GET_FILTERS,this.getRequestHeaders()).pipe();
    }

    public filterFilterList(filterList:Array<Filter>, search_value:string){
        if(filterList){
            return filterList.filter( _filter =>
                (_filter.name.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1));
        }
    }

    public filterFilterListHard(filterList:Array<Filter>, search_value:string){
        if(filterList){
            return filterList.filter( _filter =>
                (_filter.name.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1) ||
                (_filter.frame_material.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1) ||
                (_filter.frame_type.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1) ||
                (_filter.type.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1) || 
                (_filter.manufacturer.name.toLowerCase().indexOf(search_value.toLocaleLowerCase()) !== -1));
        }
    }

    public searchMatchingFilters(filter_data){
        let url = global.API_GET_ALL_FILTERS + `?frame_type=__frame_type__&frame_material=__frame_material__&media_type=__media_type__&filter_type=__filter_type__&depth_min=__depth_min__&depth_max=__depth_max__&dust_type=__dust_type__&merv_rating_min=__merv_rating_min__&merv_rating_max=__merv_rating_max__&standard=__standard__`;
        url = url.replace('__filter_type__',filter_data.filter_type)
        url = url.replace('__frame_type__',filter_data.filter_frame_type)
        url = url.replace('__frame_material__',filter_data.filter_frame_material)
        url = url.replace('__media_type__',filter_data.filter_media_material);
        url = url.replace('__depth_max__',filter_data.depth_max)
        url = url.replace('__depth_min__',filter_data.depth_min);
        url = url.replace('__dust_type__',filter_data.filter_dust_type)
        url = url.replace('__merv_rating_min__',filter_data.filter_merv_rating);
        url = url.replace('__merv_rating_max__',filter_data.filter_merv_rating);
        url = url.replace('__merv_rating_max__',filter_data.filter_merv_rating);
        url = url.replace('__standard__',filter_data.standard.id);

        if(filter_data.incinarable){
            url = url + '&incinerable=true'
        }
        if(filter_data.filter_face_screen){
            url = url + '&face_screen=true'   
        }

        return this.http.get<Filter>(url,this.getRequestHeaders()).pipe();
    }

    public searchFilters(data,part_url):Observable<any>{
        let url = global.API_SEARCH_FILTERS;
        if(part_url.length > 1){
            part_url = part_url + "&"
        }
        url = url.replace('__part_url__',part_url);
        data.name !== "" ?  url = url.replace('__name__',data.name) : url = url.replace('name=__name__&','');
        data.part_number !== "" ?url = url.replace('__part_number__',data.part_number) : url = url.replace('part_number=__part_number__&','');
        data.depth_min !== "" ? url = url.replace('__depth_min__',data.depth_min) : url = url.replace('depth_min=__depth_min__&','');
        data.depth_max !== "" ? url = url.replace('__depth_max__',data.depth_max) : url = url.replace('depth_max=__depth_max__&','');
        data.width_min !== "" ? url = url.replace('__width_min__',data.width_min) : url = url.replace('width_min=__width_min__&','');
        data.width_max !== "" ? url = url.replace('__width_max__',data.width_max) : url = url.replace('width_max=__width_max__&','');
        data.height_min !== "" ? url = url.replace('__height_min__',data.height_min) : url = url.replace('height_min=__height_min__&','');
        data.height_max !== "" ? url = url.replace('__height_max__',data.height_max) : url = url.replace('height_max=__height_max__&','');
        data.merv_rating_min !== "" ? url = url.replace('__merv_rating_min__',data.merv_rating_min) : url = url.replace('merv_rating_min=__merv_rating_min__&','');
        data.merv_rating_max !== "" ? url = url.replace('__merv_rating_max__',data.merv_rating_max) : url = url.replace('merv_rating_max=__merv_rating_max__&','');
        data.max_press_drop_min !== "" ? url = url.replace('__max_press_drop_min__',data.max_press_drop_min) : url = url.replace('max_press_drop_min=__max_press_drop_min__&','');
        data.max_press_drop_max !== "" ? url = url.replace('__max_press_drop_max__',data.max_press_drop_max) : url = url.replace('max_press_drop_max=__max_press_drop_max__&','');
        data.efficiency_max !== "" ? url = url.replace('__efficiency_max__',data.efficiency_max) : url = url.replace('efficiency_max=__efficiency_max__&','');
        data.efficiency_min !== "" ? url = url.replace('__efficiency_min__',data.efficiency_min) : url = url.replace('efficiency_min=__efficiency_min__&','');
        data.test_airflow_max !== "" ? url = url.replace('__test_airflow_max__',data.test_airflow_max) : url = url.replace('test_airflow_max=__test_airflow_max__&','');
        data.test_airflow_min !== "" ? url = url.replace('__test_airflow_min__',data.test_airflow_min) : url = url.replace('test_airflow_min=__test_airflow_min__&','');
        data.pdf_date_max !== "" ? url = url.replace('__pdf_date_max__',data.pdf_date_max) : url = url.replace('pdf_date_max=__pdf_date_max__&','');
        data.pdf_date_min !== "" ? url = url.replace('__pdf_date_min__',data.pdf_date_min) : url = url.replace('pdf_date_min=__pdf_date_min__&','');
        data.e1_max !== "" ? url = url.replace('__e1_max__',data.e1_max) : url = url.replace('e1_max=__e1_max__&','');
        data.e1_min !== "" ? url = url.replace('__e1_min__',data.e1_min) : url = url.replace('e1_min=__e1_min__&','');
        data.e2_max !== "" ? url = url.replace('__e2_max__',data.e2_max) : url = url.replace('e2_max=__e2_max__&','');
        data.e2_min !== "" ? url = url.replace('__e2_min__',data.e2_min) : url = url.replace('e2_min=__e2_min__&','');
        data.e3_max !== "" ? url = url.replace('__e3_max__',data.e3_max) : url = url.replace('e3_max=__e3_max__&','');
        data.e3_min !== "" ? url = url.replace('__e3_min__',data.e3_min) : url = url.replace('e3_min=__e3_min__','');
        if(url[url.length-1] == "&"){
            url = url.substring(0, url.length - 1);
        }
        return this.http.get<Filter>(url,this.getRequestHeaders()).pipe();
    }

    registFilterReference(id){
        let url = global.API_FILTER_REFERENCES;
        url  = url.replace('__id__',id);
        return this.http.get(url,this.getRequestHeaders()).pipe();
    }
}
