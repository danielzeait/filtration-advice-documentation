import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map, min } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { System } from '../models/system.model';
import { SystemFilter } from '../models/system-filter.model';

@Injectable()
export class CaseProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router) {}
    
    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }

    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public getPersonalCases():Observable<any>{
        localStorage.setItem('loader_message','Loading your cases');
        return this.http.get<System>(global.API_CASES,this.getRequestHeaders()).pipe();
    }

    public getPersonalComparisonCases():Observable<any>{
        let url = global.API_CASES;
        url = url.replace('ordering=-updated','ordering=-comparison,-updated')
        localStorage.setItem('loader_message','Loading your cases');
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();
    }

    public reOrderCases(param):Observable<any>{
        let url = global.API_CASES;
        url = url.replace('-updated',param);
        localStorage.setItem('loader_message','Loading your cases');
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();
    }

    public getCaseDetails(id,snapshot?):Observable<any>{
        let url = global.API_GET_CASE_BY_ID;
        url = url.replace('__case_id__',id);
        if(snapshot){
            url = url + 'snapshots/' + snapshot;
        }
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();    
    }

    public getCaseInstruments(id):Observable<any>{
        let url = global.API_GET_CASE_INSTRUMENTS;
        url = url.replace('__case_id__',id);
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();    
    }

    public getCasesByQtty(qtty):Observable<any>{
        let url = global.API_CASES;
        url = url  + '?page_size=' + qtty;
        localStorage.setItem('loader_message','Loading your cases');
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();
    }
    
    public getCasesByUrl(url):Observable<any>{
        localStorage.setItem('loader_message','Loading your cases');
        return this.http.get<System>(url,this.getRequestHeaders()).pipe();
    }
    
    public deleteStudyCase(case_id):Observable<any>{
        let url = global.API_DELETE_CASE;
        url = url.replace('__case_id__',case_id);
        return this.http.delete<any>(url,this.getRequestHeaders()).pipe();
    }

    public getEnergyCosts(data):Observable<any>{
        let url = global.API_GET_ENERGY_COSTS;
        url = url.replace('__zipcode__',data.company_zip_code);
        url = url.replace('__industry__',data.industry_type);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getDpValues(data):Observable<any>{
        let url = global.API_GET_DP_VALUES;
        url = url.replace('__report__',data.id);
        localStorage.setItem('hide_loader','true');
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getSystemDpValues(data):Observable<any>{
        let min_dp_data = {...data};
        if(min_dp_data.is_variable){
            let airflows = [];
            min_dp_data.airflows_per_filter.forEach(element => {
                airflows.push({
                    flow: element.flow / min_dp_data.num_filters,
                    percent:element.percent,
                }) 
            });
            min_dp_data.airflows_per_filter = [...airflows];
        }
        localStorage.setItem('hide_loader','true');
        return this.http.post<any>(global.API_GET_LIMITS_DP_VALUES,min_dp_data,this.getRequestHeaders()).pipe();
    }

    public createNewCase(data:any,is_monitored?):Observable<any>{
        localStorage.setItem('loader_message','Creating new case');
        let url = global.API_PAY_AND_OPTIMIZE;
        if(!data.current_filter.is_paid_with_credits && !is_monitored){
            url = url + '?payment-method=' + data.payment_id.toString();
        }
        return this.http.post<System>(
            global.API_PAY_AND_OPTIMIZE,data,this.getRequestHeaders()).pipe();
    }

    public updateCase(data:any):Observable<any>{
        localStorage.setItem('loader_message','Saving changes for your case');
        let url = global.API_EDIT_CASE;
        url = url.replace('__case_id__',data.alter_autoincrement_id.toString());
        return this.http.put<any>(
            url,data,this.getRequestHeaders()).pipe();
    }

    public patchCase(data:any):Observable<any>{
        localStorage.setItem('loader_message','Saving changes for your case');
        let url = global.API_EDIT_CASE;
        url = url.replace('__case_id__',data.alter_autoincrement_id.toString());
        return this.http.patch<any>(
            url,data,this.getRequestHeaders()).pipe();
    }

    public payAndUpdateCase(data:any):Observable<any>{
        localStorage.setItem('loader_message','Updating your case');
        let url = global.API_PAY_AND_EDIT_CASE;
        url = url.replace('__case_id__',data.alter_autoincrement_id.toString());
        if(!data.current_filter.is_paid_with_credits){
            url = url + '?payment-method=' + data.payment_id.toString();
        }
        return this.http.patch<System>(
            url,data,this.getRequestHeaders()).pipe();
    }

    public getCase(case_id):Observable<any>{
        let url = global.API_GET_CASE_BY_ID;
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    
    public getCaseById(case_id):Observable<any>{
        let url = global.API_CASES + '&&ids=' + case_id;
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }
    
    public getCaseDpVsTimeChart(case_id,snapshot?):Observable<any>{
        var url = '';
        if(snapshot){
            url = global.API_GET_SNAPSHOTS_CHART;
            url = url.replace('__timestamp__',snapshot);
        }else{
            url = global.API_CASE_OPTIMAL_CHART;
        }
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }
    
    public getCaseAirflowChart(case_id,snapshot?):Observable<any>{
        var url = ''
        if(snapshot){
            url = global.API_GET_SNAPSHOTS_FLOW_CHART;
            url = url.replace('__timestamp__',snapshot);
        }else{
            url = global.API_CASE_AIRFLOW_HISTOGRAM;
        }
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getCaseCostVsTimeChart(case_id,snapshot?):Observable<any>{
        var url = '';
        if(snapshot){
            url = global.API_GET_SNAPSHOTS_COST_CHART;
            url = url.replace('__timestamp__',snapshot);
        }else{
            url = global.API_GET_CASE_CHART_COST;
        }
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getCaseCostVsTimeChartByTime(case_id,time,time_propsed_selected?):Observable<any>{
        let url = global.API_CASE_OPTIMAL_CHART;
        url = url.replace('__case_id__',case_id);
        url = url + '?current-selected-time=' + time.change_out_time 
        if(time_propsed_selected){
            url = url + '&proposed-selected-time=' + time_propsed_selected;
        }
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getOperationChart(case_id):Observable<any>{
        let url = global.API_CASE_OPERATION_CHARTS;
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getCostsByTime(id,time,proposed?,snapshot?):Observable<any>{
        let url = '';
        if(proposed){
            url = global.API_GET_COST_BY_TIME_PROPOSED_FILTER;
            url = url.replace('__filter_id__',id);
        }else{
            if(snapshot){
                url = global.API_GEST_COST_BY_TIME_SNAPSHOT;
                url = url.replace('__snapshot__',snapshot);
            }else{
                url = global.API_GET_COST_BY_TIME;
            }
            url = url.replace('__case_id__',id);
        }
        url = url.replace('__given_time__',time);
        localStorage.setItem('hide_loader','true');
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getChangeOutList(cases):Observable<any>{
        let url = global.API_GET_CHANGEOUT_LIST;
        url = url + '?ids=' + cases; 
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public filterCaseList(caseList:Array<System>, value:string){
        return caseList.filter(_case =>
            (_case.name.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_case.model_filter_name.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_case.client_name.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) ||
            (_case.created_display.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1) || 
            (_case.updated.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1)) 
    }

    public getCaseSnapshots(case_id){
        let url = global.API_GET_SNAPSHOTS;
        url = url.replace('__case_id__',case_id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getCaseByDate(case_id,timestamp){
        let url = global.API_GET_SNAPSHOTS;
        url = url.replace('__case_id__',case_id);
        url = url + '?timestamp=' + timestamp;
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }


    public getCaseStatus(case_id){
        let url = global.API_GET_CASE_STATUS;
        url = url.replace('__case_id__',case_id);
        localStorage.setItem('hide_loader','true');
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

}
