import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';

@Injectable()
export class ComparisonProvider {
    header:HttpHeaders;
    options:any;
    token:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        this.requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        this.options = {
            headers: this.requestHeaders
        }
        return this.options;
    }


    public getSessionToken(){
        return localStorage.getItem('auth_token');
        //return "3eca9549f3dc568b6f462feadc41f66a6f65d43f"
    }

    public getPersonalCases():Observable<any>{
        return this.http.get<any>(global.API_CASES,this.getRequestHeaders()).pipe();
    }

    public getCaseComparisons(id):Observable<any>{
        let url = global.API_CASE_COMPARISONS;
        url = url.replace('__case_id__',id);
        url = url.replace('&&payment-method=__payment_id__','');
        localStorage.setItem('loader_message','Getting comparisons for your cases');

        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public orderCaseComparisons(id,param):Observable<any>{
        let url = global.API_CASE_COMPARISONS;
        url = url.replace('__case_id__',id);
        url = url.replace('-updated',param);
        localStorage.setItem('loader_message','Ordering comparisons for your case');

        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public deleteComparison(comparison_id):Observable<any>{
        let url = global.API_GET_COMPARISON_RESULT;
        url = url.replace('__filter_id__',comparison_id);
        return this.http.delete<any>(url,this.getRequestHeaders()).pipe();
    }

    public updateComparison(data):Observable<any>{
        localStorage.setItem('loader_message','Saving changes for your comparison');
        let url = global.API_GET_COMPARISON_RESULT;
        url = url.replace('__filter_id__',data.id);
        url = url.replace('__payment_id__',data.payment_id);
        return this.http.put<any>(
            url,data,this.getRequestHeaders()).pipe();
    }

    public createNewComparison(data):Observable<any>{
        let url = global.API_CASE_COMPARISONS;
        url = url.replace('__case_id__',data.alter_autoincrement_id);
        if(data.proposed_filter.is_paid_with_credits){
            url = url.replace('__payment_id__',null);
        }else{
            url = url.replace('__payment_id__',data.payment_id);    
        }
        localStorage.setItem('loader_message','Creating new comparison');
        return this.http.post<any>(url,data.proposed_filter,this.getRequestHeaders()).pipe();
    }

    public getComparisonChartDpVsTime(data):Observable<any>{
        let url = global.API_COMPARISON_OPTIMAL_CHART;
        url = url.replace('__filter_id__',data.comparison_data.id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public getDpVsTimeProposedSelected(case_id,time,time_propsed?):Observable<any>{
        let url = global.API_CASE_OPTIMAL_CHART_SELECTED;
        url = url.replace('__case_id__',case_id);
        url = url + '?current-selected-time=' + time.change_out_time;
        if(time_propsed){
            url = url + '&proposed-selected-time=' + time_propsed;
        }
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }
    public getComparisonChartCostVsTime(data):Observable<any>{
        let url = global.API_GET_COMPARISON_CHART_COST;
        url = url.replace('__filter_id__',data.comparison_data.id);
        return this.http.get<any>(url,this.getRequestHeaders()).pipe();
    }

    public filterComparisonList(comparisonList:Array<any>, value:string){
        return comparisonList.filter(_comparison =>
            (_comparison.name.toLowerCase().indexOf(value.toLocaleLowerCase()) !== -1)) 
    }

    
}
