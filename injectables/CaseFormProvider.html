import { Injectable } from '@angular/core';
import {FormControl, Validators, FormBuilder, FormGroup} from '@angular/forms';

@Injectable()
export class CaseFormProvider {
    form : any;
    min_dp_allowed: any;
    max_dp_allowed: any;
    constructor(private fb: FormBuilder,) {}

    public newCaseForm(){
        let caseForm = this.fb.group({
            username: new FormControl('',[Validators.required]),
            email: new FormControl('', Validators.compose([
              Validators.required,
              Validators.email
            ])),
            password: new FormControl('', [Validators.required,Validators.minLength(8)]),
            confirm_password: new FormControl('', [Validators.required,Validators.minLength(8)]),
          });
        return caseForm
    }

    public caseForm(){
        var phonePattern = "^[+]?[\s./0-9]*[(]?[0-9]{1,4}[)]?[-\s./0-9]*$";
        var zipCodePattern = "^[0-9]{4,7}$";

        this.form = this.fb.group({        
            airflow_per_filter: new FormControl('',[Validators.required, Validators.min(118)]),               
            carbon_footprint: new FormControl('',[Validators.required,Validators.min(0)]),
            client_address: new FormControl('',[Validators.required]),
            client_case_name: new FormControl('',[Validators.required]),
            company_name : new FormControl(''),
            client_contact_name: new FormControl(''),
            client_email: new FormControl('',[Validators.email]),
            client_phone: new FormControl('',[Validators.pattern(phonePattern)]),
            client_user_case: new FormControl(''),
            client_zip_code:  new FormControl('',[Validators.pattern(zipCodePattern)]),
            days_per_month: new FormControl('',[Validators.min(1),Validators.max(30)]),
            disposal_cost: new FormControl('',[Validators.min(0)]),
            drive_efficiency: new FormControl(90,[Validators.min(0),Validators.max(100)]),
            energy_cost: new FormControl('',[Validators.required,Validators.min(0)]),
            fan_efficiency: new FormControl(80,[Validators.min(0),Validators.max(100)]),
            filter_cost: new FormControl('',[Validators.required, Validators.min(0)]),
            filters_number: new FormControl('',[Validators.required]),
            freight_cost: new FormControl('',[Validators.min(0)]),
            has_resport: new FormControl(false,[Validators.required]),
            height: new FormControl('',[Validators.required,Validators.min(1),Validators.max(24)]),
            hours_per_day: new FormControl('',[Validators.min(1),Validators.max(24)]),
            hours_per_year:  new FormControl({value:'',disabled:true},[Validators.required, Validators.min(0)]),
            industry_type:  new FormControl('',[Validators.required]),
            labor_cost: new FormControl('',[Validators.min(0)]),
            length: new FormControl({value:'',disabled:true},[Validators.required,Validators.min(1),Validators.max(24)]),
            min_dp: new FormControl(),
            max_dp: new FormControl(),
            maximum_operational_dp: new FormControl(''),
            months_per_year: new FormControl('',[Validators.min(1),Validators.max(12)]),
            motor_efficiency: new FormControl(90,[Validators.min(0),Validators.max(100)]),
            pressure_drop:new FormControl('',[Validators.required, Validators.min(0)]),
            replacement_cycle: new FormControl('',[Validators.required]),
            total_airflow: new FormControl({value:'',disabled:true},[Validators.required]),
            variable_airflow: new FormControl(''),
            width: new FormControl('',[Validators.required,Validators.min(1),Validators.max(24)]),
        },{
            validator:[this.validTotalAirflow,this.validTotalHours,this.validDropPressure,this.validMaxAirflow, this.existTestReport]
        })

        return this.form
    }

    get min_dp() {
        return this.form.get('min_dp') as FormControl;
    }

    get max_dp() {
        return this.form.get('max_dp') as FormControl;
    }

    get pressure_drop() {
        return this.form.get('pressure_drop') as FormControl;
    }

    existTestReport(group : FormGroup){
        let test_report = group.controls.has_resport.value;
        return  test_report ? null : { caseWithoutReport: true }
    }

    validMaxAirflow(group : FormGroup){
        let airflow_per_filter = parseFloat(group.controls.airflow_per_filter.value);
        let max_airflow = parseFloat(((1968*group.controls.height.value * group.controls.width.value)/(24*24)).toFixed(2));
        return airflow_per_filter < max_airflow || airflow_per_filter == max_airflow ? null : { invalidMaxAirflow: true }
    }

    validDropPressure(group:FormGroup){
        let min_dp = group.controls.min_dp.value;
        let max_dp = group.controls.max_dp.value;
        let pressure_drop = group.controls.pressure_drop.value;
        return (pressure_drop < max_dp && pressure_drop > min_dp) || pressure_drop == max_dp || pressure_drop == min_dp ? null : { invalidDropPressure: true } 
    }

    validTotalAirflow(group:FormGroup){
        let airflow_per_filter = parseFloat(group.controls.airflow_per_filter.value);
        let filters_number = group.controls.filters_number.value;
        let total_airflow = parseFloat(group.controls.total_airflow.value);
        return total_airflow === airflow_per_filter * filters_number ? null : { invalidTotalAirflow: true }
    }

    validTotalHours(group:FormGroup){
        let hours = group.controls.hours_per_day.value;
        let days = group.controls.days_per_month.value;
        let months = group.controls.months_per_year.value;
        let hours_per_year = group.controls.hours_per_year.value;
        let total_hours = hours * days * months;
        return hours_per_year === total_hours ? null : { invalidTotalHours: true }
    }
}
