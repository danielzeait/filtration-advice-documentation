import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { global } from  '../global';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { User } from '../models/user_data.model';
import { CommonProvider } from './common.service';

@Injectable()
export class AuthProvider {
    header:any;
    options:any;
    requestHeaders:any;
    constructor(private http: HttpClient, private router:Router, private commonService:CommonProvider) {}

    public getRequestHeaders(){
        let token = this.getSessionToken();
        if (token){
            this.requestHeaders = new  HttpHeaders()
                .set('Accept', 'application/json')
                .set("Authorization","token "+ token)
            this.options = {
                headers: this.requestHeaders
            }
        }else{
            this.requestHeaders = new  HttpHeaders()
                .set('Accept', 'application/json');
            this.options = {
                headers: this.requestHeaders
            }
        }
        return this.options;
    }

    public userLogin(data):Observable<any>{
        let requestHeaders = new  HttpHeaders().set('Accept', 'application/json');
        let options = {
            headers: requestHeaders
        }
        localStorage.setItem('loader_message','Loggin in. Please wait');
        return this.http.post<any>(global.API_USER_LOGIN,data,options).pipe();
    }
        
    public userLogut(){
        var token = this.getSessionToken();
        var requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ token)
        
        var options = {
            headers: requestHeaders
        }
        return this.http.post<any>(
            global.API_USER_LOGOUT,
            "",
            options
        ).pipe();;
    }

    public registerUser(data):Observable<any>{
        localStorage.setItem('loader_message','Creating new user. Please wait');
        return this.http.post<any>(
            global.API_REGISTER_USER,data,this.getRequestHeaders()).pipe();
    }

    public validateUserEmail(token):Observable<any>{
        let url = global.API_VALIDATE_EMAIL;
        localStorage.setItem('loader_message','Validating yor email');
        url = url.replace('__token__',token);
        return this.http.post<any>(url,token,this.getRequestHeaders()).pipe();
    }

    public registerUserConfirmation(data):Observable<any>{

        if(data !== ''){
            let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
            .set("Authorization","token "+ data);
            let register_options = {
                headers: requestHeaders
            }
            return this.http.post<any>(
                global.API_REGISTER_USER_CONFIRMATION,data,register_options).pipe();
        }else{
            return this.http.post<any>(
                global.API_REGISTER_USER_CONFIRMATION,data,this.getRequestHeaders()).pipe();
        }
        //data.token
    }

    public requestResetPassword(data){
        return this.http.post<any>(
            global.API_REQUEST_RECOVER_PASSWORD,data,this.getRequestHeaders()).pipe();
    }

    public resetUserPassword(data):Observable<any>{
        let url = global.API_RESET_PASSWORD;
        url = url.replace('__token__',data.token);
        return this.http.post<any>(url,data,this.getRequestHeaders()).pipe();
    }

    public destroySession(){
        localStorage.clear();
        this.commonService.setUserValidator(false);
        this.router.navigate(['/landing'],{replaceUrl:true});
    }

    public getSessionToken(){
        return localStorage.getItem('auth_token');
    }

    public getUserInfo():Observable<any>{
        localStorage.setItem('loader_message','Getting personal info.');
        return this.http.get<User>(global.API_GET_USER_INFO,this.getRequestHeaders()).pipe();
    }

    public getUserDashboard(){
        localStorage.setItem('loader_message','Getting dashboard configuration.');
        return this.http.get<any>(global.API_GET_USER_DASHBOARD,this.getRequestHeaders()).pipe();
    }

    public getUserCredits(){
        localStorage.setItem('hide_loader','true');
        return this.http.get<any>(global.API_GET_USER_CREDITS,this.getRequestHeaders()).pipe();
    }

    public setUserDashboard(data){
        localStorage.setItem('hide_loader','true');
        return this.http.post<any>(global.API_GET_USER_DASHBOARD,data,this.getRequestHeaders()).pipe();
    }

    public sendUserMessage(data){
        localStorage.setItem('hide_loader','true');
        let requestHeaders = new  HttpHeaders()
            .set('Accept', 'application/json')
        let options = {
            headers: requestHeaders
        }
        return this.http.post<any>(global.API_SEND_CONTACT_MESSAGE,data,options).pipe();
    }

    public sendUserReview(data){
        localStorage.setItem('hide_loader','true');
        return this.http.post<any>(global.API_SEND_USER_REVIEW,data,this.getRequestHeaders()).pipe();
    }

    public sendUserInstrumentHelp(data){
        localStorage.setItem('hide_loader','true');
        return this.http.post<any>(global.API_SEND_USER_INSTRUMENT_HELP,data,this.getRequestHeaders()).pipe();
    }

    public getUsers(){
        localStorage.setItem('hide_loader','true');
        return this.http.get<User>(global.API_GET_ALL_USERS,this.getRequestHeaders()).pipe();
    }

}
