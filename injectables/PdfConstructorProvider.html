    public getIdenfiyerRow(system:System,doc_images,pdf_options){
        return {   
	        columns:[
	            pdf_options.include_client_data ? this.getPdfClient(pdf_options.client_data ? pdf_options.client_data : null) : '',
                pdf_options.include_company_info ? this.getPdfCompany(system,doc_images,pdf_options) : this.getPersonalInfo(system), 
	        ],
	    }
    }

    public getPdfClient(client){
        return [
            client.name || client.email || client.phone || client.address ? 
            {
                text:'Created for:'
            }:'',
            client.name || client.email || client.phone || client.address ?{
                canvas: 
                    [{ lineColor:'#ccc', type: 'line', x1:0, y1: 5, x2: 240, y2: 5, lineWidth: 1.5}],
            } : '',
            client.name || client.email || client.phone || client.address ? {   
                marginTop:10,
                columns:[
                    client.client_image ? {
                        width:70,
                        height:72,
                        image:client.client_image,                        
                    }:
                    {
                        width:0,
                        text:''
                        //image:client.image ? client.image:'',                        
                    } 
                    ,
                    [
                        {text:client.name ? client.name : '',fontSize:10,marginLeft:15},
                        {text:client.phone ? client.phone : '',fontSize:10,marginLeft:15},
                        {text:client.email ? client.email + '\n\n' : '',fontSize:10,marginLeft:15},
                        {text:client.address ? 'Client address' + '\n' : '',fontSize:10,marginLeft:15},
                        {text:client.address ? client.address : '',fontSize:10,marginLeft:15}

    }

    appendReportsPdf(system: System, pdfGenerator) {
        this.report_files = [];
        pdfGenerator.getBlob(async (blob_pdf_results) => {
            var new_file = new File(
                [blob_pdf_results],
                system.name + ".pdf",
                { type: 'application/pdf' }
            );
            this.report_files.push(new_file);
            await this.getPdfFileFromUrl(system.current_filter.reports);
            var print_files = [];
            print_files.push(new_file);
            let extracted_files = await this.reportProvider.extractReports(this.report_files);
            print_files = print_files.concat(extracted_files);
            this.reportProvider.concatReports(print_files).subscribe(
            (response) => {
                window.location.href = response.Files[0].Url;
            }, error => {
                this.handlerError.handleError(error);
            })
        });
    }

    getPdfFileFromUrl(files : Array<any>) {
        return new Promise((resolve) => {
            for (let index = 0; index < files.length; index++) {
                this.reportProvider.downloadPDF(files[index].pdf_file).subscribe(
                    async (response) => {
                    var file = new File(
                        [response],
                        'test_report_' + index + '.pdf',
                        { type: "aplication/pdf" }
                    );
                    this.report_files.push(file);
                    if(index == files.length - 1){
                        resolve(true);
                    }
                })
            }
        })
    }
